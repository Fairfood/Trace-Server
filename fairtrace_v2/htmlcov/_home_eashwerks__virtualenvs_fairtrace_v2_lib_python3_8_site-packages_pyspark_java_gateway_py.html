35def launch_gateway(conf=None, popen_kwargs=None): 

36 """ 

37 launch jvm gateway 

38 

39 Parameters 

40 ---------- 

41 conf : :py:class:`pyspark.SparkConf` 

42 spark configuration passed to spark-submit 

43 popen_kwargs : dict 

44 Dictionary of kwargs to pass to Popen when spawning 

45 the py4j JVM. This is a developer feature intended for use in 

46 customizing how pyspark interacts with the py4j JVM (e.g., capturing 

47 stdout/stderr). 

48 

49 Returns 

50 ------- 

51 ClientServer or JavaGateway 

52 """ 

53 if "PYSPARK_GATEWAY_PORT" in os.environ: 

54 gateway_port = int(os.environ["PYSPARK_GATEWAY_PORT"]) 

55 gateway_secret = os.environ["PYSPARK_GATEWAY_SECRET"] 

56 # Process already exists 

57 proc = None 

58 else: 

59 SPARK_HOME = _find_spark_home() 

60 # Launch the Py4j gateway using Spark's run command so that we pick up the 

61 # proper classpath and settings from spark-env.sh 

62 on_windows = platform.system() == "Windows" 

63 script = "./bin/spark-submit.cmd" if on_windows else "./bin/spark-submit" 

64 command = [os.path.join(SPARK_HOME, script)] 

65 if conf: 

66 for k, v in conf.getAll(): 

67 command += ["--conf", "%s=%s" % (k, v)] 

68 submit_args = os.environ.get("PYSPARK_SUBMIT_ARGS", "pyspark-shell") 

69 if os.environ.get("SPARK_TESTING"): 

70 submit_args = " ".join(["--conf spark.ui.enabled=false", submit_args]) 

71 command = command + shlex.split(submit_args) 

72 

73 # Create a temporary directory where the gateway server should write the connection 

74 # information. 

75 conn_info_dir = tempfile.mkdtemp() 

76 try: 

77 fd, conn_info_file = tempfile.mkstemp(dir=conn_info_dir) 

78 os.close(fd) 

79 os.unlink(conn_info_file) 

80 

81 env = dict(os.environ) 

82 env["_PYSPARK_DRIVER_CONN_INFO_PATH"] = conn_info_file 

83 

84 # Launch the Java gateway. 

85 popen_kwargs = {} if popen_kwargs is None else popen_kwargs 

86 # We open a pipe to stdin so that the Java gateway can die when the pipe is broken 

87 popen_kwargs["stdin"] = PIPE 

88 # We always set the necessary environment variables. 

89 popen_kwargs["env"] = env 

90 if not on_windows: 

91 # Don't send ctrl-c / SIGINT to the Java gateway: 

92 def preexec_func(): 

93 signal.signal(signal.SIGINT, signal.SIG_IGN) 

94 

95 popen_kwargs["preexec_fn"] = preexec_func 

96 proc = Popen(command, **popen_kwargs) 

97 else: 

98 # preexec_fn not supported on Windows 

99 proc = Popen(command, **popen_kwargs) 

100 

101 # Wait for the file to appear, or for the process to exit, whichever happens first. 

102 while not proc.poll() and not os.path.isfile(conn_info_file): 

103 time.sleep(0.1) 

104 

105 if not os.path.isfile(conn_info_file): 

106 raise RuntimeError("Java gateway process exited before sending its port number") 

107 

108 with open(conn_info_file, "rb") as info: 

109 gateway_port = read_int(info) 

110 gateway_secret = UTF8Deserializer().loads(info) 

111 finally: 

112 shutil.rmtree(conn_info_dir) 

113 

114 # In Windows, ensure the Java child processes do not linger after Python has exited. 

115 # In UNIX-based systems, the child process can kill itself on broken pipe (i.e. when 

116 # the parent process' stdin sends an EOF). In Windows, however, this is not possible 

117 # because java.lang.Process reads directly from the parent process' stdin, contending 

118 # with any opportunity to read an EOF from the parent. Note that this is only best 

119 # effort and will not take effect if the python process is violently terminated. 

120 if on_windows: 

121 # In Windows, the child process here is "spark-submit.cmd", not the JVM itself 

122 # (because the UNIX "exec" command is not available). This means we cannot simply 

123 # call proc.kill(), which kills only the "spark-submit.cmd" process but not the 

124 # JVMs. Instead, we use "taskkill" with the tree-kill option "/t" to terminate all 

125 # child processes in the tree (http://technet.microsoft.com/en-us/library/bb491009.aspx) 

126 def killChild(): 

127 Popen(["cmd", "/c", "taskkill", "/f", "/t", "/pid", str(proc.pid)]) 

128 

129 atexit.register(killChild) 

130 

131 # Connect to the gateway (or client server to pin the thread between JVM and Python) 

132 if os.environ.get("PYSPARK_PIN_THREAD", "true").lower() == "true": 

133 gateway = ClientServer( 

134 java_parameters=JavaParameters( 

135 port=gateway_port, auth_token=gateway_secret, auto_convert=True 

136 ), 

137 python_parameters=PythonParameters(port=0, eager_load=False), 

138 ) 

139 else: 

140 gateway = JavaGateway( 

141 gateway_parameters=GatewayParameters( 

142 port=gateway_port, auth_token=gateway_secret, auto_convert=True 

143 ) 

144 ) 

145 

146 # Store a reference to the Popen object for use by the caller (e.g., in reading stdout/stderr) 

147 gateway.proc = proc 

148 

149 # Import the classes used by PySpark 

150 java_import(gateway.jvm, "org.apache.spark.SparkConf") 

151 java_import(gateway.jvm, "org.apache.spark.api.java.*") 

152 java_import(gateway.jvm, "org.apache.spark.api.python.*") 

153 java_import(gateway.jvm, "org.apache.spark.ml.python.*") 

154 java_import(gateway.jvm, "org.apache.spark.mllib.api.python.*") 

155 java_import(gateway.jvm, "org.apache.spark.resource.*") 

156 # TODO(davies): move into sql 

157 java_import(gateway.jvm, "org.apache.spark.sql.*") 

158 java_import(gateway.jvm, "org.apache.spark.sql.api.python.*") 

159 java_import(gateway.jvm, "org.apache.spark.sql.hive.*") 

160 java_import(gateway.jvm, "scala.Tuple2") 

161 

162 return gateway

165def _do_server_auth(conn, auth_secret): 

166 """ 

167 Performs the authentication protocol defined by the SocketAuthHelper class on the given 

168 file-like object 'conn'. 

169 """ 

170 write_with_length(auth_secret.encode("utf-8"), conn) 

171 conn.flush() 

172 reply = UTF8Deserializer().loads(conn) 

173 if reply != "ok": 

174 conn.close() 

175 raise RuntimeError("Unexpected reply from iterator server.")

178def local_connect_and_auth(port, auth_secret): 

179 """ 

180 Connect to local host, authenticate with it, and return a (sockfile,sock) for that connection. 

181 Handles IPV4 & IPV6, does some error handling. 

182 

183 Parameters 

184 ---------- 

185 port : str or int or None 

186 auth_secret : str 

187 

188 Returns 

189 ------- 

190 tuple 

191 with (sockfile, sock) 

192 """ 

193 sock = None 

194 errors = [] 

195 # Support for both IPv4 and IPv6. 

196 addr = "127.0.0.1" 

197 if os.environ.get("SPARK_PREFER_IPV6", "false").lower() == "true": 

198 addr = "::1" 

199 for res in socket.getaddrinfo(addr, port, socket.AF_UNSPEC, socket.SOCK_STREAM): 

200 af, socktype, proto, _, sa = res 

201 try: 

202 sock = socket.socket(af, socktype, proto) 

203 sock.settimeout(int(os.environ.get("SPARK_AUTH_SOCKET_TIMEOUT", 15))) 

204 sock.connect(sa) 

205 sockfile = sock.makefile("rwb", int(os.environ.get("SPARK_BUFFER_SIZE", 65536))) 

206 _do_server_auth(sockfile, auth_secret) 

207 return (sockfile, sock) 

208 except socket.error as e: 

209 emsg = str(e) 

210 errors.append("tried to connect to %s, but an error occurred: %s" % (sa, emsg)) 

211 sock.close() 

212 sock = None 

213 raise RuntimeError("could not open socket: %s" % errors)

216def ensure_callback_server_started(gw): 

217 """ 

218 Start callback server if not already started. The callback server is needed if the Java 

219 driver process needs to callback into the Python driver process to execute Python code. 

220 """ 

221 

222 # getattr will fallback to JVM, so we cannot test by hasattr() 

223 if "_callback_server" not in gw.__dict__ or gw._callback_server is None: 

224 gw.callback_server_parameters.eager_load = True 

225 gw.callback_server_parameters.daemonize = True 

226 gw.callback_server_parameters.daemonize_connections = True 

227 gw.callback_server_parameters.port = 0 

228 gw.start_callback_server(gw.callback_server_parameters) 

229 cbport = gw._callback_server.server_socket.getsockname()[1] 

230 gw._callback_server.port = cbport 

231 # gateway with real port 

232 gw._python_proxy_port = gw._callback_server.port 

233 # get the GatewayServer object in JVM by ID 

234 jgws = JavaObject("GATEWAY_SERVER", gw._gateway_client) 

235 # update the port of CallbackClient with real port 

236 jgws.resetCallbackClient(jgws.getCallbackClient().getAddress(), gw._python_proxy_port)
