208 project = project_table 

209 entry_points = project.pop("entry-points", project.pop("entry_points", {})) 

210 renaming = {"scripts": "console_scripts", "gui_scripts": "gui_scripts"}

242def _valid_command_options(cmdclass: Mapping = EMPTY) -> Dict[str, Set[str]]: 

243 from .._importlib import metadata 

244 from setuptools.dist import Distribution 

245 

246 valid_options = {"global": _normalise_cmd_options(Distribution.global_options)} 

247 

248 unloaded_entry_points = metadata.entry_points(group='distutils.commands') 

249 loaded_entry_points = (_load_ep(ep) for ep in unloaded_entry_points) 

250 entry_points = (ep for ep in loaded_entry_points if ep) 

251 for cmd, cmd_class in chain(entry_points, cmdclass.items()): 

252 opts = valid_options.get(cmd, set()) 

253 opts = opts | _normalise_cmd_options(getattr(cmd_class, "user_options", [])) 

254 valid_options[cmd] = opts 

255 

256 return valid_options

277def _attrgetter(attr): 

278 """ 

279 Similar to ``operator.attrgetter`` but returns None if ``attr`` is not found 

280 >>> from types import SimpleNamespace 

281 >>> obj = SimpleNamespace(a=42, b=SimpleNamespace(c=13)) 

282 >>> _attrgetter("a")(obj) 

283 42 

284 >>> _attrgetter("b.c")(obj) 

285 13 

286 >>> _attrgetter("d")(obj) is None 

287 True 

288 """ 

289 return partial(reduce, lambda acc, x: getattr(acc, x, None), attr.split("."))

292def _some_attrgetter(*items): 

293 """ 

294 Return the first "truth-y" attribute or None 

295 >>> from types import SimpleNamespace 

296 >>> obj = SimpleNamespace(a=42, b=SimpleNamespace(c=13)) 

297 >>> _some_attrgetter("d", "a", "b.c")(obj) 

298 42 

299 >>> _some_attrgetter("d", "e", "b.c", "a")(obj) 

300 13 

301 >>> _some_attrgetter("d", "e", "f")(obj) is None 

302 True 

303 """ 

304 def _acessor(obj): 

305 values = (_attrgetter(i)(obj) for i in items) 

306 return next((i for i in values if i is not None), None) 

307 return _acessor

344class _WouldIgnoreField(UserWarning): 

345 """Inform users that ``pyproject.toml`` would overwrite previously defined metadata: 

346 !!\n\n 

347 ########################################################################## 

348 # configuration would be ignored/result in error due to `pyproject.toml` # 

349 ########################################################################## 

350 

351 The following seems to be defined outside of `pyproject.toml`: 

352 

353 `{field} = {value!r}` 

354 

355 According to the spec (see the link bellow), however, setuptools CANNOT 

356 consider this value unless {field!r} is listed as `dynamic`. 

357 

358 https://packaging.python.org/en/latest/specifications/declaring-project-metadata/ 

359 

360 For the time being, `setuptools` will still consider the given value (as a 

361 **transitional** measure), but please note that future releases of setuptools will 

362 follow strictly the standard. 

363 

364 To prevent this warning, you can list {field!r} under `dynamic` or alternatively 

365 remove the `[project]` table from your file and rely entirely on other means of 

366 configuration. 

367 \n\n!! 

368 """ 

369 

370 @classmethod 

371 def message(cls, field, value): 

372 from inspect import cleandoc 

373 msg = "\n".join(cls.__doc__.splitlines()[1:]) 

374 return cleandoc(msg.format(field=field, value=value))
