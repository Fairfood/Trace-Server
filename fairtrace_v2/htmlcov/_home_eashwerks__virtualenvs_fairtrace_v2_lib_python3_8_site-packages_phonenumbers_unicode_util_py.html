128def is_letter(uni_char): 

129 """Determine whether the given Unicode character is a Unicode letter""" 

130 category = Category.get(uni_char) 

131 return (category == Category.UPPERCASE_LETTER or 

132 category == Category.LOWERCASE_LETTER or 

133 category == Category.TITLECASE_LETTER or 

134 category == Category.MODIFIER_LETTER or 

135 category == Category.OTHER_LETTER)

138class _BlockRange(UnicodeMixin): 

139 """Describe the range of characters encompassed by a Unicode block""" 

140 def __init__(self, start, end, regdict=None): 

141 self.start = start 

142 self.end = end 

143 if regdict is not None: 

144 regdict[start] = self 

145 

146 def __eq__(self, other): 

147 return (self.start == other.start and self.end == other.end) 

148 

149 def __ne__(self, other): 

150 return not self == other 

151 

152 def __hash__(self): 

153 return hash((self.start, self.end)) 

154 

155 def __unicode__(self): 

156 return unicod("Block[%04x, %04x]") % (self.start, self.end)

378 def get(cls, uni_char): 

379 """Return the Unicode block of the given Unicode character""" 

380 uni_char = unicod(uni_char) # Force to Unicode 

381 code_point = ord(uni_char) 

382 if Block._RANGE_KEYS is None: 

383 Block._RANGE_KEYS = sorted(Block._RANGES.keys()) 

384 idx = bisect.bisect_left(Block._RANGE_KEYS, code_point) 

385 if (idx > 0 and 

386 code_point >= Block._RANGES[Block._RANGE_KEYS[idx - 1]].start and 

387 code_point <= Block._RANGES[Block._RANGE_KEYS[idx - 1]].end): 

388 return Block._RANGES[Block._RANGE_KEYS[idx - 1]] 

389 elif (idx < len(Block._RANGES) and 

390 code_point >= Block._RANGES[Block._RANGE_KEYS[idx]].start and 

391 code_point <= Block._RANGES[Block._RANGE_KEYS[idx]].end): 

392 return Block._RANGES[Block._RANGE_KEYS[idx]] 

393 else: 

394 return Block.UNKNOWN
