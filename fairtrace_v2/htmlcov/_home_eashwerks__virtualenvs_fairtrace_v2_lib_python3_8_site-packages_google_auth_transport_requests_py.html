78class TimeoutGuard(object): 

79 """A context manager raising an error if the suite execution took too long. 

80 

81 Args: 

82 timeout (Union[None, Union[float, Tuple[float, float]]]): 

83 The maximum number of seconds a suite can run without the context 

84 manager raising a timeout exception on exit. If passed as a tuple, 

85 the smaller of the values is taken as a timeout. If ``None``, a 

86 timeout error is never raised. 

87 timeout_error_type (Optional[Exception]): 

88 The type of the error to raise on timeout. Defaults to 

89 :class:`requests.exceptions.Timeout`. 

90 """ 

91 

92 def __init__(self, timeout, timeout_error_type=requests.exceptions.Timeout): 

93 self._timeout = timeout 

94 self.remaining_timeout = timeout 

95 self._timeout_error_type = timeout_error_type 

96 

97 def __enter__(self): 

98 self._start = time.time() 

99 return self 

100 

101 def __exit__(self, exc_type, exc_value, traceback): 

102 if exc_value: 

103 return # let the error bubble up automatically 

104 

105 if self._timeout is None: 

106 return # nothing to do, the timeout was not specified 

107 

108 elapsed = time.time() - self._start 

109 deadline_hit = False 

110 

111 if isinstance(self._timeout, numbers.Number): 

112 self.remaining_timeout = self._timeout - elapsed 

113 deadline_hit = self.remaining_timeout <= 0 

114 else: 

115 self.remaining_timeout = tuple(x - elapsed for x in self._timeout) 

116 deadline_hit = min(self.remaining_timeout) <= 0 

117 

118 if deadline_hit: 

119 raise self._timeout_error_type()

268 nothing. Assume the environment is set to ``true``, the method behaves in the 

269 following manner: 

270 

271 If client_cert_callback is provided, client certificate and private 

272 key are loaded using the callback; if client_cert_callback is None, 

273 application default SSL credentials will be used. Exceptions are raised if 

274 there are problems with the certificate, private key, or the loading process, 

275 so it should be called within a try/except block. 

276 

277 First we set the environment variable to ``true``, then create an :class:`AuthorizedSession` 

278 instance and specify the endpoints:: 

279 

280 regular_endpoint = 'https://pubsub.googleapis.com/v1/projects/{my_project_id}/topics' 

281 mtls_endpoint = 'https://pubsub.mtls.googleapis.com/v1/projects/{my_project_id}/topics' 

282 

283 authed_session = AuthorizedSession(credentials) 

284

286 

287 def my_cert_callback(): 

288 # some code to load client cert bytes and private key bytes, both in 

289 # PEM format. 

290 some_code_to_load_client_cert_and_key() 

291 if loaded: 

292 return cert, key 

293 raise MyClientCertFailureException()

431 def request( 

432 self, 

433 method, 

434 url, 

435 data=None, 

436 headers=None, 

437 max_allowed_time=None, 

438 timeout=_DEFAULT_TIMEOUT, 

439 **kwargs 

440 ): 

441 """Implementation of Requests' request. 

442 

443 Args: 

444 timeout (Optional[Union[float, Tuple[float, float]]]): 

445 The amount of time in seconds to wait for the server response 

446 with each individual request. Can also be passed as a tuple 

447 ``(connect_timeout, read_timeout)``. See :meth:`requests.Session.request` 

448 documentation for details. 

449 max_allowed_time (Optional[float]): 

450 If the method runs longer than this, a ``Timeout`` exception is 

451 automatically raised. Unlike the ``timeout`` parameter, this 

452 value applies to the total method execution time, even if 

453 multiple requests are made under the hood. 

454 

455 Mind that it is not guaranteed that the timeout error is raised 

456 at ``max_allowed_time``. It might take longer, for example, if 

457 an underlying request takes a lot of time, but the request 

458 itself does not timeout, e.g. if a large file is being 

459 transmitted. The timout error will be raised after such 

460 request completes. 

461 """ 

462 # pylint: disable=arguments-differ 

463 # Requests has a ton of arguments to request, but only two 

464 # (method, url) are required. We pass through all of the other 

465 # arguments to super, so no need to exhaustively list them here. 

466 

467 # Use a kwarg for this instead of an attribute to maintain 

468 # thread-safety. 

469 _credential_refresh_attempt = kwargs.pop("_credential_refresh_attempt", 0) 

470 

471 # Make a copy of the headers. They will be modified by the credentials 

472 # and we want to pass the original headers if we recurse. 

473 request_headers = headers.copy() if headers is not None else {} 

474 

475 # Do not apply the timeout unconditionally in order to not override the 

476 # _auth_request's default timeout. 

477 auth_request = ( 

478 self._auth_request 

479 if timeout is None 

480 else functools.partial(self._auth_request, timeout=timeout) 

481 ) 

482 

483 remaining_time = max_allowed_time 

484 

485 with TimeoutGuard(remaining_time) as guard: 

486 self.credentials.before_request(auth_request, method, url, request_headers) 

487 remaining_time = guard.remaining_timeout 

488 

489 with TimeoutGuard(remaining_time) as guard: 

490 response = super(AuthorizedSession, self).request( 

491 method, 

492 url, 

493 data=data, 

494 headers=request_headers, 

495 timeout=timeout, 

496 **kwargs 

497 ) 

498 remaining_time = guard.remaining_timeout 

499 

500 # If the response indicated that the credentials needed to be 

501 # refreshed, then refresh the credentials and re-attempt the 

502 # request. 

503 # A stored token may expire between the time it is retrieved and 

504 # the time the request is made, so we may need to try twice. 

505 if ( 

506 response.status_code in self._refresh_status_codes 

507 and _credential_refresh_attempt < self._max_refresh_attempts 

508 ): 

509 

510 _LOGGER.info( 

511 "Refreshing credentials due to a %s response. Attempt %s/%s.", 

512 response.status_code, 

513 _credential_refresh_attempt + 1, 

514 self._max_refresh_attempts, 

515 ) 

516 

517 # Do not apply the timeout unconditionally in order to not override the 

518 # _auth_request's default timeout. 

519 auth_request = ( 

520 self._auth_request 

521 if timeout is None 

522 else functools.partial(self._auth_request, timeout=timeout) 

523 ) 

524 

525 with TimeoutGuard(remaining_time) as guard: 

526 self.credentials.refresh(auth_request) 

527 remaining_time = guard.remaining_timeout 

528 

529 # Recurse. Pass in the original headers, not our modified set, but 

530 # do pass the adjusted max allowed time (i.e. the remaining total time). 

531 return self.request( 

532 method, 

533 url, 

534 data=data, 

535 headers=headers, 

536 max_allowed_time=remaining_time, 

537 timeout=timeout, 

538 _credential_refresh_attempt=_credential_refresh_attempt + 1, 

539 **kwargs 

540 ) 

541 

542 return response
