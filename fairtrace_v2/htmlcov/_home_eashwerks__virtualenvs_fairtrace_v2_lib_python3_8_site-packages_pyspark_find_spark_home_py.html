20# This script attempt to determine the correct setting for SPARK_HOME given 

21# that Spark may have been installed on the system with pip. 

22 

23import os 

24import sys 

25 

26 

27def _find_spark_home(): 

28 """Find the SPARK_HOME.""" 

29 # If the environment has SPARK_HOME set trust it. 

30 if "SPARK_HOME" in os.environ: 

31 return os.environ["SPARK_HOME"] 

32 

33 def is_spark_home(path): 

34 """Takes a path and returns true if the provided path could be a reasonable SPARK_HOME""" 

35 return os.path.isfile(os.path.join(path, "bin/spark-submit")) and ( 

36 os.path.isdir(os.path.join(path, "jars")) 

37 or os.path.isdir(os.path.join(path, "assembly")) 

38 ) 

39 

40 # Spark distribution can be downloaded when PYSPARK_HADOOP_VERSION environment variable is set. 

41 # We should look up this directory first, see also SPARK-32017. 

42 spark_dist_dir = "spark-distribution" 

43 paths = [ 

44 "../", # When we're in spark/python. 

45 ] 

46 

47 if "__file__" in globals(): 

48 paths += [ 

49 # Two case belows are valid when the current script is called as a library. 

50 os.path.join(os.path.dirname(os.path.realpath(__file__)), spark_dist_dir), 

51 os.path.dirname(os.path.realpath(__file__)), 

52 ] 

53 

54 # Add the path of the PySpark module if it exists 

55 import_error_raised = False 

56 from importlib.util import find_spec 

57 

58 try: 

59 module_home = os.path.dirname(find_spec("pyspark").origin) 

60 paths.append(os.path.join(module_home, spark_dist_dir)) 

61 paths.append(module_home) 

62 # If we are installed in edit mode also look two dirs up 

63 # Downloading different versions are not supported in edit mode. 

64 paths.append(os.path.join(module_home, "../../")) 

65 except ImportError: 

66 # Not pip installed no worries 

67 import_error_raised = True 

68 

69 # Normalize the paths 

70 paths = [os.path.abspath(p) for p in paths] 

71 

72 try: 

73 return next(path for path in paths if is_spark_home(path)) 

74 except StopIteration: 

75 print("Could not find valid SPARK_HOME while searching {0}".format(paths), file=sys.stderr) 

76 if import_error_raised: 

77 print( 

78 "\nDid you install PySpark via a package manager such as pip or Conda? If so,\n" 

79 "PySpark was not found in your Python environment. It is possible your\n" 

80 "Python environment does not properly bind with your package manager.\n" 

81 "\nPlease check your default 'python' and if you set PYSPARK_PYTHON and/or\n" 

82 "PYSPARK_DRIVER_PYTHON environment variables, and see if you can import\n" 

83 "PySpark, for example, 'python -c 'import pyspark'.\n" 

84 "\nIf you cannot import, you can install by using the Python executable directly,\n" 

85 "for example, 'python -m pip install pyspark [--user]'. Otherwise, you can also\n" 

86 "explicitly set the Python executable, that has PySpark installed, to\n" 

87 "PYSPARK_PYTHON or PYSPARK_DRIVER_PYTHON environment variables, for example,\n" 

88 "'PYSPARK_PYTHON=python3 pyspark'.\n", 

89 file=sys.stderr, 

90 ) 

91 sys.exit(-1)
