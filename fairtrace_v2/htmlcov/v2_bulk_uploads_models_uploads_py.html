103 def create_file_hash(self, file, check_type="is_used"): 

104 """Create a hash for the uploaded file. 

105 

106 This method creates a hash for the uploaded file. It checks if a file 

107 exists, calculates the hash using the 'hash_file' function, and 

108 verifies that no other uploads with the same file hash and node exist. 

109 If a file hash already exists for an active upload with the same node, 

110 an exception is raised. 

111 

112 Args: 

113 file: The uploaded file. 

114 check_type: check duplicate with 'is_used' or 'is_confirmed.' 

115 

116 Raises: 

117 Exception: If a file already exists with the same file hash and 

118 node. 

119 """ 

120 if self.file: 

121 file_hash = hash_file(file) 

122 check = ({"is_confirmed": True} 

123 if check_type == "is_confirmed" 

124 else {"is_used": True}) 

125 uploads = self.__class__.objects.filter( 

126 file_hash=file_hash, **check, node=self.node 

127 ) 

128 if uploads.exists(): 

129 raise ValidationError(f"File already" 

130 f" {check_type.split('_')[-1]}") 

131 self.file_hash = file_hash
