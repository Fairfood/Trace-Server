28 def __init__( 

29 self, 

30 check_fn: Callable, 

31 groups: Optional[Union[str, List[str]]] = None, 

32 groupby: Optional[Union[str, List[str], Callable]] = None, 

33 ignore_na: bool = True, 

34 element_wise: bool = False, 

35 name: Optional[str] = None, 

36 error: Optional[str] = None, 

37 raise_warning: bool = False, 

38 n_failure_cases: Optional[int] = None, 

39 title: Optional[str] = None, 

40 description: Optional[str] = None, 

41 statistics: Dict[str, Any] = None, 

42 strategy: Optional[SearchStrategy] = None, 

43 **check_kwargs, 

44 ) -> None: 

45 """Apply a validation function to a data object. 

46 

47 :param check_fn: A function to check pandas data structure. For Column 

48 or SeriesSchema checks, if element_wise is True, this function 

49 should have the signature: ``Callable[[pd.Series], 

50 Union[pd.Series, bool]]``, where the output series is a boolean 

51 vector. 

52 

53 If element_wise is False, this function should have the signature: 

54 ``Callable[[Any], bool]``, where ``Any`` is an element in the 

55 column. 

56 

57 For DataFrameSchema checks, if element_wise=True, fn 

58 should have the signature: ``Callable[[pd.DataFrame], 

59 Union[pd.DataFrame, pd.Series, bool]]``, where the output dataframe 

60 or series contains booleans. 

61 

62 If element_wise is True, fn is applied to each row in 

63 the dataframe with the signature ``Callable[[pd.Series], bool]`` 

64 where the series input is a row in the dataframe. 

65 :param groups: The dict input to the `fn` callable will be constrained 

66 to the groups specified by `groups`. 

67 :param groupby: If a string or list of strings is provided, these 

68 columns are used to group the Column series. If a 

69 callable is passed, the expected signature is: ``Callable[ 

70 [pd.DataFrame], pd.core.groupby.DataFrameGroupBy]`` 

71 

72 The the case of ``Column`` checks, this function has access to the 

73 entire dataframe, but ``Column.name`` is selected from this 

74 DataFrameGroupby object so that a SeriesGroupBy object is passed 

75 into ``check_fn``. 

76 

77 Specifying the groupby argument changes the ``check_fn`` signature 

78 to: 

79 

80 ``Callable[[Dict[Union[str, Tuple[str]], pd.Series]], Union[bool, pd.Series]]`` # noqa 

81 

82 where the input is a dictionary mapping 

83 keys to subsets of the column/dataframe. 

84 :param ignore_na: If True, null values will be ignored when determining 

85 if a check passed or failed. For dataframes, ignores rows with any 

86 null value. *New in version 0.4.0* 

87 :param element_wise: Whether or not to apply validator in an 

88 element-wise fashion. If bool, assumes that all checks should be 

89 applied to the column element-wise. If list, should be the same 

90 number of elements as checks. 

91 :param name: optional name for the check. 

92 :param error: custom error message if series fails validation 

93 check. 

94 :param raise_warning: if True, raise a UserWarning and do not throw 

95 exception instead of raising a SchemaError for a specific check. 

96 This option should be used carefully in cases where a failing 

97 check is informational and shouldn't stop execution of the program. 

98 :param n_failure_cases: report the first n unique failure cases. If 

99 None, report all failure cases. 

100 :param title: A human-readable label for the check. 

101 :param description: An arbitrary textual description of the check. 

102 :param statistics: kwargs to pass into the check function. These values 

103 are serialized and represent the constraints of the checks. 

104 :param strategy: A hypothesis strategy, used for implementing data 

105 synthesis strategies for this check. See the 

106 :ref:`User Guide <custom_strategies>` for more details. 

107 :param check_kwargs: key-word arguments to pass into ``check_fn`` 

108 

109 :example: 

110 

111 >>> import pandas as pd 

112 >>> import pandera as pa 

113 >>> 

114 >>> 

115 >>> # column checks are vectorized by default 

116 >>> check_positive = pa.Check(lambda s: s > 0) 

117 >>> 

118 >>> # define an element-wise check 

119 >>> check_even = pa.Check(lambda x: x % 2 == 0, element_wise=True) 

120 >>> 

121 >>> # checks can be given human-readable metadata 

122 >>> check_with_metadata = pa.Check( 

123 ... lambda x: True, 

124 ... title="Always passes", 

125 ... description="This check always passes." 

126 ... ) 

127 >>> 

128 >>> # specify assertions across categorical variables using `groupby`, 

129 >>> # for example, make sure the mean measure for group "A" is always 

130 >>> # larger than the mean measure for group "B" 

131 >>> check_by_group = pa.Check( 

132 ... lambda measures: measures["A"].mean() > measures["B"].mean(), 

133 ... groupby=["group"], 

134 ... ) 

135 >>> 

136 >>> # define a wide DataFrame-level check 

137 >>> check_dataframe = pa.Check( 

138 ... lambda df: df["measure_1"] > df["measure_2"]) 

139 >>> 

140 >>> measure_checks = [check_positive, check_even, check_by_group] 

141 >>> 

142 >>> schema = pa.DataFrameSchema( 

143 ... columns={ 

144 ... "measure_1": pa.Column(int, checks=measure_checks), 

145 ... "measure_2": pa.Column(int, checks=measure_checks), 

146 ... "group": pa.Column(str), 

147 ... }, 

148 ... checks=check_dataframe 

149 ... ) 

150 >>> 

151 >>> df = pd.DataFrame({ 

152 ... "measure_1": [10, 12, 14, 16], 

153 ... "measure_2": [2, 4, 6, 8], 

154 ... "group": ["B", "B", "A", "A"] 

155 ... }) 

156 >>> 

157 >>> schema.validate(df)[["measure_1", "measure_2", "group"]] 

158 measure_1 measure_2 group 

159 0 10 2 B 

160 1 12 4 B 

161 2 14 6 A 

162 3 16 8 A 

163 

164 See :ref:`here<checks>` for more usage details. 

165 

166 """ 

167 super().__init__(name=name, error=error) 

168 if element_wise and groupby is not None: 

169 raise errors.SchemaInitError( 

170 "Cannot use groupby when element_wise=True." 

171 ) 

172 self._check_fn = check_fn 

173 self._check_kwargs = check_kwargs 

174 self.element_wise = element_wise 

175 self.name = name or getattr( 

176 self._check_fn, "__name__", self._check_fn.__class__.__name__ 

177 ) 

178 self.ignore_na = ignore_na 

179 self.raise_warning = raise_warning 

180 self.n_failure_cases = n_failure_cases 

181 self.title = title 

182 self.description = description 

183 

184 if groupby is None and groups is not None: 

185 raise ValueError( 

186 "`groupby` argument needs to be provided when `groups` " 

187 "argument is defined" 

188 ) 

189 

190 if isinstance(groupby, str): 

191 groupby = [groupby] 

192 self.groupby = groupby 

193 if isinstance(groups, str): 

194 groups = [groups] 

195 self.groups: Optional[List[str]] = groups 

196 

197 self.statistics = statistics or check_kwargs or {} 

198 self.statistics_args = [*self.statistics.keys()] 

199 self.strategy = strategy

201 def __call__( 

202 self, 

203 check_obj: Union[pd.DataFrame, pd.Series], 

204 column: Optional[str] = None, 

205 ) -> CheckResult: 

206 # pylint: disable=too-many-branches 

207 """Validate pandas DataFrame or Series. 

208 

209 :param check_obj: pandas DataFrame of Series to validate. 

210 :param column: for dataframe checks, apply the check function to this 

211 column. 

212 :returns: CheckResult tuple containing: 

213 

214 ``check_output``: boolean scalar, ``Series`` or ``DataFrame`` 

215 indicating which elements passed the check. 

216 

217 ``check_passed``: boolean scalar that indicating whether the check 

218 passed overall. 

219 

220 ``checked_object``: the checked object itself. Depending on the 

221 options provided to the ``Check``, this will be a pandas Series, 

222 DataFrame, or if the ``groupby`` option is specified, a 

223 ``Dict[str, Series]`` or ``Dict[str, DataFrame]`` where the keys 

224 are distinct groups. 

225 

226 ``failure_cases``: subset of the check_object that failed. 

227 """ 

228 backend = self.get_backend(check_obj)(self) 

229 return backend(check_obj, column)
