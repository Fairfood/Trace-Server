17def _rel_helper(lhs, rhs, ident=None, relation_type=None, direction=None, relation_properties=None, **kwargs): 

18 """ 

19 Generate a relationship matching string, with specified parameters. 

20 Examples: 

21 relation_direction = OUTGOING: (lhs)-[relation_ident:relation_type]->(rhs) 

22 relation_direction = INCOMING: (lhs)<-[relation_ident:relation_type]-(rhs) 

23 relation_direction = EITHER: (lhs)-[relation_ident:relation_type]-(rhs) 

24 

25 :param lhs: The left hand statement. 

26 :type lhs: str 

27 :param rhs: The right hand statement. 

28 :type rhs: str 

29 :param ident: A specific identity to name the relationship, or None. 

30 :type ident: str 

31 :param relation_type: None for all direct rels, * for all of any length, or a name of an explicit rel. 

32 :type relation_type: str 

33 :param direction: None or EITHER for all OUTGOING,INCOMING,EITHER. Otherwise OUTGOING or INCOMING. 

34 :param relation_properties: dictionary of relationship properties to match 

35 :returns: string 

36 """ 

37 

38 if direction == OUTGOING: 

39 stmt = '-{0}->' 

40 elif direction == INCOMING: 

41 stmt = '<-{0}-' 

42 else: 

43 stmt = '-{0}-' 

44 

45 rel_props = '' 

46 

47 if relation_properties: 

48 rel_props = ' {{{0}}}'.format(', '.join( 

49 ['{0}: {1}'.format(key, value) for key, value in relation_properties.items()])) 

50 

51 # direct, relation_type=None is unspecified, relation_type 

52 if relation_type is None: 

53 stmt = stmt.format('') 

54 # all("*" wildcard) relation_type 

55 elif relation_type == '*': 

56 stmt = stmt.format('[*]') 

57 else: 

58 # explicit relation_type 

59 stmt = stmt.format('[{0}:`{1}`{2}]'.format(ident if ident else '', relation_type, rel_props)) 

60 

61 return "({0}){1}({2})".format(lhs, stmt, rhs)

181def process_filter_args(cls, kwargs): 

182 """ 

183 loop through properties in filter parameters check they match class definition 

184 deflate them and convert into something easy to generate cypher from 

185 """ 

186 

187 output = {} 

188 

189 for key, value in kwargs.items(): 

190 if '__' in key: 

191 prop, operator = key.rsplit('__') 

192 operator = OPERATOR_TABLE[operator] 

193 else: 

194 prop = key 

195 operator = '=' 

196 

197 if prop not in cls.defined_properties(rels=False): 

198 raise ValueError("No such property {0} on {1}".format(prop, cls.__name__)) 

199 

200 property_obj = getattr(cls, prop) 

201 if isinstance(property_obj, AliasProperty): 

202 prop = property_obj.aliased_to() 

203 deflated_value = getattr(cls, prop).deflate(value) 

204 else: 

205 # handle special operators 

206 if operator == _SPECIAL_OPERATOR_IN: 

207 if not isinstance(value, tuple) and not isinstance(value, list): 

208 raise ValueError('Value must be a tuple or list for IN operation {0}={1}'.format(key, value)) 

209 deflated_value = [property_obj.deflate(v) for v in value] 

210 elif operator == _SPECIAL_OPERATOR_ISNULL: 

211 if not isinstance(value, bool): 

212 raise ValueError('Value must be a bool for isnull operation on {0}'.format(key)) 

213 operator = 'IS NULL' if value else 'IS NOT NULL' 

214 deflated_value = None 

215 elif operator in _REGEX_OPERATOR_TABLE.values(): 

216 deflated_value = property_obj.deflate(value) 

217 if not isinstance(deflated_value, basestring): 

218 raise ValueError('Must be a string value for {0}'.format(key)) 

219 if operator in _STRING_REGEX_OPERATOR_TABLE.values(): 

220 deflated_value = re.escape(deflated_value) 

221 deflated_value = operator.format(deflated_value) 

222 operator = _SPECIAL_OPERATOR_REGEX 

223 else: 

224 deflated_value = property_obj.deflate(value) 

225 

226 # map property to correct property name in the database 

227 db_property = cls.defined_properties(rels=False)[prop].db_property or prop 

228 

229 output[db_property] = (operator, deflated_value) 

230 

231 return output

390 def _parse_q_filters(self, ident, q, source_class): 

391 target = [] 

392 for child in q.children: 

393 if isinstance(child, QBase): 

394 q_childs = self._parse_q_filters(ident, child, source_class) 

395 if child.connector == Q.OR: 

396 q_childs = "(" + q_childs + ")" 

397 target.append(q_childs) 

398 else: 

399 kwargs = {child[0]: child[1]} 

400 filters = process_filter_args(source_class, kwargs) 

401 for prop, op_and_val in filters.items(): 

402 op, val = op_and_val 

403 if op in _UNARY_OPERATORS: 

404 # unary operators do not have a parameter 

405 statement = '{0}.{1} {2}'.format(ident, prop, op) 

406 else: 

407 place_holder = self._register_place_holder(ident + '_' + prop) 

408 statement = '{0}.{1} {2} ${3}'.format(ident, prop, op, place_holder) 

409 self._query_params[place_holder] = val 

410 target.append(statement) 

411 ret = ' {0} '.format(q.connector).join(target) 

412 if q.negated: 

413 ret = 'NOT ({0})'.format(ret) 

414 return ret

416 def build_where_stmt(self, ident, filters, q_filters=None, source_class=None): 

417 """ 

418 construct a where statement from some filters 

419 """ 

420 if q_filters is not None: 

421 stmts = self._parse_q_filters(ident, q_filters, source_class) 

422 if stmts: 

423 self._ast['where'].append(stmts) 

424 else: 

425 stmts = [] 

426 for row in filters: 

427 negate = False 

428 

429 # pre-process NOT cases as they are nested dicts 

430 if '__NOT__' in row and len(row) == 1: 

431 negate = True 

432 row = row['__NOT__'] 

433 

434 for prop, op_and_val in row.items(): 

435 op, val = op_and_val 

436 if op in _UNARY_OPERATORS: 

437 # unary operators do not have a parameter 

438 statement = '{0} {1}.{2} {3}'.format('NOT' if negate else '', ident, prop, op) 

439 else: 

440 place_holder = self._register_place_holder(ident + '_' + prop) 

441 statement = '{0} {1}.{2} {3} ${4}'.format('NOT' if negate else '', ident, prop, op, place_holder) 

442 self._query_params[place_holder] = val 

443 stmts.append(statement) 

444 

445 self._ast['where'].append(' AND '.join(stmts))

648 def filter(self, *args, **kwargs): 

649 """ 

650 Apply filters to the existing nodes in the set. 

651 

652 :param kwargs: filter parameters 

653 

654 Filters mimic Django's syntax with the double '__' to separate field and operators. 

655 

656 e.g `.filter(salary__gt=20000)` results in `salary > 20000`. 

657 

658 The following operators are available: 

659 

660 * 'lt': less than 

661 * 'gt': greater than 

662 * 'lte': less than or equal to 

663 * 'gte': greater than or equal to 

664 * 'ne': not equal to 

665 * 'in': matches one of list (or tuple) 

666 * 'isnull': is null 

667 * 'regex': matches supplied regex (neo4j regex format) 

668 * 'exact': exactly match string (just '=') 

669 * 'iexact': case insensitive match string 

670 * 'contains': contains string 

671 * 'icontains': case insensitive contains 

672 * 'startswith': string starts with 

673 * 'istartswith': case insensitive string starts with 

674 * 'endswith': string ends with 

675 * 'iendswith': case insensitive string ends with 

676 

677 :return: self 

678 """ 

679 if args or kwargs: 

680 self.q_filters = Q(self.q_filters & Q(*args, **kwargs)) 

681 return self

700 def order_by(self, *props): 

701 """ 

702 Order by properties. Prepend with minus to do descending. Pass None to 

703 remove ordering. 

704 """ 

705 should_remove = len(props) == 1 and props[0] is None 

706 if not hasattr(self, '_order_by') or should_remove: 

707 self._order_by = [] 

708 if should_remove: 

709 return self 

710 if '?' in props: 

711 self._order_by.append('?') 

712 else: 

713 for prop in props: 

714 prop = prop.strip() 

715 if prop.startswith('-'): 

716 prop = prop[1:] 

717 desc = True 

718 else: 

719 desc = False 

720 

721 if prop not in self.source_class.defined_properties(rels=False): 

722 raise ValueError("No such property {0} on {1}".format( 

723 prop, self.source_class.__name__)) 

724 

725 property_obj = getattr(self.source_class, prop) 

726 if isinstance(property_obj, AliasProperty): 

727 prop = property_obj.aliased_to() 

728 

729 self._order_by.append(prop + (' DESC' if desc else '')) 

730 

731 return self
