47 @staticmethod 

48 def majorMinorVersion(sparkVersion: str) -> Tuple[int, int]: 

49 """ 

50 Given a Spark version string, return the (major version number, minor version number). 

51 E.g., for 2.0.1-SNAPSHOT, return (2, 0). 

52 

53 Examples 

54 -------- 

55 >>> sparkVersion = "2.4.0" 

56 >>> VersionUtils.majorMinorVersion(sparkVersion) 

57 (2, 4) 

58 >>> sparkVersion = "2.3.0-SNAPSHOT" 

59 >>> VersionUtils.majorMinorVersion(sparkVersion) 

60 (2, 3) 

61 """ 

62 m = re.search(r"^(\d+)\.(\d+)(\..*)?$", sparkVersion) 

63 if m is not None: 

64 return (int(m.group(1)), int(m.group(2))) 

65 else: 

66 raise ValueError( 

67 "Spark tried to parse '%s' as a Spark" % sparkVersion 

68 + " version string, but it could not find the major and minor" 

69 + " version numbers." 

70 )

74 """

76 prevents silent loss of data when 'f' is used in a for loop in Spark code

96def try_simplify_traceback(tb: TracebackType) -> Optional[TracebackType]: 

97 """ 

98 Simplify the traceback. It removes the tracebacks in the current package, and only 

99 shows the traceback that is related to the thirdparty and user-specified codes. 

100 

101 Returns 

102 ------- 

103 TracebackType or None 

104 Simplified traceback instance. It returns None if it fails to simplify. 

105 

106 Notes 

107 ----- 

108 This keeps the tracebacks once it sees they are from a different file even 

109 though the following tracebacks are from the current package. 

110 

111 Examples 

112 -------- 

113 >>> import importlib 

114 >>> import sys 

115 >>> import traceback 

116 >>> import tempfile 

117 >>> with tempfile.TemporaryDirectory() as tmp_dir: 

118 ... with open("%s/dummy_module.py" % tmp_dir, "w") as f: 

119 ... _ = f.write( 

120 ... 'def raise_stop_iteration():\\n' 

121 ... ' raise StopIteration()\\n\\n' 

122 ... 'def simple_wrapper(f):\\n' 

123 ... ' def wrapper(*a, **k):\\n' 

124 ... ' return f(*a, **k)\\n' 

125 ... ' return wrapper\\n') 

126 ... f.flush() 

127 ... spec = importlib.util.spec_from_file_location( 

128 ... "dummy_module", "%s/dummy_module.py" % tmp_dir) 

129 ... dummy_module = importlib.util.module_from_spec(spec) 

130 ... spec.loader.exec_module(dummy_module) 

131 >>> def skip_doctest_traceback(tb): 

132 ... import pyspark 

133 ... root = os.path.dirname(pyspark.__file__) 

134 ... pairs = zip(walk_tb(tb), traceback.extract_tb(tb)) 

135 ... for cur_tb, cur_frame in pairs: 

136 ... if cur_frame.filename.startswith(root): 

137 ... return cur_tb 

138 

139 Regular exceptions should show the file name of the current package as below. 

140 

141 >>> exc_info = None 

142 >>> try: 

143 ... fail_on_stopiteration(dummy_module.raise_stop_iteration)() 

144 ... except Exception as e: 

145 ... tb = sys.exc_info()[-1] 

146 ... e.__cause__ = None 

147 ... exc_info = "".join( 

148 ... traceback.format_exception(type(e), e, tb)) 

149 >>> print(exc_info) # doctest: +NORMALIZE_WHITESPACE, +ELLIPSIS 

150 Traceback (most recent call last): 

151 File ... 

152 ... 

153 File "/.../pyspark/util.py", line ... 

154 ... 

155 RuntimeError: ... 

156 >>> "pyspark/util.py" in exc_info 

157 True 

158 

159 If the traceback is simplified with this method, it hides the current package file name: 

160 

161 >>> exc_info = None 

162 >>> try: 

163 ... fail_on_stopiteration(dummy_module.raise_stop_iteration)() 

164 ... except Exception as e: 

165 ... tb = try_simplify_traceback(sys.exc_info()[-1]) 

166 ... e.__cause__ = None 

167 ... exc_info = "".join( 

168 ... traceback.format_exception( 

169 ... type(e), e, try_simplify_traceback(skip_doctest_traceback(tb)))) 

170 >>> print(exc_info) # doctest: +NORMALIZE_WHITESPACE, +ELLIPSIS 

171 RuntimeError: ... 

172 >>> "pyspark/util.py" in exc_info 

173 False 

174 

175 In the case below, the traceback contains the current package in the middle. 

176 In this case, it just hides the top occurrence only. 

177 

178 >>> exc_info = None 

179 >>> try: 

180 ... fail_on_stopiteration(dummy_module.simple_wrapper( 

181 ... fail_on_stopiteration(dummy_module.raise_stop_iteration)))() 

182 ... except Exception as e: 

183 ... tb = sys.exc_info()[-1] 

184 ... e.__cause__ = None 

185 ... exc_info_a = "".join( 

186 ... traceback.format_exception(type(e), e, tb)) 

187 ... exc_info_b = "".join( 

188 ... traceback.format_exception( 

189 ... type(e), e, try_simplify_traceback(skip_doctest_traceback(tb)))) 

190 >>> exc_info_a.count("pyspark/util.py") 

191 2 

192 >>> exc_info_b.count("pyspark/util.py") 

193 1 

194 """ 

195 if "pypy" in platform.python_implementation().lower(): 

196 # Traceback modification is not supported with PyPy in PySpark. 

197 return None 

198 if sys.version_info[:2] < (3, 7): 

199 # Traceback creation is not supported Python < 3.7. 

200 # See https://bugs.python.org/issue30579. 

201 return None 

202 

203 import pyspark 

204 

205 root = os.path.dirname(pyspark.__file__) 

206 tb_next = None 

207 new_tb = None 

208 pairs = zip(walk_tb(tb), traceback.extract_tb(tb)) 

209 last_seen = [] 

210 

211 for cur_tb, cur_frame in pairs: 

212 if not cur_frame.filename.startswith(root): 

213 # Filter the stacktrace from the PySpark source itself. 

214 last_seen = [(cur_tb, cur_frame)] 

215 break 

216 

217 for cur_tb, cur_frame in reversed(list(itertools.chain(last_seen, pairs))): 

218 # Once we have seen the file names outside, don't skip. 

219 new_tb = TracebackType( 

220 tb_next=tb_next, 

221 tb_frame=cur_tb.tb_frame, 

222 tb_lasti=cur_tb.tb_frame.f_lasti, 

223 tb_lineno=cur_tb.tb_frame.f_lineno if cur_tb.tb_frame.f_lineno is not None else -1, 

224 ) 

225 tb_next = new_tb 

226 return new_tb

259def _parse_memory(s: str) -> int: 

260 """ 

261 Parse a memory string in the format supported by Java (e.g. 1g, 200m) and 

262 return the value in MiB 

263 

264 Examples 

265 -------- 

266 >>> _parse_memory("256m") 

267 256 

268 >>> _parse_memory("2g") 

269 2048 

270 """ 

271 units = {"g": 1024, "m": 1, "t": 1 << 20, "k": 1.0 / 1024} 

272 if s[-1].lower() not in units: 

273 raise ValueError("invalid format: " + s) 

274 return int(float(s[:-1]) * units[s[-1].lower()])

277def inheritable_thread_target(f: Callable) -> Callable: 

278 """ 

279 Return thread target wrapper which is recommended to be used in PySpark when the 

280 pinned thread mode is enabled. The wrapper function, before calling original 

281 thread target, it inherits the inheritable properties specific 

282 to JVM thread such as ``InheritableThreadLocal``. 

283 

284 Also, note that pinned thread mode does not close the connection from Python 

285 to JVM when the thread is finished in the Python side. With this wrapper, Python 

286 garbage-collects the Python thread instance and also closes the connection 

287 which finishes JVM thread correctly. 

288 

289 When the pinned thread mode is off, it return the original ``f``. 

290 

291 .. versionadded:: 3.2.0 

292 

293 Parameters 

294 ---------- 

295 f : function 

296 the original thread target. 

297 

298 Notes 

299 ----- 

300 This API is experimental. 

301 

302 It is important to know that it captures the local properties when you decorate it 

303 whereas :class:`InheritableThread` captures when the thread is started. 

304 Therefore, it is encouraged to decorate it when you want to capture the local 

305 properties. 

306 

307 For example, the local properties from the current Spark context is captured 

308 when you define a function here instead of the invocation: 

309 

310 >>> @inheritable_thread_target 

311 ... def target_func(): 

312 ... pass # your codes. 

313 

314 If you have any updates on local properties afterwards, it would not be reflected to 

315 the Spark context in ``target_func()``. 

316 

317 The example below mimics the behavior of JVM threads as close as possible: 

318 

319 >>> Thread(target=inheritable_thread_target(target_func)).start() # doctest: +SKIP 

320 """ 

321 from pyspark import SparkContext 

322 

323 if isinstance(SparkContext._gateway, ClientServer): 

324 # Here's when the pinned-thread mode (PYSPARK_PIN_THREAD) is on. 

325 

326 # NOTICE the internal difference vs `InheritableThread`. `InheritableThread` 

327 # copies local properties when the thread starts but `inheritable_thread_target` 

328 # copies when the function is wrapped. 

329 assert SparkContext._active_spark_context is not None 

330 properties = SparkContext._active_spark_context._jsc.sc().getLocalProperties().clone() 

331 

332 @functools.wraps(f) 

333 def wrapped(*args: Any, **kwargs: Any) -> Any: 

334 # Set local properties in child thread. 

335 assert SparkContext._active_spark_context is not None 

336 SparkContext._active_spark_context._jsc.sc().setLocalProperties(properties) 

337 return f(*args, **kwargs) 

338 

339 return wrapped 

340 else: 

341 return f

344class InheritableThread(threading.Thread): 

345 """ 

346 Thread that is recommended to be used in PySpark instead of :class:`threading.Thread` 

347 when the pinned thread mode is enabled. The usage of this class is exactly same as 

348 :class:`threading.Thread` but correctly inherits the inheritable properties specific 

349 to JVM thread such as ``InheritableThreadLocal``. 

350 

351 Also, note that pinned thread mode does not close the connection from Python 

352 to JVM when the thread is finished in the Python side. With this class, Python 

353 garbage-collects the Python thread instance and also closes the connection 

354 which finishes JVM thread correctly. 

355 

356 When the pinned thread mode is off, this works as :class:`threading.Thread`. 

357 

358 .. versionadded:: 3.1.0 

359 

360 Notes 

361 ----- 

362 This API is experimental. 

363 """ 

364 

365 _props: JavaObject 

366 

367 def __init__(self, target: Callable, *args: Any, **kwargs: Any): 

368 from pyspark import SparkContext 

369 

370 if isinstance(SparkContext._gateway, ClientServer): 

371 # Here's when the pinned-thread mode (PYSPARK_PIN_THREAD) is on. 

372 def copy_local_properties(*a: Any, **k: Any) -> Any: 

373 # self._props is set before starting the thread to match the behavior with JVM. 

374 assert hasattr(self, "_props") 

375 assert SparkContext._active_spark_context is not None 

376 SparkContext._active_spark_context._jsc.sc().setLocalProperties(self._props) 

377 return target(*a, **k) 

378 

379 super(InheritableThread, self).__init__( 

380 target=copy_local_properties, *args, **kwargs # type: ignore[misc] 

381 ) 

382 else: 

383 super(InheritableThread, self).__init__( 

384 target=target, *args, **kwargs # type: ignore[misc] 

385 ) 

386 

387 def start(self) -> None: 

388 from pyspark import SparkContext 

389 

390 if isinstance(SparkContext._gateway, ClientServer): 

391 # Here's when the pinned-thread mode (PYSPARK_PIN_THREAD) is on. 

392 

393 # Local property copy should happen in Thread.start to mimic JVM's behavior. 

394 assert SparkContext._active_spark_context is not None 

395 self._props = SparkContext._active_spark_context._jsc.sc().getLocalProperties().clone() 

396 return super(InheritableThread, self).start()
