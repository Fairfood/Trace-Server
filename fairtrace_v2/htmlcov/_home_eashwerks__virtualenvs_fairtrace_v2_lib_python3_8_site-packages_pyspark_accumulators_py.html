59class Accumulator(Generic[T]): 

60 

61 """ 

62 A shared variable that can be accumulated, i.e., has a commutative and associative "add" 

63 operation. Worker tasks on a Spark cluster can add values to an Accumulator with the `+=` 

64 operator, but only the driver program is allowed to access its value, using `value`. 

65 Updates from the workers get propagated automatically to the driver program. 

66 

67 While :class:`SparkContext` supports accumulators for primitive data types like :class:`int` and 

68 :class:`float`, users can also define accumulators for custom types by providing a custom 

69 :py:class:`AccumulatorParam` object. Refer to its doctest for an example. 

70 

71 Examples 

72 -------- 

73 >>> a = sc.accumulator(1) 

74 >>> a.value 

75 1 

76 >>> a.value = 2 

77 >>> a.value 

78 2 

79 >>> a += 5 

80 >>> a.value 

81 7 

82 >>> sc.accumulator(1.0).value 

83 1.0 

84 >>> sc.accumulator(1j).value 

85 1j 

86 >>> rdd = sc.parallelize([1,2,3]) 

87 >>> def f(x): 

88 ... global a 

89 ... a += x 

90 >>> rdd.foreach(f) 

91 >>> a.value 

92 13 

93 >>> b = sc.accumulator(0) 

94 >>> def g(x): 

95 ... b.add(x) 

96 >>> rdd.foreach(g) 

97 >>> b.value 

98 6 

99 

100 >>> rdd.map(lambda x: a.value).collect() # doctest: +IGNORE_EXCEPTION_DETAIL 

101 Traceback (most recent call last): 

102 ... 

103 Py4JJavaError: ... 

104 

105 >>> def h(x): 

106 ... global a 

107 ... a.value = 7 

108 >>> rdd.foreach(h) # doctest: +IGNORE_EXCEPTION_DETAIL 

109 Traceback (most recent call last): 

110 ... 

111 Py4JJavaError: ... 

112 

113 >>> sc.accumulator([1.0, 2.0, 3.0]) # doctest: +IGNORE_EXCEPTION_DETAIL 

114 Traceback (most recent call last): 

115 ... 

116 TypeError: ... 

117 """ 

118

124 self.accum_param = accum_param

151 self._value = value

169class AccumulatorParam(Generic[T]): 

170 

171 """ 

172 Helper object that defines how to accumulate values of a given type. 

173 

174 Examples 

175 -------- 

176 >>> from pyspark.accumulators import AccumulatorParam 

177 >>> class VectorAccumulatorParam(AccumulatorParam): 

178 ... def zero(self, value): 

179 ... return [0.0] * len(value) 

180 ... def addInPlace(self, val1, val2): 

181 ... for i in range(len(val1)): 

182 ... val1[i] += val2[i] 

183 ... return val1 

184 >>> va = sc.accumulator([1.0, 2.0, 3.0], VectorAccumulatorParam()) 

185 >>> va.value 

186 [1.0, 2.0, 3.0] 

187 >>> def g(x): 

188 ... global va 

189 ... va += [x] * 3 

190 >>> rdd = sc.parallelize([1,2,3]) 

191 >>> rdd.foreach(g) 

192 >>> va.value 

193 [7.0, 8.0, 9.0] 

194 """

236class _UpdateRequestHandler(SocketServer.StreamRequestHandler): 

237 

238 """ 

239 This handler will keep polling updates from the same socket until the 

240 server is shutdown. 

241 """ 

242 

243 def handle(self) -> None: 

244 from pyspark.accumulators import _accumulatorRegistry 

245 

246 auth_token = self.server.auth_token # type: ignore[attr-defined] 

247 

248 def poll(func: Callable[[], bool]) -> None: 

249 while not self.server.server_shutdown: # type: ignore[attr-defined] 

250 # Poll every 1 second for new data -- don't block in case of shutdown. 

251 r, _, _ = select.select([self.rfile], [], [], 1) 

252 if self.rfile in r: 

253 if func(): 

254 break 

255 

256 def accum_updates() -> bool: 

257 num_updates = read_int(self.rfile) 

258 for _ in range(num_updates): 

259 (aid, update) = pickleSer._read_with_length(self.rfile) 

260 _accumulatorRegistry[aid] += update 

261 # Write a byte in acknowledgement 

262 self.wfile.write(struct.pack("!b", 1)) 

263 return False 

264 

265 def authenticate_and_accum_updates() -> bool: 

266 received_token: Union[bytes, str] = self.rfile.read(len(auth_token)) 

267 if isinstance(received_token, bytes): 

268 received_token = received_token.decode("utf-8") 

269 if received_token == auth_token: 

270 accum_updates() 

271 # we've authenticated, we can break out of the first loop now 

272 return True 

273 else: 

274 raise ValueError( 

275 "The value of the provided token to the AccumulatorServer is not correct." 

276 ) 

277 

278 # first we keep polling till we've received the authentication token 

279 poll(authenticate_and_accum_updates) 

280 # now we've authenticated, don't need to check for the token anymore 

281 poll(accum_updates)

306def _start_update_server(auth_token: str) -> AccumulatorServer: 

307 """Start a TCP server to receive accumulator updates in a daemon thread, and returns it""" 

308 server = AccumulatorServer(("localhost", 0), _UpdateRequestHandler, auth_token) 

309 thread = threading.Thread(target=server.serve_forever) 

310 thread.daemon = True 

311 thread.start() 

312 return server
