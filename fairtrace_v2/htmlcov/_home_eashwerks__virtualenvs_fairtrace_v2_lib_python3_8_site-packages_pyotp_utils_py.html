15def build_uri(secret, name, initial_count=None, issuer_name=None, 

16 algorithm=None, digits=None, period=None): 

17 """ 

18 Returns the provisioning URI for the OTP; works for either TOTP or HOTP. 

19 

20 This can then be encoded in a QR Code and used to provision the Google 

21 Authenticator app. 

22 

23 For module-internal use. 

24 

25 See also: 

26 https://github.com/google/google-authenticator/wiki/Key-Uri-Format 

27 

28 :param secret: the hotp/totp secret used to generate the URI 

29 :type secret: str 

30 :param name: name of the account 

31 :type name: str 

32 :param initial_count: starting counter value, defaults to None. 

33 If none, the OTP type will be assumed as TOTP. 

34 :type initial_count: int 

35 :param issuer_name: the name of the OTP issuer; this will be the 

36 organization title of the OTP entry in Authenticator 

37 :type issuer_name: str 

38 :param algorithm: the algorithm used in the OTP generation. 

39 :type algorithm: str 

40 :param digits: the length of the OTP generated code. 

41 :type digits: int 

42 :param period: the number of seconds the OTP generator is set to 

43 expire every code. 

44 :type period: int 

45 :returns: provisioning uri 

46 :rtype: str 

47 """ 

48 # initial_count may be 0 as a valid param 

49 is_initial_count_present = (initial_count is not None) 

50 

51 # Handling values different from defaults 

52 is_algorithm_set = (algorithm is not None and algorithm != 'sha1') 

53 is_digits_set = (digits is not None and digits != 6) 

54 is_period_set = (period is not None and period != 30) 

55 

56 otp_type = 'hotp' if is_initial_count_present else 'totp' 

57 base_uri = 'otpauth://{0}/{1}?{2}' 

58 

59 url_args = {'secret': secret} 

60 

61 label = quote(name) 

62 if issuer_name is not None: 

63 label = quote(issuer_name) + ':' + label 

64 url_args['issuer'] = issuer_name 

65 

66 if is_initial_count_present: 

67 url_args['counter'] = initial_count 

68 if is_algorithm_set: 

69 url_args['algorithm'] = algorithm.upper() 

70 if is_digits_set: 

71 url_args['digits'] = digits 

72 if is_period_set: 

73 url_args['period'] = period 

74 

75 uri = base_uri.format(otp_type, label, urlencode(url_args).replace("+", "%20")) 

76 return uri

79def _compare_digest(s1, s2): 

80 differences = 0 

81 for c1, c2 in izip_longest(s1, s2): 

82 if c1 is None or c2 is None: 

83 differences = 1 

84 continue 

85 differences |= ord(c1) ^ ord(c2) 

86 return differences == 0

98def strings_equal(s1, s2): 

99 """ 

100 Timing-attack resistant string comparison. 

101 

102 Normal comparison using == will short-circuit on the first mismatching 

103 character. This avoids that by scanning the whole string, though we 

104 still reveal to a timing attack whether the strings are the same 

105 length. 

106 """
