111 @classmethod 

112 def get_handshake(cls): 

113 """ Return the supported Bolt versions as bytes. 

114 The length is 16 bytes as specified in the Bolt version negotiation. 

115 :return: bytes 

116 """ 

117 offered_versions = sorted(cls.protocol_handlers().keys(), reverse=True)[:4] 

118 return b"".join(version.to_bytes() for version in offered_versions).ljust(16, b"\x00")

120 @classmethod 

121 def protocol_handlers(cls, protocol_version=None): 

122 """ Return a dictionary of available Bolt protocol handlers, 

123 keyed by version tuple. If an explicit protocol version is 

124 provided, the dictionary will contain either zero or one items, 

125 depending on whether that version is supported. If no protocol 

126 version is provided, all available versions will be returned. 

127 

128 :param protocol_version: tuple identifying a specific protocol 

129 version (e.g. (3, 5)) or None 

130 :return: dictionary of version tuple to handler class for all 

131 relevant and supported protocol versions 

132 :raise TypeError: if protocol version is not passed in a tuple 

133 """

373 def _acquire(self, address, timeout): 

374 """ Acquire a connection to a given address from the pool. 

375 The address supplied should always be an IP address, not 

376 a host name. 

377 

378 This method is thread safe. 

379 """ 

380 t0 = perf_counter() 

381 if timeout is None: 

382 timeout = self.workspace_config.connection_acquisition_timeout 

383 

384 with self.lock: 

385 try: 

386 connections = self.connections[address] 

387 except KeyError: 

388 connections = self.connections[address] = deque() 

389 

390 def time_remaining(): 

391 t = timeout - (perf_counter() - t0) 

392 return t if t > 0 else 0 

393 

394 while True: 

395 # try to find a free connection in pool 

396 for connection in list(connections): 

397 if connection.closed() or connection.defunct() or connection.timedout(): 

398 connections.remove(connection) 

399 continue 

400 if not connection.in_use: 

401 connection.in_use = True 

402 return connection 

403 # all connections in pool are in-use 

404 infinite_pool_size = (self.pool_config.max_connection_pool_size < 0 or self.pool_config.max_connection_pool_size == float("inf")) 

405 can_create_new_connection = infinite_pool_size or len(connections) < self.pool_config.max_connection_pool_size 

406 if can_create_new_connection: 

407 timeout = min(self.pool_config.connection_timeout, time_remaining()) 

408 try: 

409 connection = self.opener(address, timeout) 

410 except ServiceUnavailable: 

411 self.remove(address) 

412 raise 

413 else: 

414 connection.pool = self 

415 connection.in_use = True 

416 connections.append(connection) 

417 return connection 

418 

419 # failed to obtain a connection from pool because the 

420 # pool is full and no free connection in the pool 

421 if time_remaining(): 

422 self.cond.wait(time_remaining()) 

423 # if timed out, then we throw error. This time 

424 # computation is needed, as with python 2.7, we 

425 # cannot tell if the condition is notified or 

426 # timed out when we come to this line 

427 if not time_remaining(): 

428 raise ClientError("Failed to obtain a connection from pool " 

429 "within {!r}s".format(timeout)) 

430 else: 

431 raise ClientError("Failed to obtain a connection from pool " 

432 "within {!r}s".format(timeout))

442 def release(self, *connections): 

443 """ Release a connection back into the pool. 

444 This method is thread safe. 

445 """ 

446 with self.lock: 

447 for connection in connections: 

448 connection.in_use = False 

449 self.cond.notify_all()

451 def in_use_connection_count(self, address): 

452 """ Count the number of connections currently in use to a given 

453 address. 

454 """ 

455 try: 

456 connections = self.connections[address] 

457 except KeyError: 

458 return 0 

459 else: 

460 return sum(1 if connection.in_use else 0 for connection in connections) 

461 

462 def deactivate(self, address): 

463 """ Deactivate an address from the connection pool, if present, closing 

464 all idle connection to that address 

465 """ 

466 with self.lock: 

467 try: 

468 connections = self.connections[address] 

469 except KeyError: # already removed from the connection pool 

470 return 

471 for conn in list(connections): 

472 if not conn.in_use: 

473 connections.remove(conn) 

474 try: 

475 conn.close() 

476 except IOError: 

477 pass 

478 if not connections: 

479 self.remove(address)

481 def on_write_failure(self, address):

483 

484 def remove(self, address): 

485 """ Remove an address from the connection pool, if present, closing 

486 all connections to that address. 

487 """ 

488 with self.lock: 

489 for connection in self.connections.pop(address, ()): 

490 try: 

491 connection.close() 

492 except IOError: 

493 pass

495 def close(self): 

496 """ Close all connections and empty the pool. 

497 This method is thread safe. 

498 """ 

499 try: 

500 with self.lock: 

501 for address in list(self.connections): 

502 self.remove(address) 

503 except TypeError: 

504 pass

636 def fetch_routing_info(self, *, address, timeout, database): 

637 """ Fetch raw routing info from a given router address. 

638 

639 :param address: router address 

640 :param timeout: seconds 

641 :param database: the data base name to get routing table for 

642 :param address: the address by which the client initially contacted the server as a hint for inclusion in the returned routing table. 

643 

644 :return: list of routing records or 

645 None if no connection could be established 

646 :raise ServiceUnavailable: if the server does not support routing or 

647 if routing support is broken 

648 """ 

649 

650 # The name of system database is fixed and named as "system". 

651 # It cannot be changed for a single instance or a cluster. (We can reliably assume that the system db exists on each instance.) 

652 # 

653 # Database name is NOT case sensitive. 

654 # 

655 # Each cluster member will host the exact same databases. For example, if a cluster member A has databases "foo" and 

656 # "system", then all other members in the cluster should also have and only have "foo" and "system". 

657 # However at a certain time, the cluster members may or may not up-to-date, as a result, cluster members may contain different databases. 

658 # 

659 # Maintain a routing table for each database. 

660 # 

661 # Default database is named "neo4j", (this can be renamed on the Neo4j server). 

662 # 

663 # Any core member in a cluster can provide a routing table for any database inside the cluster. 

664 # The seed_url can be used to find all databases in the cluster. 

665 # 

666 # If the driver failed to refresh routing table with all known routers, then the driver should retry a few times before it raises a ServiceUnavailable. 

667 # 

668 # A valid routing table should at least have one router and one reader. 

669 # 

670 # To prevent the routing tables from growing infinitely. 

671 # Stale/Aged routing tables is removed when there is a failure to obtain a routing table. 

672 # Remove a routing table if it have been aged, timeout = TTL + RoutingConfig.routing_table_purge_delay 

673

702 cx.pull(on_success=metadata.update, on_records=records.extend) 

703 cx.send_all() 

704 cx.fetch_all() 

705 routing_info = [dict(zip(metadata.get("fields", ()), values)) for values in records] 

706 log.debug("[#%04X] S: <ROUTING> info=%r", cx.local_port, routing_info) 

707 return routing_info 

708 except BoltRoutingError as error: 

709 raise ServiceUnavailable(*error.args) 

710 except ServiceUnavailable: 

711 self.deactivate(address=address) 

712 return None

714 def fetch_routing_table(self, *, address, timeout, database): 

715 """ Fetch a routing table from a given router address. 

716 

717 :param address: router address 

718 :param timeout: seconds 

719 :param database: the database name 

720 :type: str 

721 

722 :return: a new RoutingTable instance or None if the given router is 

723 currently unable to provide routing information 

724 

725 :raise neo4j.exceptions.ServiceUnavailable: if no writers are available 

726 :raise neo4j._exceptions.BoltProtocolError: if the routing information received is unusable 

727 """ 

728 new_routing_info = self.fetch_routing_info(address=address, timeout=timeout, database=database) 

729 if new_routing_info is None: 

730 return None 

731 elif not new_routing_info: 

732 raise BoltRoutingError("Invalid routing table", address) 

733 else: 

734 servers = new_routing_info[0]["servers"] 

735 ttl = new_routing_info[0]["ttl"] 

736 new_routing_table = RoutingTable.parse_routing_info(database=database, servers=servers, ttl=ttl) 

737 

738 # Parse routing info and count the number of each type of server 

739 num_routers = len(new_routing_table.routers) 

740 num_readers = len(new_routing_table.readers) 

741 

742 # num_writers = len(new_routing_table.writers) 

743 # If no writers are available. This likely indicates a temporary state, 

744 # such as leader switching, so we should not signal an error. 

745 

746 # No routers 

747 if num_routers == 0: 

748 raise BoltRoutingError("No routing servers returned from server", address) 

749 

750 # No readers 

751 if num_readers == 0: 

752 raise BoltRoutingError("No read servers returned from server", address) 

753 

754 # At least one of each is fine, so return this table 

755 return new_routing_table

809 def ensure_routing_table_is_fresh(self, *, access_mode, database): 

810 """ Update the routing table if stale. 

811 

812 This method performs two freshness checks, before and after acquiring 

813 the refresh lock. If the routing table is already fresh on entry, the 

814 method exits immediately; otherwise, the refresh lock is acquired and 

815 the second freshness check that follows determines whether an update 

816 is still required. 

817 

818 This method is thread-safe. 

819 

820 :return: `True` if an update was required, `False` otherwise. 

821 """ 

822 from neo4j.api import READ_ACCESS 

823 if self.routing_tables[database].is_fresh(readonly=(access_mode == READ_ACCESS)): 

824 # Readers are fresh. 

825 return False 

826 with self.refresh_lock: 

827 

828 self.update_routing_table(database=database) 

829 self.update_connection_pool(database=database) 

830 

831 for database in list(self.routing_tables.keys()): 

832 # Remove unused databases in the routing table 

833 # Remove the routing table after a timeout = TTL + 30s 

834 log.debug("[#0000] C: <ROUTING AGED> database=%s", database) 

835 if self.routing_tables[database].should_be_purged_from_memory() and database != self.workspace_config.database: 

836 del self.routing_tables[database] 

837 

838 return True

840 def _select_address(self, *, access_mode, database): 

841 from neo4j.api import READ_ACCESS 

842 """ Selects the address with the fewest in-use connections. 

843 """ 

844 self.create_routing_table(database) 

845 self.ensure_routing_table_is_fresh(access_mode=access_mode, database=database) 

846 log.debug("[#0000] C: <ROUTING TABLE ENSURE FRESH> %r", self.routing_tables) 

847 if access_mode == READ_ACCESS: 

848 addresses = self.routing_tables[database].readers 

849 else: 

850 addresses = self.routing_tables[database].writers 

851 addresses_by_usage = {} 

852 for address in addresses: 

853 addresses_by_usage.setdefault(self.in_use_connection_count(address), []).append(address) 

854 if not addresses_by_usage: 

855 if access_mode == READ_ACCESS: 

856 raise ReadServiceUnavailable("No read service currently available") 

857 else: 

858 raise WriteServiceUnavailable("No write service currently available") 

859 return choice(addresses_by_usage[min(addresses_by_usage)])

861 def acquire(self, access_mode=None, timeout=None, database=None): 

862 if access_mode not in (WRITE_ACCESS, READ_ACCESS): 

863 raise ClientError("Non valid 'access_mode'; {}".format(access_mode)) 

864 if not timeout: 

865 raise ClientError("'timeout' must be a float larger than 0; {}".format(timeout)) 

866 

867 from neo4j.api import check_access_mode 

868 access_mode = check_access_mode(access_mode) 

869 while True: 

870 try: 

871 # Get an address for a connection that have the fewest in-use connections. 

872 address = self._select_address(access_mode=access_mode, database=database) 

873 log.debug("[#0000] C: <ACQUIRE ADDRESS> database=%r address=%r", database, address) 

874 except (ReadServiceUnavailable, WriteServiceUnavailable) as err: 

875 raise SessionExpired("Failed to obtain connection towards '%s' server." % access_mode) from err 

876 try: 

877 connection = self._acquire(address, timeout=timeout) # should always be a resolved address 

878 except ServiceUnavailable: 

879 self.deactivate(address=address) 

880 else: 

881 return connection

883 def deactivate(self, address): 

884 """ Deactivate an address from the connection pool, 

885 if present, remove from the routing table and also closing 

886 all idle connections to that address. 

887 """ 

888 log.debug("[#0000] C: <ROUTING> Deactivating address %r", address) 

889 # We use `discard` instead of `remove` here since the former 

890 # will not fail if the address has already been removed. 

891 for database in self.routing_tables.keys(): 

892 self.routing_tables[database].routers.discard(address) 

893 self.routing_tables[database].readers.discard(address) 

894 self.routing_tables[database].writers.discard(address) 

895 log.debug("[#0000] C: <ROUTING> table=%r", self.routing_tables) 

896 super(Neo4jPool, self).deactivate(address)

907def _connect(resolved_address, timeout, keep_alive): 

908 """ 

909 

910 :param resolved_address: 

911 :param timeout: seconds 

912 :param keep_alive: True or False 

913 :return: socket object 

914 """ 

915 

916 s = None # The socket 

917 

918 try: 

919 if len(resolved_address) == 2: 

920 s = socket(AF_INET) 

921 elif len(resolved_address) == 4: 

922 s = socket(AF_INET6) 

923 else: 

924 raise ValueError("Unsupported address {!r}".format(resolved_address)) 

925 t = s.gettimeout() 

926 if timeout: 

927 s.settimeout(timeout) 

928 log.debug("[#0000] C: <OPEN> %s", resolved_address) 

929 s.connect(resolved_address) 

930 s.settimeout(t) 

931 keep_alive = 1 if keep_alive else 0 

932 s.setsockopt(SOL_SOCKET, SO_KEEPALIVE, keep_alive) 

933 except SocketTimeout: 

934 log.debug("[#0000] C: <TIMEOUT> %s", resolved_address) 

935 log.debug("[#0000] C: <CLOSE> %s", resolved_address) 

936 s.close() 

937 raise ServiceUnavailable("Timed out trying to establish connection to {!r}".format(resolved_address)) 

938 except OSError as error: 

939 log.debug("[#0000] C: <ERROR> %s %s", type(error).__name__, 

940 " ".join(map(repr, error.args))) 

941 log.debug("[#0000] C: <CLOSE> %s", resolved_address) 

942 s.close() 

943 raise ServiceUnavailable("Failed to establish connection to {!r} (reason {})".format(resolved_address, error)) 

944 else: 

945 return s

948def _secure(s, host, ssl_context): 

949 local_port = s.getsockname()[1] 

950 # Secure the connection if an SSL context has been provided 

951 if ssl_context: 

952 log.debug("[#%04X] C: <SECURE> %s", local_port, host) 

953 try: 

954 sni_host = host if HAS_SNI and host else None 

955 s = ssl_context.wrap_socket(s, server_hostname=sni_host) 

956 except (SSLError, OSError) as cause: 

957 s.close() 

958 error = BoltSecurityError(message="Failed to establish encrypted connection.", address=(host, local_port)) 

959 error.__cause__ = cause 

960 raise error 

961 else: 

962 # Check that the server provides a certificate 

963 der_encoded_server_certificate = s.getpeercert(binary_form=True) 

964 if der_encoded_server_certificate is None: 

965 s.close() 

966 raise BoltProtocolError("When using an encrypted socket, the server should always provide a certificate", address=(host, local_port)) 

967 return s

970def _handshake(s, resolved_address): 

971 """ 

972 

973 :param s: Socket 

974 :param resolved_address: 

975 

976 :return: (socket, version, client_handshake, server_response_data) 

977 """ 

978 local_port = s.getsockname()[1] 

979 

980 # TODO: Optimize logging code 

981 handshake = Bolt.get_handshake() 

982 import struct 

983 handshake = struct.unpack(">16B", handshake) 

984 handshake = [handshake[i:i + 4] for i in range(0, len(handshake), 4)] 

985 

986 supported_versions = [("0x%02X%02X%02X%02X" % (vx[0], vx[1], vx[2], vx[3])) for vx in handshake] 

987 

988 log.debug("[#%04X] C: <MAGIC> 0x%08X", local_port, int.from_bytes(Bolt.MAGIC_PREAMBLE, byteorder="big")) 

989 log.debug("[#%04X] C: <HANDSHAKE> %s %s %s %s", local_port, *supported_versions) 

990 

991 data = Bolt.MAGIC_PREAMBLE + Bolt.get_handshake() 

992 s.sendall(data) 

993 

994 # Handle the handshake response 

995 ready_to_read = False 

996 while not ready_to_read: 

997 ready_to_read, _, _ = select((s,), (), (), 1) 

998 try: 

999 data = s.recv(4) 

1000 except OSError: 

1001 raise ServiceUnavailable("Failed to read any data from server {!r} " 

1002 "after connected".format(resolved_address)) 

1003 data_size = len(data) 

1004 if data_size == 0: 

1005 # If no data is returned after a successful select 

1006 # response, the server has closed the connection 

1007 log.debug("[#%04X] S: <CLOSE>", local_port) 

1008 s.close() 

1009 raise BoltHandshakeError("Connection to {address} closed without handshake response".format(address=resolved_address), address=resolved_address, request_data=handshake, response_data=None) 

1010 if data_size != 4: 

1011 # Some garbled data has been received 

1012 log.debug("[#%04X] S: @*#!", local_port) 

1013 s.close() 

1014 raise BoltProtocolError("Expected four byte Bolt handshake response from %r, received %r instead; check for incorrect port number" % (resolved_address, data), address=resolved_address) 

1015 elif data == b"HTTP": 

1016 log.debug("[#%04X] S: <CLOSE>", local_port) 

1017 s.close() 

1018 raise ServiceUnavailable("Cannot to connect to Bolt service on {!r} " 

1019 "(looks like HTTP)".format(resolved_address)) 

1020 agreed_version = data[-1], data[-2] 

1021 log.debug("[#%04X] S: <HANDSHAKE> 0x%06X%02X", local_port, agreed_version[1], agreed_version[0]) 

1022 return s, agreed_version, handshake, data

1025def connect(address, *, timeout, custom_resolver, ssl_context, keep_alive): 

1026 """ Connect and perform a handshake and return a valid Connection object, 

1027 assuming a protocol version can be agreed. 

1028 """ 

1029 last_error = None 

1030 # Establish a connection to the host and port specified 

1031 # Catches refused connections see: 

1032 # https://docs.python.org/2/library/errno.html 

1033 log.debug("[#0000] C: <RESOLVE> %s", address) 

1034 

1035 for resolved_address in Address(address).resolve(resolver=custom_resolver): 

1036 s = None 

1037 try: 

1038 host = address[0] 

1039 s = _connect(resolved_address, timeout, keep_alive) 

1040 s = _secure(s, host, ssl_context) 

1041 return _handshake(s, address) 

1042 except Exception as error: 

1043 if s: 

1044 s.close() 

1045 last_error = error 

1046 if last_error is None: 

1047 raise ServiceUnavailable("Failed to resolve addresses for %s" % address) 

1048 else: 

1049 raise last_error
