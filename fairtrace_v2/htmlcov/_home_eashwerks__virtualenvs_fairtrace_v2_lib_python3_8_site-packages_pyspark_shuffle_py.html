150class ExternalMerger(Merger): 

151 

152 """ 

153 External merger will dump the aggregated data into disks when 

154 memory usage goes above the limit, then merge them together. 

155 

156 This class works as follows: 

157 

158 - It repeatedly combine the items and save them in one dict in 

159 memory. 

160 

161 - When the used memory goes above memory limit, it will split 

162 the combined data into partitions by hash code, dump them 

163 into disk, one file per partition. 

164 

165 - Then it goes through the rest of the iterator, combine items 

166 into different dict by hash. Until the used memory goes over 

167 memory limit, it dump all the dicts into disks, one file per 

168 dict. Repeat this again until combine all the items. 

169 

170 - Before return any items, it will load each partition and 

171 combine them separately. Yield them before loading next 

172 partition. 

173 

174 - During loading a partition, if the memory goes over limit, 

175 it will partition the loaded data and dump them into disks 

176 and load them partition by partition again. 

177 

178 `data` and `pdata` are used to hold the merged items in memory. 

179 At first, all the data are merged into `data`. Once the used 

180 memory goes over limit, the items in `data` are dumped into 

181 disks, `data` will be cleared, all rest of items will be merged 

182 into `pdata` and then dumped into disks. Before returning, all 

183 the items in `pdata` will be dumped into disks. 

184 

185 Finally, if any items were spilled into disks, each partition 

186 will be merged into `data` and be yielded, then cleared. 

187 

188 Examples 

189 -------- 

190 >>> agg = SimpleAggregator(lambda x, y: x + y) 

191 >>> merger = ExternalMerger(agg, 10) 

192 >>> N = 10000 

193 >>> merger.mergeValues(zip(range(N), range(N))) 

194 >>> assert merger.spills > 0 

195 >>> sum(v for k,v in merger.items()) 

196 49995000 

197 

198 >>> merger = ExternalMerger(agg, 10) 

199 >>> merger.mergeCombiners(zip(range(N), range(N))) 

200 >>> assert merger.spills > 0 

201 >>> sum(v for k,v in merger.items()) 

202 49995000 

203 """ 

204 

205 # the max total partitions created recursively 

206 MAX_TOTAL_PARTITIONS = 4096

208 def __init__( 

209 self, 

210 aggregator, 

211 memory_limit=512, 

212 serializer=None, 

213 localdirs=None, 

214 scale=1, 

215 partitions=59, 

216 batch=1000, 

217 ): 

218 Merger.__init__(self, aggregator) 

219 self.memory_limit = memory_limit 

220 self.serializer = _compressed_serializer(serializer) 

221 self.localdirs = localdirs or _get_local_dirs(str(id(self))) 

222 # number of partitions when spill data into disks 

223 self.partitions = partitions 

224 # check the memory after # of items merged 

225 self.batch = batch 

226 # scale is used to scale down the hash of key for recursive hash map 

227 self.scale = scale 

228 # un-partitioned merged data 

229 self.data = {} 

230 # partitioned merged data, list of dicts 

231 self.pdata = [] 

232 # number of chunks dumped into disks 

233 self.spills = 0 

234 # randomize the hash of key, id(o) is the address of o (aligned by 8) 

235 self._seed = id(self) + 7

237 def _get_spill_dir(self, n): 

238 """Choose one directory for spill by number n""" 

239 return os.path.join(self.localdirs[n % len(self.localdirs)], str(n))

241 def _next_limit(self): 

242 """ 

243 Return the next memory limit. If the memory is not released 

244 after spilling, it will dump the data only when the used memory 

245 starts to increase. 

246 """ 

247 return max(self.memory_limit, get_used_memory() * 1.05)

249 def mergeValues(self, iterator): 

250 """Combine the items by creator and combiner""" 

251 # speedup attribute lookup 

252 creator, comb = self.agg.createCombiner, self.agg.mergeValue 

253 c, data, pdata, hfun, batch = 0, self.data, self.pdata, self._partition, self.batch 

254 limit = self.memory_limit 

255 

256 for k, v in iterator: 

257 d = pdata[hfun(k)] if pdata else data 

258 d[k] = comb(d[k], v) if k in d else creator(v) 

259 

260 c += 1 

261 if c >= batch: 

262 if get_used_memory() >= limit: 

263 self._spill() 

264 limit = self._next_limit() 

265 batch /= 2 

266 c = 0 

267 else: 

268 batch *= 1.5 

269 

270 if get_used_memory() >= limit: 

271 self._spill()

273 def _partition(self, key): 

274 """Return the partition for key""" 

275 return hash((key, self._seed)) % self.partitions 

276 

277 def _object_size(self, obj): 

278 """How much of memory for this obj, assume that all the objects 

279 consume similar bytes of memory 

280 """ 

281 return 1

283 def mergeCombiners(self, iterator, limit=None): 

284 """Merge (K,V) pair by mergeCombiner""" 

285 if limit is None: 

286 limit = self.memory_limit 

287 # speedup attribute lookup 

288 comb, hfun, objsize = self.agg.mergeCombiners, self._partition, self._object_size 

289 c, data, pdata, batch = 0, self.data, self.pdata, self.batch 

290 for k, v in iterator: 

291 d = pdata[hfun(k)] if pdata else data 

292 d[k] = comb(d[k], v) if k in d else v 

293 if not limit: 

294 continue 

295 

296 c += objsize(v) 

297 if c > batch: 

298 if get_used_memory() > limit: 

299 self._spill() 

300 limit = self._next_limit() 

301 batch /= 2 

302 c = 0 

303 else: 

304 batch *= 1.5 

305 

306 if limit and get_used_memory() >= limit: 

307 self._spill()

309 def _spill(self): 

310 """ 

311 dump already partitioned data into disks. 

312 

313 It will dump the data in batch for better performance. 

314 """ 

315 global MemoryBytesSpilled, DiskBytesSpilled 

316 path = self._get_spill_dir(self.spills) 

317 if not os.path.exists(path): 

318 os.makedirs(path) 

319 

320 used_memory = get_used_memory() 

321 if not self.pdata: 

322 # The data has not been partitioned, it will iterator the 

323 # dataset once, write them into different files, has no 

324 # additional memory. It only called when the memory goes 

325 # above limit at the first time. 

326 

327 # open all the files for writing 

328 streams = [open(os.path.join(path, str(i)), "wb") for i in range(self.partitions)] 

329 

330 for k, v in self.data.items(): 

331 h = self._partition(k) 

332 # put one item in batch, make it compatible with load_stream 

333 # it will increase the memory if dump them in batch 

334 self.serializer.dump_stream([(k, v)], streams[h]) 

335 

336 for s in streams: 

337 DiskBytesSpilled += s.tell() 

338 s.close() 

339 

340 self.data.clear() 

341 self.pdata.extend([{} for i in range(self.partitions)]) 

342 

343 else: 

344 for i in range(self.partitions): 

345 p = os.path.join(path, str(i)) 

346 with open(p, "wb") as f: 

347 # dump items in batch 

348 self.serializer.dump_stream(iter(self.pdata[i].items()), f) 

349 self.pdata[i].clear() 

350 DiskBytesSpilled += os.path.getsize(p) 

351 

352 self.spills += 1 

353 gc.collect() # release the memory as much as possible 

354 MemoryBytesSpilled += max(used_memory - get_used_memory(), 0) << 20 

355 

356 def items(self): 

357 """Return all merged items as iterator""" 

358 if not self.pdata and not self.spills: 

359 return iter(self.data.items()) 

360 return self._external_items()

362 def _external_items(self): 

363 """Return all partitioned items as iterator""" 

364 assert not self.data 

365 if any(self.pdata): 

366 self._spill() 

367 # disable partitioning and spilling when merge combiners from disk 

368 self.pdata = [] 

369 

370 try: 

371 for i in range(self.partitions): 

372 for v in self._merged_items(i): 

373 yield v 

374 self.data.clear() 

375 

376 # remove the merged partition 

377 for j in range(self.spills): 

378 path = self._get_spill_dir(j) 

379 os.remove(os.path.join(path, str(i))) 

380 finally: 

381 self._cleanup() 

382 

383 def _merged_items(self, index): 

384 self.data = {} 

385 limit = self._next_limit() 

386 for j in range(self.spills): 

387 path = self._get_spill_dir(j) 

388 p = os.path.join(path, str(index)) 

389 # do not check memory during merging 

390 with open(p, "rb") as f: 

391 self.mergeCombiners(self.serializer.load_stream(f), 0) 

392 

393 # limit the total partitions 

394 if ( 

395 self.scale * self.partitions < self.MAX_TOTAL_PARTITIONS 

396 and j < self.spills - 1 

397 and get_used_memory() > limit 

398 ): 

399 self.data.clear() # will read from disk again 

400 gc.collect() # release the memory as much as possible 

401 return self._recursive_merged_items(index) 

402 

403 return self.data.items()

405 def _recursive_merged_items(self, index): 

406 """ 

407 merge the partitioned items and return the as iterator 

408 

409 If one partition can not be fit in memory, then them will be 

410 partitioned and merged recursively. 

411 """ 

412 subdirs = [os.path.join(d, "parts", str(index)) for d in self.localdirs] 

413 m = ExternalMerger( 

414 self.agg, 

415 self.memory_limit, 

416 self.serializer, 

417 subdirs, 

418 self.scale * self.partitions, 

419 self.partitions, 

420 self.batch, 

421 ) 

422 m.pdata = [{} for _ in range(self.partitions)] 

423 limit = self._next_limit() 

424 

425 for j in range(self.spills): 

426 path = self._get_spill_dir(j) 

427 p = os.path.join(path, str(index)) 

428 with open(p, "rb") as f: 

429 m.mergeCombiners(self.serializer.load_stream(f), 0) 

430 

431 if get_used_memory() > limit: 

432 m._spill() 

433 limit = self._next_limit() 

434 

435 return m._external_items()

438 """Clean up all the files in disks""" 

439 for d in self.localdirs: 

440 shutil.rmtree(d, True)

443class ExternalSorter: 

444 """ 

445 ExternalSorter will divide the elements into chunks, sort them in 

446 memory and dump them into disks, finally merge them back. 

447 

448 The spilling will only happen when the used memory goes above 

449 the limit. 

450 

451 Examples 

452 -------- 

453 >>> sorter = ExternalSorter(1) # 1M 

454 >>> import random 

455 >>> l = list(range(1024)) 

456 >>> random.shuffle(l) 

457 >>> sorted(l) == list(sorter.sorted(l)) 

458 True 

459 >>> sorted(l) == list(sorter.sorted(l, key=lambda x: -x, reverse=True)) 

460 True 

461 """ 

462 

463 def __init__(self, memory_limit, serializer=None): 

464 self.memory_limit = memory_limit 

465 self.local_dirs = _get_local_dirs("sort") 

466 self.serializer = _compressed_serializer(serializer) 

467 

468 def _get_path(self, n): 

469 """Choose one directory for spill by number n""" 

470 d = self.local_dirs[n % len(self.local_dirs)] 

471 if not os.path.exists(d): 

472 os.makedirs(d) 

473 return os.path.join(d, str(n)) 

474 

475 def _next_limit(self): 

476 """ 

477 Return the next memory limit. If the memory is not released 

478 after spilling, it will dump the data only when the used memory 

479 starts to increase. 

480 """ 

481 return max(self.memory_limit, get_used_memory() * 1.05) 

482 

483 def sorted(self, iterator, key=None, reverse=False): 

484 """ 

485 Sort the elements in iterator, do external sort when the memory 

486 goes above the limit. 

487 """ 

488 global MemoryBytesSpilled, DiskBytesSpilled 

489 batch, limit = 100, self._next_limit() 

490 chunks, current_chunk = [], [] 

491 iterator = iter(iterator) 

492 while True: 

493 # pick elements in batch 

494 chunk = list(itertools.islice(iterator, batch)) 

495 current_chunk.extend(chunk) 

496 if len(chunk) < batch: 

497 break 

498 

499 used_memory = get_used_memory() 

500 if used_memory > limit: 

501 # sort them inplace will save memory 

502 current_chunk.sort(key=key, reverse=reverse) 

503 path = self._get_path(len(chunks)) 

504 with open(path, "wb") as f: 

505 self.serializer.dump_stream(current_chunk, f) 

506 

507 def load(f): 

508 for v in self.serializer.load_stream(f): 

509 yield v 

510 # close the file explicit once we consume all the items 

511 # to avoid ResourceWarning in Python3 

512 f.close() 

513 

514 chunks.append(load(open(path, "rb"))) 

515 current_chunk = [] 

516 MemoryBytesSpilled += max(used_memory - get_used_memory(), 0) << 20 

517 DiskBytesSpilled += os.path.getsize(path) 

518 os.unlink(path) # data will be deleted after close 

519 

520 elif not chunks: 

521 batch = min(int(batch * 1.5), 10000) 

522 

523 current_chunk.sort(key=key, reverse=reverse) 

524 if not chunks: 

525 return current_chunk 

526 

527 if current_chunk: 

528 chunks.append(iter(current_chunk)) 

529 

530 return heapq.merge(*chunks, key=key, reverse=reverse)

534 """

572 else:

634 MemoryBytesSpilled += max(used_memory - get_used_memory(), 0) << 20

698class ExternalGroupBy(ExternalMerger): 

699 

700 """ 

701 Group by the items by key. If any partition of them can not been 

702 hold in memory, it will do sort based group by. 

703 

704 This class works as follows: 

705 

706 - It repeatedly group the items by key and save them in one dict in 

707 memory. 

708 

709 - When the used memory goes above memory limit, it will split 

710 the combined data into partitions by hash code, dump them 

711 into disk, one file per partition. If the number of keys 

712 in one partitions is smaller than 1000, it will sort them 

713 by key before dumping into disk. 

714 

715 - Then it goes through the rest of the iterator, group items 

716 by key into different dict by hash. Until the used memory goes over 

717 memory limit, it dump all the dicts into disks, one file per 

718 dict. Repeat this again until combine all the items. It 

719 also will try to sort the items by key in each partition 

720 before dumping into disks. 

721 

722 - It will yield the grouped items partitions by partitions. 

723 If the data in one partitions can be hold in memory, then it 

724 will load and combine them in memory and yield. 

725 

726 - If the dataset in one partition cannot be hold in memory, 

727 it will sort them first. If all the files are already sorted, 

728 it merge them by heap.merge(), so it will do external sort 

729 for all the files. 

730 

731 - After sorting, `GroupByKey` class will put all the continuous 

732 items with the same key as a group, yield the values as 

733 an iterator. 

734 """ 

735 

736 SORT_KEY_LIMIT = 1000 

737 

738 def flattened_serializer(self): 

739 assert isinstance(self.serializer, BatchedSerializer) 

740 ser = self.serializer 

741 return FlattenedValuesSerializer(ser, 20) 

742 

743 def _object_size(self, obj): 

744 return len(obj) 

745 

746 def _spill(self): 

747 """ 

748 dump already partitioned data into disks. 

749 """ 

750 global MemoryBytesSpilled, DiskBytesSpilled 

751 path = self._get_spill_dir(self.spills) 

752 if not os.path.exists(path): 

753 os.makedirs(path) 

754 

755 used_memory = get_used_memory() 

756 if not self.pdata: 

757 # The data has not been partitioned, it will iterator the 

758 # data once, write them into different files, has no 

759 # additional memory. It only called when the memory goes 

760 # above limit at the first time. 

761 

762 # open all the files for writing 

763 streams = [open(os.path.join(path, str(i)), "wb") for i in range(self.partitions)] 

764 

765 # If the number of keys is small, then the overhead of sort is small 

766 # sort them before dumping into disks 

767 self._sorted = len(self.data) < self.SORT_KEY_LIMIT 

768 if self._sorted: 

769 self.serializer = self.flattened_serializer() 

770 for k in sorted(self.data.keys()): 

771 h = self._partition(k) 

772 self.serializer.dump_stream([(k, self.data[k])], streams[h]) 

773 else: 

774 for k, v in self.data.items(): 

775 h = self._partition(k) 

776 self.serializer.dump_stream([(k, v)], streams[h]) 

777 

778 for s in streams: 

779 DiskBytesSpilled += s.tell() 

780 s.close() 

781 

782 self.data.clear() 

783 # self.pdata is cached in `mergeValues` and `mergeCombiners` 

784 self.pdata.extend([{} for i in range(self.partitions)]) 

785 

786 else: 

787 for i in range(self.partitions): 

788 p = os.path.join(path, str(i)) 

789 with open(p, "wb") as f: 

790 # dump items in batch 

791 if self._sorted: 

792 # sort by key only (stable) 

793 sorted_items = sorted(self.pdata[i].items(), key=operator.itemgetter(0)) 

794 self.serializer.dump_stream(sorted_items, f) 

795 else: 

796 self.serializer.dump_stream(self.pdata[i].items(), f) 

797 self.pdata[i].clear() 

798 DiskBytesSpilled += os.path.getsize(p) 

799 

800 self.spills += 1 

801 gc.collect() # release the memory as much as possible 

802 MemoryBytesSpilled += max(used_memory - get_used_memory(), 0) << 20 

803 

804 def _merged_items(self, index): 

805 size = sum( 

806 os.path.getsize(os.path.join(self._get_spill_dir(j), str(index))) 

807 for j in range(self.spills) 

808 ) 

809 # if the memory can not hold all the partition, 

810 # then use sort based merge. Because of compression, 

811 # the data on disks will be much smaller than needed memory 

812 if size >= self.memory_limit << 17: # * 1M / 8 

813 return self._merge_sorted_items(index) 

814 

815 self.data = {} 

816 for j in range(self.spills): 

817 path = self._get_spill_dir(j) 

818 p = os.path.join(path, str(index)) 

819 # do not check memory during merging 

820 with open(p, "rb") as f: 

821 self.mergeCombiners(self.serializer.load_stream(f), 0) 

822 return self.data.items() 

823 

824 def _merge_sorted_items(self, index): 

825 """load a partition from disk, then sort and group by key""" 

826 

827 def load_partition(j): 

828 path = self._get_spill_dir(j) 

829 p = os.path.join(path, str(index)) 

830 with open(p, "rb", 65536) as f: 

831 for v in self.serializer.load_stream(f): 

832 yield v 

833 

834 disk_items = [load_partition(j) for j in range(self.spills)] 

835 

836 if self._sorted: 

837 # all the partitions are already sorted 

838 sorted_items = heapq.merge(*disk_items, key=operator.itemgetter(0)) 

839 

840 else: 

841 # Flatten the combined values, so it will not consume huge 

842 # memory during merging sort. 

843 ser = self.flattened_serializer() 

844 sorter = ExternalSorter(self.memory_limit, ser) 

845 sorted_items = sorter.sorted(itertools.chain(*disk_items), key=operator.itemgetter(0)) 

846 return ((k, vs) for k, vs in GroupByKey(sorted_items))
