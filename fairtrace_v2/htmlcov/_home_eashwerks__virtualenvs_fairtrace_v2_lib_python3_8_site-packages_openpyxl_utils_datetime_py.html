38def from_ISO8601(formatted_string): 

39 """Convert from a timestamp string to a datetime object. According to 

40 18.17.4 in the specification the following ISO 8601 formats are 

41 supported. 

42 

43 Dates B.1.1 and B.2.1 

44 Times B.1.2 and B.2.2 

45 Datetimes B.1.3 and B.2.3 

46 

47 There is no concept of timedeltas in the specification, but Excel 

48 writes them (in strict OOXML mode), so these are also understood. 

49 """ 

50 if not formatted_string: 

51 return None 

52 

53 match = ISO_REGEX.match(formatted_string) 

54 if match and any(match.groups()): 

55 parts = match.groupdict(0) 

56 for key in ["year", "month", "day", "hour", "minute", "second"]: 

57 if parts[key]: 

58 parts[key] = int(parts[key]) 

59 

60 if parts["microsecond"]: 

61 parts["microsecond"] = int(float(parts['microsecond']) * 1_000_000) 

62 

63 if not parts["date"]: 

64 dt = datetime.time(parts['hour'], parts['minute'], parts['second'], parts["microsecond"]) 

65 elif not parts["time"]: 

66 dt = datetime.date(parts['year'], parts['month'], parts['day']) 

67 else: 

68 del parts["time"] 

69 del parts["date"] 

70 dt = datetime.datetime(**parts) 

71 return dt 

72 

73 match = ISO_DURATION.match(formatted_string) 

74 if match and any(match.groups()): 

75 parts = match.groupdict(0) 

76 for key, val in parts.items(): 

77 if val: 

78 parts[key] = float(val) 

79 return datetime.timedelta(**parts) 

80 

81 raise ValueError("Invalid datetime value {}".format(formatted_string))

84def to_excel(dt, epoch=WINDOWS_EPOCH): 

85 """Convert Python datetime to Excel serial""" 

86 if isinstance(dt, datetime.time): 

87 return time_to_days(dt) 

88 if isinstance(dt, datetime.timedelta): 

89 return timedelta_to_days(dt) 

90 if isnan(dt.year): # Pandas supports Not a Date 

91 return 

92 

93 if not hasattr(dt, "date"): 

94 dt = datetime.datetime.combine(dt, datetime.time()) 

95 

96 # rebase on epoch and adjust for < 1900-03-01 

97 days = (dt - epoch).days 

98 if 0 < days <= 60 and epoch == WINDOWS_EPOCH: 

99 days -= 1 

100 return days + time_to_days(dt)

103def from_excel(value, epoch=WINDOWS_EPOCH, timedelta=False): 

104 """Convert Excel serial to Python datetime""" 

105 if value is None: 

106 return 

107 

108 if timedelta: 

109 td = datetime.timedelta(days=value) 

110 if td.microseconds: 

111 # round to millisecond precision 

112 td = datetime.timedelta(seconds=td.total_seconds() // 1, 

113 microseconds=round(td.microseconds, -3)) 

114 return td 

115 

116 day, fraction = divmod(value, 1) 

117 diff = datetime.timedelta(milliseconds=round(fraction * SECS_PER_DAY * 1000)) 

118 if 0 <= value < 1 and diff.days == 0: 

119 return days_to_time(diff) 

120 if 0 < value < 60 and epoch == WINDOWS_EPOCH: 

121 day += 1 

122 return epoch + datetime.timedelta(days=day) + diff

129 """Convert a time value to fractions of day"""
