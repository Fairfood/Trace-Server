405 def _wrapped(self) -> "UserDefinedFunctionLike": 

406 """ 

407 Wrap this udf with a function and attach docstring from func 

408 """ 

409 

410 # It is possible for a callable instance without __name__ attribute or/and 

411 # __module__ attribute to be wrapped here. For example, functools.partial. In this case, 

412 # we should avoid wrapping the attributes from the wrapped function to the wrapper 

413 # function. So, we take out these attribute names from the default names to set and 

414 # then manually assign it after being wrapped. 

415 assignments = tuple( 

416 a for a in functools.WRAPPER_ASSIGNMENTS if a != "__name__" and a != "__module__" 

417 ) 

418 

419 @functools.wraps(self.func, assigned=assignments) 

420 def wrapper(*args: "ColumnOrName") -> Column: 

421 return self(*args) 

422 

423 wrapper.__name__ = self._name 

424 wrapper.__module__ = ( 

425 self.func.__module__ 

426 if hasattr(self.func, "__module__") 

427 else self.func.__class__.__module__ 

428 )

430 wrapper.func = self.func # type: ignore[attr-defined] 

431 wrapper.returnType = self.returnType # type: ignore[attr-defined] 

432 wrapper.evalType = self.evalType # type: ignore[attr-defined] 

433 wrapper.deterministic = self.deterministic # type: ignore[attr-defined] 

434 wrapper.asNondeterministic = functools.wraps( # type: ignore[attr-defined] 

435 self.asNondeterministic 

436 )(lambda: self.asNondeterministic()._wrapped()) 

437 wrapper._unwrapped = self # type: ignore[attr-defined]

458 .. versionadded:: 1.3.1

464 def register( 

465 self, 

466 name: str, 

467 f: Union[Callable[..., Any], "UserDefinedFunctionLike"], 

468 returnType: Optional["DataTypeOrString"] = None, 

469 ) -> "UserDefinedFunctionLike": 

470 """Register a Python function (including lambda function) or a user-defined function 

471 as a SQL function. 

472 

473 .. versionadded:: 1.3.1 

474 

475 .. versionchanged:: 3.4.0 

476 Supports Spark Connect. 

477 

478 Parameters 

479 ---------- 

480 name : str, 

481 name of the user-defined function in SQL statements. 

482 f : function, :meth:`pyspark.sql.functions.udf` or :meth:`pyspark.sql.functions.pandas_udf` 

483 a Python function, or a user-defined function. The user-defined function can 

484 be either row-at-a-time or vectorized. See :meth:`pyspark.sql.functions.udf` and 

485 :meth:`pyspark.sql.functions.pandas_udf`. 

486 returnType : :class:`pyspark.sql.types.DataType` or str, optional 

487 the return type of the registered user-defined function. The value can 

488 be either a :class:`pyspark.sql.types.DataType` object or a DDL-formatted type string. 

489 `returnType` can be optionally specified when `f` is a Python function but not 

490 when `f` is a user-defined function. Please see the examples below. 

491 

492 Returns 

493 ------- 

494 function 

495 a user-defined function 

496 

497 Notes 

498 ----- 

499 To register a nondeterministic Python function, users need to first build 

500 a nondeterministic user-defined function for the Python function and then register it 

501 as a SQL function. 

502 

503 Examples 

504 -------- 

505 1. When `f` is a Python function: 

506 

507 `returnType` defaults to string type and can be optionally specified. The produced 

508 object must match the specified type. In this case, this API works as if 

509 `register(name, f, returnType=StringType())`. 

510 

511 >>> strlen = spark.udf.register("stringLengthString", lambda x: len(x)) 

512 >>> spark.sql("SELECT stringLengthString('test')").collect() 

513 [Row(stringLengthString(test)='4')] 

514 

515 >>> spark.sql("SELECT 'foo' AS text").select(strlen("text")).collect() 

516 [Row(stringLengthString(text)='3')] 

517 

518 >>> from pyspark.sql.types import IntegerType 

519 >>> _ = spark.udf.register("stringLengthInt", lambda x: len(x), IntegerType()) 

520 >>> spark.sql("SELECT stringLengthInt('test')").collect() 

521 [Row(stringLengthInt(test)=4)] 

522 

523 >>> from pyspark.sql.types import IntegerType 

524 >>> _ = spark.udf.register("stringLengthInt", lambda x: len(x), IntegerType()) 

525 >>> spark.sql("SELECT stringLengthInt('test')").collect() 

526 [Row(stringLengthInt(test)=4)] 

527 

528 2. When `f` is a user-defined function (from Spark 2.3.0): 

529 

530 Spark uses the return type of the given user-defined function as the return type of 

531 the registered user-defined function. `returnType` should not be specified. 

532 In this case, this API works as if `register(name, f)`. 

533 

534 >>> from pyspark.sql.types import IntegerType 

535 >>> from pyspark.sql.functions import udf 

536 >>> slen = udf(lambda s: len(s), IntegerType()) 

537 >>> _ = spark.udf.register("slen", slen) 

538 >>> spark.sql("SELECT slen('test')").collect() 

539 [Row(slen(test)=4)] 

540 

541 >>> import random 

542 >>> from pyspark.sql.functions import udf 

543 >>> from pyspark.sql.types import IntegerType 

544 >>> random_udf = udf(lambda: random.randint(0, 100), IntegerType()).asNondeterministic() 

545 >>> new_random_udf = spark.udf.register("random_udf", random_udf) 

546 >>> spark.sql("SELECT random_udf()").collect() # doctest: +SKIP 

547 [Row(random_udf()=82)] 

548

550 >>> from pyspark.sql.functions import pandas_udf

607 def registerJavaFunction( 

608 self, 

609 name: str, 

610 javaClassName: str, 

611 returnType: Optional["DataTypeOrString"] = None, 

612 ) -> None: 

613 """Register a Java user-defined function as a SQL function. 

614 

615 In addition to a name and the function itself, the return type can be optionally specified. 

616 When the return type is not specified we would infer it via reflection. 

617 

618 .. versionadded:: 2.3.0 

619 

620 .. versionchanged:: 3.4.0 

621 Supports Spark Connect. 

622 

623 Parameters 

624 ---------- 

625 name : str 

626 name of the user-defined function 

627 javaClassName : str 

628 fully qualified name of java class 

629 returnType : :class:`pyspark.sql.types.DataType` or str, optional 

630 the return type of the registered Java function. The value can be either 

631 a :class:`pyspark.sql.types.DataType` object or a DDL-formatted type string. 

632 

633 Examples 

634 -------- 

635 >>> from pyspark.sql.types import IntegerType 

636 >>> spark.udf.registerJavaFunction( 

637 ... "javaStringLength", "test.org.apache.spark.sql.JavaStringLength", IntegerType()) 

638 ... # doctest: +SKIP 

639 >>> spark.sql("SELECT javaStringLength('test')").collect() # doctest: +SKIP 

640 [Row(javaStringLength(test)=4)] 

641 

642 >>> spark.udf.registerJavaFunction( 

643 ... "javaStringLength2", "test.org.apache.spark.sql.JavaStringLength") 

644 ... # doctest: +SKIP 

645 >>> spark.sql("SELECT javaStringLength2('test')").collect() # doctest: +SKIP 

646 [Row(javaStringLength2(test)=4)] 

647 

648 >>> spark.udf.registerJavaFunction( 

649 ... "javaStringLength3", "test.org.apache.spark.sql.JavaStringLength", "integer") 

650 ... # doctest: +SKIP 

651 >>> spark.sql("SELECT javaStringLength3('test')").collect() # doctest: +SKIP 

652 [Row(javaStringLength3(test)=4)] 

653 """ 

654 

655 jdt = None 

656 if returnType is not None: 

657 if not isinstance(returnType, DataType): 

658 returnType = _parse_datatype_string(returnType) 

659 jdt = self.sparkSession._jsparkSession.parseDataType(returnType.json()) 

660 self.sparkSession._jsparkSession.udf().registerJava(name, javaClassName, jdt)
