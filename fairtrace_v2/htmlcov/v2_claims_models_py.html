121 def create_new_version(self): 

122 """Most of the claims details except some minor changes cannot be 

123 updated, since it might already be attached to some batched. 

124 

125 If a major change needs to be done, a new version of the claim 

126 needs to be created. This model method creates a new version of 

127 the claims with the same details, criterion and fields. This new 

128 claim can be updated because it won't be attached to any 

129 batches. The old claims will then be disabled from showing up in 

130 the interface while adding. 

131 """ 

132 latest_cl = max( 

133 [i.version for i in Claim.objects.filter(reference=self.reference)] 

134 ) 

135 new_claim = copy.deepcopy(self) 

136 new_claim.id = None 

137 new_claim.version = latest_cl + 1 

138 new_claim.save() 

139 self.active = False 

140 self.latest = False 

141 self.save() 

142 for criterion in self.criteria.all(): 

143 ref = criterion.reference 

144 latest_cr = max( 

145 [i.version for i in Criterion.objects.filter(reference=ref)] 

146 ) 

147 new_criterion = copy.deepcopy(criterion) 

148 new_criterion.id = None 

149 new_criterion.version = latest_cr + 1 

150 new_criterion.claim = new_claim 

151 new_criterion.save() 

152 for field in criterion.fields.all(): 

153 new_field = copy.deepcopy(field) 

154 new_field.criterion = new_criterion 

155 new_field.id = None 

156 new_field.save() 

157 return new_claim

619 def verify(self): 

620 """Some claims can be automatically verified. 

621 

622 If a claims need not be verified by anyone, it's automatically 

623 marked as verified. If a claims is system verified, the 

624 particular business logic is computed and the claim is verified 

625 automatically by the system. System claims cannot be configured 

626 by Fairfood admin as of now. It needs to be added from the 

627 developer side. 

628 """ 

629 if ( 

630 not self.verifier 

631 and not self.claim.verifiers.exists() 

632 and self.claim.verified_by == constants.VERIFIED_BY_NONE 

633 ): 

634 self.status = constants.STATUS_APPROVED 

635 self.save() 

636 for criterion in self.criteria.all(): 

637 criterion.verify() 

638 if self.attached_from == constants.ATTACHED_BY_INHERITANCE: 

639 total = 0 

640 verified = 0 

641 for ( 

642 batch_obj 

643 ) in self.batch.source_transaction.source_batch_objects.all(): 

644 total += float(batch_obj.quantity) 

645 try: 

646 batch_claim = batch_obj.batch.claims.get( 

647 claim=self.claim, status=constants.STATUS_APPROVED 

648 ) 

649 verified += float(batch_obj.quantity) * ( 

650 batch_claim.verification_percentage / 100 

651 ) 

652 except ObjectDoesNotExist: 

653 pass 

654 self.verification_percentage = common_lib._percentage( 

655 verified, total 

656 ) 

657 self.save() 

658 else: 

659 self.check_and_approve()

666 if len(criteria_status) == 1: 

667 self.status = criteria_status[0] 

668 if criteria_status[0] == constants.STATUS_APPROVED: 

669 self.verification_percentage = 100
