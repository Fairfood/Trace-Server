278class TraceableGujiClaim(SystemClaim): 

279 """This claim checks if the batch originated from a farmer and the farmer 

280 is from guji region, ie., all the root batches of the batch is owned by a 

281 farmer and the location of the farmers is inside a preset co-ordinates of 

282 the guji region.""" 

283 

284 boundary = Polygon(GUJI_REGION) 

285 

286 def check_point(self, coords): 

287 """Check point.""" 

288 point = Point(coords) 

289 return self.boundary.contains(point) 

290 

291 def verify(self, batch_criterion): 

292 """Verify with batch criterion.""" 

293 source_region_checks = True 

294 traceable_checks = True 

295 farmer_coordinates = [] 

296 info = "" 

297 _s_transaction = batch_criterion.batch_claim.batch.source_transaction 

298 root_transactions = _s_transaction.get_root_nodes() 

299 for transaction in root_transactions: 

300 # Checking if root is a farm 

301 n_type = NODE_TYPE_FARM 

302 is_farm = transaction.externaltransaction.source.type == n_type 

303 if not is_farm: 

304 info += ( 

305 "%s is not a farm. " 

306 % transaction.externaltransaction.source.full_name 

307 ) 

308 traceable_checks &= is_farm 

309 

310 coordinates = [ 

311 transaction.externaltransaction.source.latitude, 

312 transaction.externaltransaction.source.longitude, 

313 ] 

314 # Checking if root is in Guji region 

315 source_in_guji = self.check_point(coordinates) 

316 farmer_coordinates.append(coordinates) 

317 source_region_checks &= source_in_guji 

318 if not source_in_guji: 

319 info += ( 

320 "%s not in region. " 

321 % transaction.externaltransaction.source.full_name 

322 ) 

323 passed = source_region_checks & traceable_checks 

324 evidence = {"farmer_coordinates": farmer_coordinates} 

325 

326 return passed, info, evidence
