112 def applyInPandas( 

113 self, func: "PandasGroupedMapFunction", schema: Union[StructType, str] 

114 ) -> DataFrame: 

115 """ 

116 Maps each group of the current :class:`DataFrame` using a pandas udf and returns the result 

117 as a `DataFrame`. 

118 

119 The function should take a `pandas.DataFrame` and return another 

120 `pandas.DataFrame`. Alternatively, the user can pass a function that takes 

121 a tuple of the grouping key(s) and a `pandas.DataFrame`. 

122 For each group, all columns are passed together as a `pandas.DataFrame` 

123 to the user-function and the returned `pandas.DataFrame` are combined as a 

124 :class:`DataFrame`. 

125 

126 The `schema` should be a :class:`StructType` describing the schema of the returned 

127 `pandas.DataFrame`. The column labels of the returned `pandas.DataFrame` must either match 

128 the field names in the defined schema if specified as strings, or match the 

129 field data types by position if not strings, e.g. integer indices. 

130 The length of the returned `pandas.DataFrame` can be arbitrary. 

131 

132 .. versionadded:: 3.0.0 

133 

134 .. versionchanged:: 3.4.0 

135 Support Spark Connect. 

136 

137 Parameters 

138 ---------- 

139 func : function 

140 a Python native function that takes a `pandas.DataFrame` and outputs a 

141 `pandas.DataFrame`, or that takes one tuple (grouping keys) and a 

142 `pandas.DataFrame` and outputs a `pandas.DataFrame`. 

143 schema : :class:`pyspark.sql.types.DataType` or str 

144 the return type of the `func` in PySpark. The value can be either a 

145 :class:`pyspark.sql.types.DataType` object or a DDL-formatted type string. 

146 

147 Examples 

148 -------- 

149 >>> import pandas as pd # doctest: +SKIP 

150 >>> from pyspark.sql.functions import pandas_udf, ceil 

151 >>> df = spark.createDataFrame( 

152 ... [(1, 1.0), (1, 2.0), (2, 3.0), (2, 5.0), (2, 10.0)], 

153 ... ("id", "v")) # doctest: +SKIP 

154 >>> def normalize(pdf): 

155 ... v = pdf.v 

156 ... return pdf.assign(v=(v - v.mean()) / v.std()) 

157 >>> df.groupby("id").applyInPandas( 

158 ... normalize, schema="id long, v double").show() # doctest: +SKIP 

159 +---+-------------------+ 

160 | id| v| 

161 +---+-------------------+ 

162 | 1|-0.7071067811865475| 

163 | 1| 0.7071067811865475| 

164 | 2|-0.8320502943378437| 

165 | 2|-0.2773500981126146| 

166 | 2| 1.1094003924504583| 

167 +---+-------------------+ 

168 

169 Alternatively, the user can pass a function that takes two arguments. 

170 In this case, the grouping key(s) will be passed as the first argument and the data will 

171 be passed as the second argument. The grouping key(s) will be passed as a tuple of numpy 

172 data types, e.g., `numpy.int32` and `numpy.float64`. The data will still be passed in 

173 as a `pandas.DataFrame` containing all columns from the original Spark DataFrame. 

174 This is useful when the user does not want to hardcode grouping key(s) in the function. 

175 

176 >>> df = spark.createDataFrame( 

177 ... [(1, 1.0), (1, 2.0), (2, 3.0), (2, 5.0), (2, 10.0)], 

178 ... ("id", "v")) # doctest: +SKIP 

179 >>> def mean_func(key, pdf): 

180 ... # key is a tuple of one numpy.int64, which is the value 

181 ... # of 'id' for the current group 

182 ... return pd.DataFrame([key + (pdf.v.mean(),)]) 

183 >>> df.groupby('id').applyInPandas( 

184 ... mean_func, schema="id long, v double").show() # doctest: +SKIP 

185 +---+---+ 

186 | id| v| 

187 +---+---+ 

188 | 1|1.5| 

189 | 2|6.0| 

190 +---+---+ 

191 

192 >>> def sum_func(key, pdf): 

193 ... # key is a tuple of two numpy.int64s, which is the values 

194 ... # of 'id' and 'ceil(df.v / 2)' for the current group 

195 ... return pd.DataFrame([key + (pdf.v.sum(),)]) 

196 >>> df.groupby(df.id, ceil(df.v / 2)).applyInPandas( 

197 ... sum_func, schema="id long, `ceil(v / 2)` long, v double").show() # doctest: +SKIP 

198 +---+-----------+----+ 

199 | id|ceil(v / 2)| v| 

200 +---+-----------+----+ 

201 | 2| 5|10.0| 

202 | 1| 1| 3.0| 

203 | 2| 3| 5.0| 

204 | 2| 2| 3.0| 

205 +---+-----------+----+ 

206 

207 Notes 

208 ----- 

209 This function requires a full shuffle. All the data of a group will be loaded 

210 into memory, so the user should be aware of the potential OOM risk if data is skewed 

211 and certain groups are too large to fit in memory. 

212 

213 This API is experimental. 

214 

215 See Also 

216 -------- 

217 pyspark.sql.functions.pandas_udf 

218 """ 

219 from pyspark.sql import GroupedData 

220 from pyspark.sql.functions import pandas_udf, PandasUDFType 

221 

222 assert isinstance(self, GroupedData) 

223 

224 udf = pandas_udf(func, returnType=schema, functionType=PandasUDFType.GROUPED_MAP) 

225 df = self._df 

226 udf_column = udf(*[df[col] for col in df.columns]) 

227 jdf = self._jgd.flatMapGroupsInPandas(udf_column._jc.expr()) 

228 return DataFrame(jdf, self.session)

230 def applyInPandasWithState( 

231 self, 

232 func: "PandasGroupedMapFunctionWithState", 

233 outputStructType: Union[StructType, str], 

234 stateStructType: Union[StructType, str], 

235 outputMode: str, 

236 timeoutConf: str, 

237 ) -> DataFrame: 

238 """ 

239 Applies the given function to each group of data, while maintaining a user-defined 

240 per-group state. The result Dataset will represent the flattened record returned by the 

241 function. 

242 

243 For a streaming :class:`DataFrame`, the function will be invoked first for all input groups 

244 and then for all timed out states where the input data is set to be empty. Updates to each 

245 group's state will be saved across invocations. 

246 

247 The function should take parameters (key, Iterator[`pandas.DataFrame`], state) and 

248 return another Iterator[`pandas.DataFrame`]. The grouping key(s) will be passed as a tuple 

249 of numpy data types, e.g., `numpy.int32` and `numpy.float64`. The state will be passed as 

250 :class:`pyspark.sql.streaming.state.GroupState`. 

251 

252 For each group, all columns are passed together as `pandas.DataFrame` to the user-function, 

253 and the returned `pandas.DataFrame` across all invocations are combined as a 

254 :class:`DataFrame`. Note that the user function should not make a guess of the number of 

255 elements in the iterator. To process all data, the user function needs to iterate all 

256 elements and process them. On the other hand, the user function is not strictly required to 

257 iterate through all elements in the iterator if it intends to read a part of data. 

258 

259 The `outputStructType` should be a :class:`StructType` describing the schema of all 

260 elements in the returned value, `pandas.DataFrame`. The column labels of all elements in 

261 returned `pandas.DataFrame` must either match the field names in the defined schema if 

262 specified as strings, or match the field data types by position if not strings, 

263 e.g. integer indices. 

264 

265 The `stateStructType` should be :class:`StructType` describing the schema of the 

266 user-defined state. The value of the state will be presented as a tuple, as well as the 

267 update should be performed with the tuple. The corresponding Python types for 

268 :class:DataType are supported. Please refer to the page 

269 https://spark.apache.org/docs/latest/sql-ref-datatypes.html (Python tab). 

270 

271 The size of each `pandas.DataFrame` in both the input and output can be arbitrary. The 

272 number of `pandas.DataFrame` in both the input and output can also be arbitrary. 

273 

274 .. versionadded:: 3.4.0 

275 

276 Parameters 

277 ---------- 

278 func : function 

279 a Python native function to be called on every group. It should take parameters 

280 (key, Iterator[`pandas.DataFrame`], state) and return Iterator[`pandas.DataFrame`]. 

281 Note that the type of the key is tuple and the type of the state is 

282 :class:`pyspark.sql.streaming.state.GroupState`. 

283 outputStructType : :class:`pyspark.sql.types.DataType` or str 

284 the type of the output records. The value can be either a 

285 :class:`pyspark.sql.types.DataType` object or a DDL-formatted type string. 

286 stateStructType : :class:`pyspark.sql.types.DataType` or str 

287 the type of the user-defined state. The value can be either a 

288 :class:`pyspark.sql.types.DataType` object or a DDL-formatted type string. 

289 outputMode : str 

290 the output mode of the function. 

291 timeoutConf : str 

292 timeout configuration for groups that do not receive data for a while. valid values 

293 are defined in :class:`pyspark.sql.streaming.state.GroupStateTimeout`. 

294 

295 Examples 

296 -------- 

297 >>> import pandas as pd # doctest: +SKIP 

298 >>> from pyspark.sql.streaming.state import GroupStateTimeout 

299 >>> def count_fn(key, pdf_iter, state): 

300 ... assert isinstance(state, GroupStateImpl) 

301 ... total_len = 0 

302 ... for pdf in pdf_iter: 

303 ... total_len += len(pdf) 

304 ... state.update((total_len,)) 

305 ... yield pd.DataFrame({"id": [key[0]], "countAsString": [str(total_len)]}) 

306 ... 

307 >>> df.groupby("id").applyInPandasWithState( 

308 ... count_fn, outputStructType="id long, countAsString string", 

309 ... stateStructType="len long", outputMode="Update", 

310 ... timeoutConf=GroupStateTimeout.NoTimeout) # doctest: +SKIP 

311 

312 Notes 

313 ----- 

314 This function requires a full shuffle. 

315 

316 This API is experimental. 

317 """ 

318 

319 from pyspark.sql import GroupedData 

320 from pyspark.sql.functions import pandas_udf 

321 

322 assert isinstance(self, GroupedData) 

323 assert timeoutConf in [ 

324 GroupStateTimeout.NoTimeout, 

325 GroupStateTimeout.ProcessingTimeTimeout, 

326 GroupStateTimeout.EventTimeTimeout, 

327 ] 

328 

329 if isinstance(outputStructType, str): 

330 outputStructType = cast(StructType, _parse_datatype_string(outputStructType)) 

331 if isinstance(stateStructType, str): 

332 stateStructType = cast(StructType, _parse_datatype_string(stateStructType)) 

333 

334 udf = pandas_udf( 

335 func, # type: ignore[call-overload] 

336 returnType=outputStructType, 

337 functionType=PythonEvalType.SQL_GROUPED_MAP_PANDAS_UDF_WITH_STATE, 

338 ) 

339 df = self._df 

340 udf_column = udf(*[df[col] for col in df.columns]) 

341 jdf = self._jgd.applyInPandasWithState( 

342 udf_column._jc.expr(), 

343 self.session._jsparkSession.parseDataType(outputStructType.json()), 

344 self.session._jsparkSession.parseDataType(stateStructType.json()), 

345 outputMode, 

346 timeoutConf, 

347 ) 

348 return DataFrame(jdf, self.session)

388 self, func: "PandasCogroupedMapFunction", schema: Union[StructType, str] 

389 ) -> DataFrame: 

390 """ 

391 Applies a function to each cogroup using pandas and returns the result 

392 as a `DataFrame`. 

393 

394 The function should take two `pandas.DataFrame`\\s and return another 

395 `pandas.DataFrame`. Alternatively, the user can pass a function that takes 

396 a tuple of the grouping key(s) and the two `pandas.DataFrame`\\s. 

397 For each side of the cogroup, all columns are passed together as a 

398 `pandas.DataFrame` to the user-function and the returned `pandas.DataFrame` are combined as 

399 a :class:`DataFrame`. 

400 

401 The `schema` should be a :class:`StructType` describing the schema of the returned 

402 `pandas.DataFrame`. The column labels of the returned `pandas.DataFrame` must either match 

403 the field names in the defined schema if specified as strings, or match the 

404 field data types by position if not strings, e.g. integer indices. 

405 The length of the returned `pandas.DataFrame` can be arbitrary. 

406 

407 .. versionadded:: 3.0.0 

408 

409 .. versionchanged:: 3.4.0 

410 Support Spark Connect. 

411 

412 Parameters 

413 ---------- 

414 func : function 

415 a Python native function that takes two `pandas.DataFrame`\\s, and 

416 outputs a `pandas.DataFrame`, or that takes one tuple (grouping keys) and two 

417 ``pandas.DataFrame``\\s, and outputs a ``pandas.DataFrame``. 

418 schema : :class:`pyspark.sql.types.DataType` or str 

419 the return type of the `func` in PySpark. The value can be either a 

420 :class:`pyspark.sql.types.DataType` object or a DDL-formatted type string. 

421 

422 Examples 

423 -------- 

424 >>> from pyspark.sql.functions import pandas_udf 

425 >>> df1 = spark.createDataFrame( 

426 ... [(20000101, 1, 1.0), (20000101, 2, 2.0), (20000102, 1, 3.0), (20000102, 2, 4.0)], 

427 ... ("time", "id", "v1")) 

428 >>> df2 = spark.createDataFrame( 

429 ... [(20000101, 1, "x"), (20000101, 2, "y")], 

430 ... ("time", "id", "v2")) 

431 >>> def asof_join(l, r): 

432 ... return pd.merge_asof(l, r, on="time", by="id") 

433 >>> df1.groupby("id").cogroup(df2.groupby("id")).applyInPandas( 

434 ... asof_join, schema="time int, id int, v1 double, v2 string" 

435 ... ).show() # doctest: +SKIP 

436 +--------+---+---+---+ 

437 | time| id| v1| v2| 

438 +--------+---+---+---+ 

439 |20000101| 1|1.0| x| 

440 |20000102| 1|3.0| x| 

441 |20000101| 2|2.0| y| 

442 |20000102| 2|4.0| y| 

443 +--------+---+---+---+ 

444 

445 Alternatively, the user can define a function that takes three arguments. In this case, 

446 the grouping key(s) will be passed as the first argument and the data will be passed as the 

447 second and third arguments. The grouping key(s) will be passed as a tuple of numpy data 

448 types, e.g., `numpy.int32` and `numpy.float64`. The data will still be passed in as two 

449 `pandas.DataFrame` containing all columns from the original Spark DataFrames. 

450 

451 >>> def asof_join(k, l, r): 

452 ... if k == (1,): 

453 ... return pd.merge_asof(l, r, on="time", by="id") 

454 ... else: 

455 ... return pd.DataFrame(columns=['time', 'id', 'v1', 'v2']) 

456 >>> df1.groupby("id").cogroup(df2.groupby("id")).applyInPandas( 

457 ... asof_join, "time int, id int, v1 double, v2 string").show() # doctest: +SKIP 

458 +--------+---+---+---+ 

459 | time| id| v1| v2| 

460 +--------+---+---+---+ 

461 |20000101| 1|1.0| x| 

462 |20000102| 1|3.0| x| 

463 +--------+---+---+---+ 

464

468 into memory, so the user should be aware of the potential OOM risk if data is skewed 

469 and certain groups are too large to fit in memory. 

470 

471 This API is experimental. 

472 

473 See Also

475 pyspark.sql.functions.pandas_udf

477 from pyspark.sql.pandas.functions import pandas_udf

480 udf = pandas_udf( 

481 func, returnType=schema, functionType=PythonEvalType.SQL_COGROUPED_MAP_PANDAS_UDF

485 udf_column = udf(*all_cols) 

486 jdf = self._gd1._jgd.flatMapCoGroupsInPandas(self._gd2._jgd, udf_column._jc.expr())
