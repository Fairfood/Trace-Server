102## The __fspath__ protocol

104try: 

105 from os import PathLike # pylint:disable=unused-import 

106except ImportError: 

107 class PathLike(ABC): 

108 @classmethod 

109 def __subclasshook__(cls, subclass): 

110 return hasattr(subclass, '__fspath__')

112# fspath from 3.6 os.py, but modified to raise the same exceptions as the 

113# real native implementation. 

114# Define for testing 

115def _fspath(path): 

116 """ 

117 Return the path representation of a path-like object. 

118 

119 If str or bytes is passed in, it is returned unchanged. Otherwise the 

120 os.PathLike interface is used to get the path representation. If the 

121 path representation is not str or bytes, TypeError is raised. If the 

122 provided path is not str, bytes, or os.PathLike, TypeError is raised. 

123 """ 

124 if isinstance(path, native_path_types): 

125 return path 

126 

127 # Work from the object's type to match method resolution of other magic 

128 # methods. 

129 path_type = type(path) 

130 try: 

131 path_type_fspath = path_type.__fspath__ 

132 except AttributeError: 

133 raise TypeError("expected str, bytes or os.PathLike object, " 

134 "not " + path_type.__name__) 

135 

136 path_repr = path_type_fspath(path) 

137 if isinstance(path_repr, native_path_types): 

138 return path_repr 

139 

140 raise TypeError("expected {}.__fspath__() to return str or bytes, " 

141 "not {}".format(path_type.__name__, 

142 type(path_repr).__name__)) 

143try: 

144 from os import fspath # pylint: disable=unused-import,no-name-in-module 

145except ImportError: 

146 # if not available, use the Python version as transparently as 

147 # possible 

148 fspath = _fspath 

149 fspath.__name__ = 'fspath'

153except ImportError: 

154 encoding = sys.getfilesystemencoding() or ('utf-8' if not WIN else 'mbcs')

157 # Added in 3.2, so this is for Python 2.7. Note that it doesn't have 

158 # sys.getfilesystemencodeerrors(), which was added in 3.6 

159 def fsencode(filename): 

160 """Encode filename (an os.PathLike, bytes, or str) to the filesystem 

161 encoding with 'surrogateescape' error handler, return bytes unchanged. 

162 On Windows, use 'strict' error handler if the file system encoding is 

163 'mbcs' (which is the default encoding). 

164 """ 

165 filename = fspath(filename) # Does type-checking of `filename`. 

166 if isinstance(filename, bytes): 

167 return filename 

168 

169 try: 

170 return filename.encode(encoding, errors) 

171 except LookupError: 

172 # Can't encode it, and the error handler doesn't 

173 # exist. Probably on Python 2 with an astral character. 

174 # Not sure how to handle this. 

175 raise UnicodeEncodeError("Can't encode path to filesystem encoding")
