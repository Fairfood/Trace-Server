19def code_challenge_method_s256(verifier, challenge): 

20 """ 

21 If the "code_challenge_method" from `Section 4.3`_ was "S256", the 

22 received "code_verifier" is hashed by SHA-256, base64url-encoded, and 

23 then compared to the "code_challenge", i.e.: 

24 

25 BASE64URL-ENCODE(SHA256(ASCII(code_verifier))) == code_challenge 

26 

27 How to implement a base64url-encoding 

28 function without padding, based upon the standard base64-encoding 

29 function that uses padding. 

30 

31 To be concrete, example C# code implementing these functions is shown 

32 below. Similar code could be used in other languages. 

33 

34 static string base64urlencode(byte [] arg) 

35 { 

36 string s = Convert.ToBase64String(arg); // Regular base64 encoder 

37 s = s.Split('=')[0]; // Remove any trailing '='s 

38 s = s.Replace('+', '-'); // 62nd char of encoding 

39 s = s.Replace('/', '_'); // 63rd char of encoding 

40 return s; 

41 } 

42 

43 In python urlsafe_b64encode is already replacing '+' and '/', but preserve 

44 the trailing '='. So we have to remove it. 

45 

46 .. _`Section 4.3`: https://tools.ietf.org/html/rfc7636#section-4.3 

47 """ 

48 return base64.urlsafe_b64encode( 

49 hashlib.sha256(verifier.encode()).digest() 

50 ).decode().rstrip('=') == challenge

55 If the "code_challenge_method" from `Section 4.3`_ was "plain", they are 

56 compared directly, i.e.: 

57 

58 code_verifier == code_challenge. 

59 

60 .. _`Section 4.3`: https://tools.ietf.org/html/rfc7636#section-4.3

65class AuthorizationCodeGrant(GrantTypeBase): 

66 

67 """`Authorization Code Grant`_ 

68 

69 The authorization code grant type is used to obtain both access 

70 tokens and refresh tokens and is optimized for confidential clients. 

71 Since this is a redirection-based flow, the client must be capable of 

72 interacting with the resource owner's user-agent (typically a web 

73 browser) and capable of receiving incoming requests (via redirection) 

74 from the authorization server:: 

75 

76 +----------+ 

77 | Resource | 

78 | Owner | 

79 | | 

80 +----------+ 

81 ^ 

82 | 

83 (B) 

84 +----|-----+ Client Identifier +---------------+ 

85 | -+----(A)-- & Redirection URI ---->| | 

86 | User- | | Authorization | 

87 | Agent -+----(B)-- User authenticates --->| Server | 

88 | | | | 

89 | -+----(C)-- Authorization Code ---<| | 

90 +-|----|---+ +---------------+ 

91 | | ^ v 

92 (A) (C) | | 

93 | | | | 

94 ^ v | | 

95 +---------+ | | 

96 | |>---(D)-- Authorization Code ---------' | 

97 | Client | & Redirection URI | 

98 | | | 

99 | |<---(E)----- Access Token -------------------' 

100 +---------+ (w/ Optional Refresh Token) 

101 

102 Note: The lines illustrating steps (A), (B), and (C) are broken into 

103 two parts as they pass through the user-agent. 

104 

105 Figure 3: Authorization Code Flow 

106 

107 The flow illustrated in Figure 3 includes the following steps: 

108 

109 (A) The client initiates the flow by directing the resource owner's 

110 user-agent to the authorization endpoint. The client includes 

111 its client identifier, requested scope, local state, and a 

112 redirection URI to which the authorization server will send the 

113 user-agent back once access is granted (or denied). 

114 

115 (B) The authorization server authenticates the resource owner (via 

116 the user-agent) and establishes whether the resource owner 

117 grants or denies the client's access request. 

118 

119 (C) Assuming the resource owner grants access, the authorization 

120 server redirects the user-agent back to the client using the 

121 redirection URI provided earlier (in the request or during 

122 client registration). The redirection URI includes an 

123 authorization code and any local state provided by the client 

124 earlier. 

125 

126 (D) The client requests an access token from the authorization 

127 server's token endpoint by including the authorization code 

128 received in the previous step. When making the request, the 

129 client authenticates with the authorization server. The client 

130 includes the redirection URI used to obtain the authorization 

131 code for verification. 

132 

133 (E) The authorization server authenticates the client, validates the 

134 authorization code, and ensures that the redirection URI 

135 received matches the URI used to redirect the client in 

136 step (C). If valid, the authorization server responds back with 

137 an access token and, optionally, a refresh token. 

138 

139 OAuth 2.0 public clients utilizing the Authorization Code Grant are 

140 susceptible to the authorization code interception attack. 

141 

142 A technique to mitigate against the threat through the use of Proof Key for Code 

143 Exchange (PKCE, pronounced "pixy") is implemented in the current oauthlib 

144 implementation. 

145 

146 .. _`Authorization Code Grant`: https://tools.ietf.org/html/rfc6749#section-4.1 

147 .. _`PKCE`: https://tools.ietf.org/html/rfc7636 

148 """ 

149 

150 default_response_mode = 'query' 

151 response_types = ['code'] 

152 

153 # This dict below is private because as RFC mention it: 

154 # "S256" is Mandatory To Implement (MTI) on the server. 

155 # 

156 _code_challenge_methods = { 

157 'plain': code_challenge_method_plain, 

158 'S256': code_challenge_method_s256 

159 }

284 def create_token_response(self, request, token_handler): 

285 """Validate the authorization code. 

286 

287 The client MUST NOT use the authorization code more than once. If an 

288 authorization code is used more than once, the authorization server 

289 MUST deny the request and SHOULD revoke (when possible) all tokens 

290 previously issued based on that authorization code. The authorization 

291 code is bound to the client identifier and redirection URI. 

292 

293 :param request: OAuthlib request. 

294 :type request: oauthlib.common.Request 

295 :param token_handler: A token handler instance, for example of type 

296 oauthlib.oauth2.BearerToken. 

297 

298 """ 

299 headers = self._get_default_headers() 

300 try: 

301 self.validate_token_request(request) 

302 log.debug('Token request validation ok for %r.', request) 

303 except errors.OAuth2Error as e: 

304 log.debug('Client error during validation of %r. %r.', request, e) 

305 headers.update(e.headers) 

306 return headers, e.json, e.status_code 

307 

308 token = token_handler.create_token(request, refresh_token=self.refresh_token)

319 def validate_authorization_request(self, request): 

320 """Check the authorization request for normal and fatal errors. 

321 

322 A normal error could be a missing response_type parameter or the client 

323 attempting to access scope it is not allowed to ask authorization for. 

324 Normal errors can safely be included in the redirection URI and 

325 sent back to the client. 

326 

327 Fatal errors occur when the client_id or redirect_uri is invalid or 

328 missing. These must be caught by the provider and handled, how this 

329 is done is outside of the scope of OAuthLib but showing an error 

330 page describing the issue is a good idea. 

331 

332 :param request: OAuthlib request. 

333 :type request: oauthlib.common.Request 

334 """ 

335 

336 # First check for fatal errors 

337 

338 # If the request fails due to a missing, invalid, or mismatching 

339 # redirection URI, or if the client identifier is missing or invalid, 

340 # the authorization server SHOULD inform the resource owner of the 

341 # error and MUST NOT automatically redirect the user-agent to the 

342 # invalid redirection URI. 

343 

344 # First check duplicate parameters 

345 for param in ('client_id', 'response_type', 'redirect_uri', 'scope', 'state'): 

346 try: 

347 duplicate_params = request.duplicate_params 

348 except ValueError: 

349 raise errors.InvalidRequestFatalError(description='Unable to parse query string', request=request) 

350 if param in duplicate_params: 

351 raise errors.InvalidRequestFatalError(description='Duplicate %s parameter.' % param, request=request) 

352 

353 # REQUIRED. The client identifier as described in Section 2.2. 

354 # https://tools.ietf.org/html/rfc6749#section-2.2 

355 if not request.client_id: 

356 raise errors.MissingClientIdError(request=request) 

357 

358 if not self.request_validator.validate_client_id(request.client_id, request): 

359 raise errors.InvalidClientIdError(request=request) 

360 

361 # OPTIONAL. As described in Section 3.1.2. 

362 # https://tools.ietf.org/html/rfc6749#section-3.1.2 

363 log.debug('Validating redirection uri %s for client %s.', 

364 request.redirect_uri, request.client_id) 

365 

366 # OPTIONAL. As described in Section 3.1.2. 

367 # https://tools.ietf.org/html/rfc6749#section-3.1.2 

368 self._handle_redirects(request) 

369 

370 # Then check for normal errors. 

371 

372 # If the resource owner denies the access request or if the request 

373 # fails for reasons other than a missing or invalid redirection URI, 

374 # the authorization server informs the client by adding the following 

375 # parameters to the query component of the redirection URI using the 

376 # "application/x-www-form-urlencoded" format, per Appendix B. 

377 # https://tools.ietf.org/html/rfc6749#appendix-B 

378 

379 # Note that the correct parameters to be added are automatically 

380 # populated through the use of specific exceptions. 

381 

382 request_info = {} 

383 for validator in self.custom_validators.pre_auth: 

384 request_info.update(validator(request)) 

385 

386 # REQUIRED. 

387 if request.response_type is None: 

388 raise errors.MissingResponseTypeError(request=request) 

389 # Value MUST be set to "code" or one of the OpenID authorization code including 

390 # response_types "code token", "code id_token", "code token id_token" 

391 elif not 'code' in request.response_type and request.response_type != 'none': 

392 raise errors.UnsupportedResponseTypeError(request=request) 

393 

394 if not self.request_validator.validate_response_type(request.client_id, 

395 request.response_type, 

396 request.client, request): 

397 

398 log.debug('Client %s is not authorized to use response_type %s.', 

399 request.client_id, request.response_type) 

400 raise errors.UnauthorizedClientError(request=request) 

401 

402 # OPTIONAL. Validate PKCE request or reply with "error"/"invalid_request" 

403 # https://tools.ietf.org/html/rfc6749#section-4.4.1 

404 if self.request_validator.is_pkce_required(request.client_id, request) is True: 

405 if request.code_challenge is None: 

406 raise errors.MissingCodeChallengeError(request=request) 

407 

408 if request.code_challenge is not None: 

409 request_info["code_challenge"] = request.code_challenge 

410 

411 # OPTIONAL, defaults to "plain" if not present in the request. 

412 if request.code_challenge_method is None: 

413 request.code_challenge_method = "plain" 

414 

415 if request.code_challenge_method not in self._code_challenge_methods: 

416 raise errors.UnsupportedCodeChallengeMethodError(request=request) 

417 request_info["code_challenge_method"] = request.code_challenge_method 

418 

419 # OPTIONAL. The scope of the access request as described by Section 3.3 

420 # https://tools.ietf.org/html/rfc6749#section-3.3 

421 self.validate_scopes(request) 

422 

423 request_info.update({ 

424 'client_id': request.client_id, 

425 'redirect_uri': request.redirect_uri, 

426 'response_type': request.response_type, 

427 'state': request.state, 

428 'request': request 

429 }) 

430 

431 for validator in self.custom_validators.post_auth: 

432 request_info.update(validator(request)) 

433 

434 return request.scopes, request_info

495 if request.code_verifier is None: 

496 raise errors.MissingCodeVerifierError(request=request) 

497 

498 challenge_method = self.request_validator.get_code_challenge_method(request.code, request) 

499 if challenge_method is None: 

500 raise errors.InvalidGrantError(request=request, description="Challenge method not found") 

501 

502 if challenge_method not in self._code_challenge_methods: 

503 raise errors.ServerError( 

504 description="code_challenge_method {} is not supported.".format(challenge_method), 

505 request=request 

506 ) 

507 

508 if not self.validate_code_challenge(challenge, 

509 challenge_method, 

510 request.code_verifier): 

511 log.debug('request provided a invalid code_verifier.') 

512 raise errors.InvalidGrantError(request=request) 

513 elif self.request_validator.is_pkce_required(request.client_id, request) is True: 

514 if request.code_verifier is None: 

515 raise errors.MissingCodeVerifierError(request=request) 

516 raise errors.InvalidGrantError(request=request, description="Challenge not found") 

517 

518 for attr in ('user', 'scopes'): 

519 if getattr(request, attr, None) is None: 

520 log.debug('request.%s was not set on code validation.', attr) 

521 

522 # REQUIRED, if the "redirect_uri" parameter was included in the 

523 # authorization request as described in Section 4.1.1, and their 

524 # values MUST be identical.

527 request.redirect_uri = self.request_validator.get_default_redirect_uri(

530 if not request.redirect_uri: 

531 raise errors.MissingRedirectURIError(request=request)

551 def _create_cors_headers(self, request): 

552 """If CORS is allowed, create the appropriate headers."""
