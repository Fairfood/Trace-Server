435def encode_unicode(f): 

436 """Cerberus error messages expect regular binary strings. 

437 If unicode is used in a ValidationError message can't be printed. 

438 

439 This decorator ensures that if legacy Python is used unicode 

440 strings are encoded before passing to a function. 

441 """ 

442 

443 @wraps(f) 

444 def wrapped(obj, error): 

445 def _encode(value): 

446 """Helper encoding unicode strings into binary utf-8""" 

447 if isinstance(value, unicode): # noqa: F821 

448 return value.encode('utf-8') 

449 return value 

450 

451 error = copy(error) 

452 error.document_path = _encode(error.document_path) 

453 error.schema_path = _encode(error.schema_path) 

454 error.constraint = _encode(error.constraint) 

455 error.value = _encode(error.value) 

456 error.info = _encode(error.info) 

457 return f(obj, error) 

458 

459 return wrapped if PYTHON_VERSION < 3 else f

553 def _insert_error(self, path, node): 

554 """ 

555 Adds an error or sub-tree to :attr:tree. 

556 

557 :param path: Path to the error. 

558 :type path: Tuple of strings and integers. 

559 :param node: An error message or a sub-tree. 

560 :type node: String or dictionary. 

561 """ 

562 field = path[0] 

563 if len(path) == 1: 

564 if field in self.tree: 

565 subtree = self.tree[field].pop() 

566 self.tree[field] += [node, subtree] 

567 else: 

568 self.tree[field] = [node, {}] 

569 elif len(path) >= 1: 

570 if field not in self.tree: 

571 self.tree[field] = [{}] 

572 subtree = self.tree[field][-1] 

573 

574 if subtree: 

575 new = self.__class__(tree=copy(subtree)) 

576 else: 

577 new = self.__class__() 

578 new._insert_error(path[1:], node) 

579 subtree.update(new.tree)

617 def _rewrite_error_path(self, error, offset=0): 

618 """ 

619 Recursively rewrites the error path to correctly represent logic errors 

620 """ 

621 if error.is_logic_error: 

622 self._rewrite_logic_error_path(error, offset) 

623 elif error.is_group_error:

626 def _rewrite_group_error_path(self, error, offset=0): 

627 child_start = len(error.document_path) - offset 

628 

629 for child_error in error.child_errors: 

630 relative_path = child_error.document_path[child_start:] 

631 child_error.document_path = error.document_path + relative_path 

632 

633 self._rewrite_error_path(child_error, offset)

635 def _rewrite_logic_error_path(self, error, offset=0): 

636 child_start = len(error.document_path) - offset 

637 

638 for i, definition_errors in error.definitions_errors.items(): 

639 if not definition_errors: 

640 continue 

641 

642 nodename = '%s definition %s' % (error.rule, i) 

643 path = error.document_path + (nodename,) 

644 

645 for child_error in definition_errors: 

646 rel_path = child_error.document_path[child_start:] 

647 child_error.document_path = path + rel_path 

648 

649 self._rewrite_error_path(child_error, offset + 1)
