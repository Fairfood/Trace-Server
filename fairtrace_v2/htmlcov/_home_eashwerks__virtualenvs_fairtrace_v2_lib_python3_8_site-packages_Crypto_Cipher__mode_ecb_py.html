51class EcbMode(object): 

52 """*Electronic Code Book (ECB)*. 

53 

54 This is the simplest encryption mode. Each of the plaintext blocks 

55 is directly encrypted into a ciphertext block, independently of 

56 any other block. 

57 

58 This mode is dangerous because it exposes frequency of symbols 

59 in your plaintext. Other modes (e.g. *CBC*) should be used instead. 

60 

61 See `NIST SP800-38A`_ , Section 6.1. 

62 

63 .. _`NIST SP800-38A` : http://csrc.nist.gov/publications/nistpubs/800-38a/sp800-38a.pdf 

64 

65 :undocumented: __init__ 

66 """ 

67 

68 def __init__(self, block_cipher): 

69 """Create a new block cipher, configured in ECB mode. 

70 

71 :Parameters: 

72 block_cipher : C pointer 

73 A smart pointer to the low-level block cipher instance. 

74 """ 

75 

76 self._state = VoidPointer() 

77 result = raw_ecb_lib.ECB_start_operation(block_cipher.get(), 

78 self._state.address_of()) 

79 if result: 

80 raise ValueError("Error %d while instantiating the ECB mode" 

81 % result) 

82 

83 # Ensure that object disposal of this Python object will (eventually) 

84 # free the memory allocated by the raw library for the cipher 

85 # mode 

86 self._state = SmartPointer(self._state.get(), 

87 raw_ecb_lib.ECB_stop_operation) 

88 

89 # Memory allocated for the underlying block cipher is now owned 

90 # by the cipher mode 

91 block_cipher.release()

93 def encrypt(self, plaintext, output=None): 

94 """Encrypt data with the key set at initialization. 

95 

96 The data to encrypt can be broken up in two or 

97 more pieces and `encrypt` can be called multiple times. 

98 

99 That is, the statement: 

100 

101 >>> c.encrypt(a) + c.encrypt(b) 

102 

103 is equivalent to: 

104 

105 >>> c.encrypt(a+b) 

106 

107 This function does not add any padding to the plaintext. 

108 

109 :Parameters: 

110 plaintext : bytes/bytearray/memoryview 

111 The piece of data to encrypt. 

112 The length must be multiple of the cipher block length. 

113 :Keywords: 

114 output : bytearray/memoryview 

115 The location where the ciphertext must be written to. 

116 If ``None``, the ciphertext is returned. 

117 :Return: 

118 If ``output`` is ``None``, the ciphertext is returned as ``bytes``. 

119 Otherwise, ``None``. 

120 """ 

121 

122 if output is None: 

123 ciphertext = create_string_buffer(len(plaintext)) 

124 else: 

125 ciphertext = output 

126 

127 if not is_writeable_buffer(output): 

128 raise TypeError("output must be a bytearray or a writeable memoryview") 

129 

130 if len(plaintext) != len(output): 

131 raise ValueError("output must have the same length as the input" 

132 " (%d bytes)" % len(plaintext)) 

133 

134 result = raw_ecb_lib.ECB_encrypt(self._state.get(), 

135 c_uint8_ptr(plaintext), 

136 c_uint8_ptr(ciphertext), 

137 c_size_t(len(plaintext))) 

138 if result: 

139 if result == 3: 

140 raise ValueError("Data must be aligned to block boundary in ECB mode") 

141 raise ValueError("Error %d while encrypting in ECB mode" % result) 

142 

143 if output is None: 

144 return get_raw_buffer(ciphertext) 

145 else: 

146 return None
