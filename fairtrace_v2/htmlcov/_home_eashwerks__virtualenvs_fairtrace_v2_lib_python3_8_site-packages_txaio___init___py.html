107def use_twisted(): 

108 global _explicit_framework 

109 if _explicit_framework is not None and _explicit_framework != 'twisted': 

110 raise RuntimeError("Explicitly using '{}' already".format(_explicit_framework)) 

111 _explicit_framework = 'twisted' 

112 from txaio import tx 

113 _use_framework(tx) 

114 import txaio 

115 txaio.using_twisted = True 

116 txaio.using_asyncio = False 

117

119def use_asyncio(): 

120 global _explicit_framework 

121 if _explicit_framework is not None and _explicit_framework != 'asyncio': 

122 raise RuntimeError("Explicitly using '{}' already".format(_explicit_framework)) 

123 _explicit_framework = 'asyncio' 

124 from txaio import aio 

125 _use_framework(aio) 

126 import txaio 

127 txaio.using_twisted = False 

128 txaio.using_asyncio = True

131def _use_framework(module): 

132 """ 

133 Internal helper, to set this modules methods to a specified 

134 framework helper-methods. 

135 """ 

136 import txaio 

137 for method_name in __all__: 

138 if method_name in ['use_twisted', 'use_asyncio']: 

139 continue 

140 setattr(txaio, method_name, getattr(module, method_name))
