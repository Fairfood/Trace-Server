59def read_configuration( 

60 filepath: _Path, 

61 expand=True, 

62 ignore_option_errors=False, 

63 dist: Optional["Distribution"] = None, 

64): 

65 """Read given configuration file and returns options from it as a dict. 

66 

67 :param str|unicode filepath: Path to configuration file in the ``pyproject.toml`` 

68 format. 

69 

70 :param bool expand: Whether to expand directives and other computed values 

71 (i.e. post-process the given configuration) 

72 

73 :param bool ignore_option_errors: Whether to silently ignore 

74 options, values of which could not be resolved (e.g. due to exceptions 

75 in directives such as file:, attr:, etc.). 

76 If False exceptions are propagated as expected. 

77 

78 :param Distribution|None: Distribution object to which the configuration refers. 

79 If not given a dummy object will be created and discarded after the 

80 configuration is read. This is used for auto-discovery of packages in the case 

81 a dynamic configuration (e.g. ``attr`` or ``cmdclass``) is expanded. 

82 When ``expand=False`` this object is simply ignored. 

83 

84 :rtype: dict 

85 """ 

86 filepath = os.path.abspath(filepath) 

87 

88 if not os.path.isfile(filepath): 

89 raise FileError(f"Configuration file {filepath!r} does not exist.") 

90 

91 asdict = load_file(filepath) or {} 

92 project_table = asdict.get("project", {}) 

93 tool_table = asdict.get("tool", {}) 

94 setuptools_table = tool_table.get("setuptools", {}) 

95 if not asdict or not (project_table or setuptools_table): 

96 return {} # User is not using pyproject to configure setuptools 

97 

98 # TODO: Remove the following once the feature stabilizes: 

99 msg = ( 

100 "Support for project metadata in `pyproject.toml` is still experimental " 

101 "and may be removed (or change) in future releases." 

102 ) 

103 warnings.warn(msg, _ExperimentalProjectMetadata) 

104 

105 # There is an overall sense in the community that making include_package_data=True 

106 # the default would be an improvement. 

107 # `ini2toml` backfills include_package_data=False when nothing is explicitly given, 

108 # therefore setting a default here is backwards compatible. 

109 orig_setuptools_table = setuptools_table.copy() 

110 if dist and getattr(dist, "include_package_data") is not None: 

111 setuptools_table.setdefault("include-package-data", dist.include_package_data) 

112 else: 

113 setuptools_table.setdefault("include-package-data", True) 

114 # Persist changes: 

115 asdict["tool"] = tool_table 

116 tool_table["setuptools"] = setuptools_table 

117 

118 try: 

119 # Don't complain about unrelated errors (e.g. tools not using the "tool" table) 

120 subset = {"project": project_table, "tool": {"setuptools": setuptools_table}} 

121 validate(subset, filepath) 

122 except Exception as ex: 

123 # TODO: Remove the following once the feature stabilizes: 

124 if _skip_bad_config(project_table, orig_setuptools_table, dist): 

125 return {} 

126 # TODO: After the previous statement is removed the try/except can be replaced 

127 # by the _ignore_errors context manager. 

128 if ignore_option_errors: 

129 _logger.debug(f"ignored error: {ex.__class__.__name__} - {ex}") 

130 else: 

131 raise # re-raise exception 

132 

133 if expand: 

134 root_dir = os.path.dirname(filepath) 

135 return expand_configuration(asdict, root_dir, ignore_option_errors, dist) 

136 

137 return asdict

140def _skip_bad_config( 

141 project_cfg: dict, setuptools_cfg: dict, dist: Optional["Distribution"] 

142) -> bool: 

143 """Be temporarily forgiving with invalid ``pyproject.toml``""" 

144 # See pypa/setuptools#3199 and pypa/cibuildwheel#1064 

145 

146 if dist is None or ( 

147 dist.metadata.name is None 

148 and dist.metadata.version is None 

149 and dist.install_requires is None 

150 ): 

151 # It seems that the build is not getting any configuration from other places 

152 return False 

153 

154 if setuptools_cfg: 

155 # If `[tool.setuptools]` is set, then `pyproject.toml` config is intentional 

156 return False 

157 

158 given_config = set(project_cfg.keys()) 

159 popular_subset = {"name", "version", "python_requires", "requires-python"} 

160 if given_config <= popular_subset: 

161 # It seems that the docs in cibuildtool has been inadvertently encouraging users 

162 # to create `pyproject.toml` files that are not compliant with the standards. 

163 # Let's be forgiving for the time being. 

164 warnings.warn(_InvalidFile.message(), _InvalidFile, stacklevel=2) 

165 return True 

166 

167 return False

382 def __enter__(self): 

383 """When entering the context, the values of ``packages``, ``py_modules`` and 

384 ``package_dir`` that are missing in ``dist`` are copied from ``setuptools_cfg``. 

385 """ 

386 dist, cfg = self._dist, self._setuptools_cfg 

387 package_dir: Dict[str, str] = cfg.setdefault("package-dir", {}) 

388 package_dir.update(dist.package_dir or {}) 

389 dist.package_dir = package_dir # needs to be the same object 

390 

391 dist.set_defaults._ignore_ext_modules() # pyproject.toml-specific behaviour 

392 

393 # Set `py_modules` and `packages` in dist to short-circuit auto-discovery, 

394 # but avoid overwriting empty lists purposefully set by users. 

395 if dist.py_modules is None: 

396 dist.py_modules = cfg.get("py-modules") 

397 if dist.packages is None: 

398 dist.packages = cfg.get("packages") 

399 

400 return super().__enter__()

402 def __exit__(self, exc_type, exc_value, traceback): 

403 """When exiting the context, if values of ``packages``, ``py_modules`` and 

404 ``package_dir`` are missing in ``setuptools_cfg``, copy from ``dist``. 

405 """ 

406 # If anything was discovered set them back, so they count in the final config. 

407 self._setuptools_cfg.setdefault("packages", self._dist.packages) 

408 self._setuptools_cfg.setdefault("py-modules", self._dist.py_modules)
