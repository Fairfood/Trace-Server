61def make_swagger_name(attribute_name): 

62 """ 

63 Convert a python variable name into a Swagger spec attribute name. 

64 

65 In particular, 

66 * if name starts with ``x_``, return ``x-{camelCase}`` 

67 * if name is ``ref``, return ``$ref`` 

68 * else return the name converted to camelCase, with trailing underscores stripped 

69 

70 :param str attribute_name: python attribute name 

71 :return: swagger name 

72 """ 

73 if attribute_name == 'ref': 

74 return "$ref" 

75 if attribute_name.startswith("x_"): 

76 return "x-" + camelize(attribute_name[2:], uppercase_first_letter=False) 

77 return camelize(attribute_name.rstrip('_'), uppercase_first_letter=False)

87class SwaggerDict(OrderedDict): 

88 """A particular type of OrderedDict, which maps all attribute accesses to dict lookups using 

89 :func:`.make_swagger_name`. Attribute names starting with ``_`` are set on the object as-is and are not included 

90 in the specification output. 

91 

92 Used as a base class for all Swagger helper models. 

93 """ 

94 

95 def __init__(self, **attrs): 

96 super(SwaggerDict, self).__init__() 

97 self._extras__ = attrs 

98 if type(self) == SwaggerDict: 

99 self._insert_extras__()

101 def __setattr__(self, key, value): 

102 if key.startswith('_'): 

103 super(SwaggerDict, self).__setattr__(key, value) 

104 return 

105 if value is not None: 

106 self[make_swagger_name(key)] = value 

107 

108 def __getattr__(self, item): 

109 if item.startswith('_'): 

110 raise AttributeError 

111 try: 

112 return self[make_swagger_name(item)] 

113 except KeyError: 

114 # raise_from is EXTREMELY slow, replaced with plain raise 

115 raise AttributeError("object of class " + type(self).__name__ + " has no attribute " + item)

117 def __delattr__(self, item): 

118 if item.startswith('_'): 

119 super(SwaggerDict, self).__delattr__(item) 

120 return

123 def _insert_extras__(self): 

124 """ 

125 From an ordering perspective, it is desired that extra attributes such as vendor extensions stay at the 

126 bottom of the object. However, python2.7's OrderdDict craps out if you try to insert into it before calling 

127 init. This means that subclasses must call super().__init__ as the first statement of their own __init__, 

128 which would result in the extra attributes being added first. For this reason, we defer the insertion of the 

129 attributes and require that subclasses call ._insert_extras__ at the end of their __init__ method. 

130 """ 

131 for attr, val in sorted(self._extras__.items()): 

132 setattr(self, attr, val) 

133 

134 @staticmethod 

135 def _as_odict(obj, memo): 

136 """Implementation detail of :meth:`.as_odict`""" 

137 if id(obj) in memo: 

138 return memo[id(obj)] 

139 

140 if isinstance(obj, Promise) and hasattr(obj, '_proxy____cast'): 

141 # handle __proxy__ objects from django.utils.functional.lazy 

142 obj = obj._proxy____cast() 

143 

144 if isinstance(obj, collections_abc.Mapping): 

145 result = OrderedDict() 

146 memo[id(obj)] = result 

147 items = obj.items() 

148 if not dict_has_ordered_keys(obj): 

149 items = sorted(items) 

150 for attr, val in items: 

151 result[attr] = SwaggerDict._as_odict(val, memo) 

152 return result 

153 elif isinstance(obj, six.string_types): 

154 return force_real_str(obj) 

155 elif isinstance(obj, collections_abc.Iterable) and not isinstance(obj, collections_abc.Iterator): 

156 return type(obj)(SwaggerDict._as_odict(elem, memo) for elem in obj) 

157 

158 return obj

160 def as_odict(self): 

161 """Convert this object into an ``OrderedDict`` instance. 

162 

163 :rtype: OrderedDict 

164 """ 

165 return SwaggerDict._as_odict(self, {})

167 def __reduce__(self): 

168 # for pickle supprt; this skips calls to all SwaggerDict __init__ methods and relies 

169 # on the already set attributes instead 

170 attrs = {k: v for k, v in vars(self).items() if not k.startswith('_NP_')} 

171 return _bare_SwaggerDict, (type(self),), attrs, None, iter(self.items())

516

600 or type[rest_framework.serializers.Serializer] 

601 :param dict examples: example bodies mapped by mime type

603 super(Response, self).__init__(**extra) 

604 self.description = description

606 self.examples = examples

612class ReferenceResolver(object): 

613 """A mapping type intended for storing objects pointed at by Swagger Refs. 

614 Provides support and checks for different refernce scopes, e.g. 'definitions'. 

615 

616 For example: 

617 

618 :: 

619 

620 > components = ReferenceResolver('definitions', 'parameters') 

621 > definitions = components.with_scope('definitions') 

622 > definitions.set('Article', Schema(...)) 

623 > print(components) 

624 {'definitions': OrderedDict([('Article', Schema(...)]), 'parameters': OrderedDict()} 

625 """ 

626 

627 def __init__(self, *scopes, **kwargs): 

628 """ 

629 :param str scopes: an enumeration of the valid scopes this resolver will contain 

630 """ 

631 force_init = kwargs.pop('force_init', False) 

632 if not force_init: 

633 raise AssertionError( 

634 "Creating an instance of ReferenceResolver almost certainly won't do what you want it to do.\n" 

635 "See https://github.com/axnsan12/drf-yasg/issues/211, " 

636 "https://github.com/axnsan12/drf-yasg/issues/271, " 

637 "https://github.com/axnsan12/drf-yasg/issues/325.\n" 

638 "Pass `force_init=True` to override this." 

639 ) 

640 

641 self._objects = OrderedDict() 

642 self._force_scope = None 

643 for scope in scopes: 

644 assert isinstance(scope, str), "scope names must be strings" 

645 self._objects[scope] = OrderedDict() 

646 

647 def with_scope(self, scope): 

648 """Return a view into this :class:`.ReferenceResolver` whose scope is defaulted and forced to `scope`. 

649 

650 :param str scope: target scope, must be in this resolver's `scopes` 

651 :return: the bound resolver 

652 :rtype: .ReferenceResolver 

653 """ 

654 assert scope in self.scopes, "unknown scope %s" % scope 

655 ret = ReferenceResolver(force_init=True) 

656 ret._objects = self._objects 

657 ret._force_scope = scope 

658 return ret 

659 

660 def _check_scope(self, scope): 

661 real_scope = self._force_scope or scope 

662 if scope is not None: 

663 assert not self._force_scope or scope == self._force_scope, "cannot overrride forced scope" 

664 assert real_scope and real_scope in self._objects, "invalid scope %s" % scope 

665 return real_scope 

666 

667 def set(self, name, obj, scope=None): 

668 """Set an object in the given scope, raise an error if it already exists. 

669 

670 :param str name: reference name 

671 :param obj: referenced object 

672 :param str scope: reference scope 

673 """ 

674 scope = self._check_scope(scope) 

675 assert obj is not None, "referenced objects cannot be None/null" 

676 assert name not in self._objects[scope], "#/%s/%s already exists" % (scope, name) 

677 self._objects[scope][name] = obj 

678 

679 def setdefault(self, name, maker, scope=None): 

680 """Set an object in the given scope only if it does not exist. 

681 

682 :param str name: reference name 

683 :param function maker: object factory, called only if necessary 

684 :param str scope: reference scope 

685 """ 

686 scope = self._check_scope(scope) 

687 assert callable(maker), "setdefault expects a callable, not %s" % type(maker).__name__ 

688 ret = self.getdefault(name, None, scope) 

689 if ret is None: 

690 ret = maker() 

691 value = self.getdefault(name, None, scope) 

692 assert ret is not None, "maker returned None; referenced objects cannot be None/null" 

693 if value is None: 

694 self.set(name, ret, scope) 

695 elif value != ret: 

696 logger.debug("during setdefault, maker for %s inserted a value and returned a different value", name) 

697 ret = value 

698 

699 return ret 

700 

701 def get(self, name, scope=None): 

702 """Get an object from the given scope, raise an error if it does not exist. 

703 

704 :param str name: reference name 

705 :param str scope: reference scope 

706 :return: the object 

707 """ 

708 scope = self._check_scope(scope) 

709 assert name in self._objects[scope], "#/%s/%s is not defined" % (scope, name) 

710 return self._objects[scope][name] 

711 

712 def getdefault(self, name, default=None, scope=None): 

713 """Get an object from the given scope or a default value if it does not exist. 

714 

715 :param str name: reference name 

716 :param default: the default value 

717 :param str scope: reference scope 

718 :return: the object or `default` 

719 """ 

720 scope = self._check_scope(scope) 

721 return self._objects[scope].get(name, default) 

722 

723 def has(self, name, scope=None): 

724 """Check if an object exists in the given scope. 

725 

726 :param str name: reference name 

727 :param str scope: reference scope 

728 :return: True if the object exists 

729 :rtype: bool 

730 """ 

731 scope = self._check_scope(scope) 

732 return name in self._objects[scope] 

733 

734 def __iter__(self): 

735 if self._force_scope: 

736 return iter(self._objects[self._force_scope]) 

737 return iter(self._objects) 

738 

739 @property 

740 def scopes(self): 

741 if self._force_scope: 

742 return [self._force_scope] 

743 return list(self._objects.keys()) 

744 

745 # act as mapping 

746 def keys(self): 

747 if self._force_scope: 

748 return self._objects[self._force_scope].keys() 

749 return self._objects.keys() 

750 

751 def __getitem__(self, item): 

752 if self._force_scope: 

753 return self._objects[self._force_scope][item] 

754 return self._objects[item]

757 return str(dict(self))
