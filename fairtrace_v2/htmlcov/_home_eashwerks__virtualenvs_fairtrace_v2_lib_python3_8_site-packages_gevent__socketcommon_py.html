319def __send_chunk(socket, data_memory, flags, timeleft, end, timeout=_timeout_error): 

320 """ 

321 Send the complete contents of ``data_memory`` before returning. 

322 This is the core loop around :meth:`send`. 

323 

324 :param timeleft: Either ``None`` if there is no timeout involved, 

325 or a float indicating the timeout to use. 

326 :param end: Either ``None`` if there is no timeout involved, or 

327 a float giving the absolute end time. 

328 :return: An updated value for ``timeleft`` (or None) 

329 :raises timeout: If ``timeleft`` was given and elapsed while 

330 sending this chunk. 

331 """ 

332 data_sent = 0 

333 len_data_memory = len(data_memory) 

334 started_timer = 0 

335 while data_sent < len_data_memory: 

336 chunk = data_memory[data_sent:] 

337 if timeleft is None: 

338 data_sent += socket.send(chunk, flags) 

339 elif started_timer and timeleft <= 0: 

340 # Check before sending to guarantee a check 

341 # happens even if each chunk successfully sends its data 

342 # (especially important for SSL sockets since they have large 

343 # buffers). But only do this if we've actually tried to 

344 # send something once to avoid spurious timeouts on non-blocking 

345 # sockets. 

346 raise timeout('timed out') 

347 else: 

348 started_timer = 1 

349 data_sent += socket.send(chunk, flags, timeout=timeleft) 

350 timeleft = end - time.time() 

351 

352 return timeleft

354def _sendall(socket, data_memory, flags, 

355 SOL_SOCKET=__socket__.SOL_SOCKET, # pylint:disable=no-member 

356 SO_SNDBUF=__socket__.SO_SNDBUF): # pylint:disable=no-member 

357 """ 

358 Send the *data_memory* (which should be a memoryview) 

359 using the gevent *socket*, performing well on PyPy. 

360 """ 

361 

362 # On PyPy up through 5.10.0, both PyPy2 and PyPy3, subviews 

363 # (slices) of a memoryview() object copy the underlying bytes the 

364 # first time the builtin socket.send() method is called. On a 

365 # non-blocking socket (that thus calls socket.send() many times) 

366 # with a large input, this results in many repeated copies of an 

367 # ever smaller string, depending on the networking buffering. For 

368 # example, if each send() can process 1MB of a 50MB input, and we 

369 # naively pass the entire remaining subview each time, we'd copy 

370 # 49MB, 48MB, 47MB, etc, thus completely killing performance. To 

371 # workaround this problem, we work in reasonable, fixed-size 

372 # chunks. This results in a 10x improvement to bench_sendall.py, 

373 # while having no measurable impact on CPython (since it doesn't 

374 # copy at all the only extra overhead is a few python function 

375 # calls, which is negligible for large inputs). 

376 

377 # On one macOS machine, PyPy3 5.10.1 produced ~ 67.53 MB/s before this change, 

378 # and ~ 616.01 MB/s after. 

379 

380 # See https://bitbucket.org/pypy/pypy/issues/2091/non-blocking-socketsend-slow-gevent 

381 

382 # Too small of a chunk (the socket's buf size is usually too 

383 # small) results in reduced perf due to *too many* calls to send and too many 

384 # small copies. With a buffer of 143K (the default on my system), for 

385 # example, bench_sendall.py yields ~264MB/s, while using 1MB yields 

386 # ~653MB/s (matching CPython). 1MB is arbitrary and might be better 

387 # chosen, say, to match a page size? 

388 

389 len_data_memory = len(data_memory) 

390 if not len_data_memory: 

391 # Don't try to send empty data at all, no point, and breaks ssl 

392 # See issue 719 

393 return 0 

394 

395 

396 chunk_size = max(socket.getsockopt(SOL_SOCKET, SO_SNDBUF), 1024 * 1024) 

397 

398 data_sent = 0 

399 end = None 

400 timeleft = None 

401 if socket.timeout is not None: 

402 timeleft = socket.timeout 

403 end = time.time() + timeleft 

404 

405 while data_sent < len_data_memory: 

406 chunk_end = min(data_sent + chunk_size, len_data_memory) 

407 chunk = data_memory[data_sent:chunk_end] 

408 

409 timeleft = __send_chunk(socket, chunk, flags, timeleft, end) 

410 data_sent += len(chunk) # Guaranteed it sent the whole thing

417def _resolve_addr(sock, address): 

418 # Internal method: resolve the AF_INET[6] address using 

419 # getaddrinfo. 

420 if sock.family not in _RESOLVABLE_FAMILIES or not isinstance(address, tuple): 

421 return address 

422 # address is (host, port) (ipv4) or (host, port, flowinfo, scopeid) (ipv6). 

423 # If it's already resolved, no need to go through getaddrinfo() again. 

424 # That can lose precision (e.g., on IPv6, it can lose scopeid). The standard library 

425 # does this in socketmodule.c:setipaddr. (This is only part of the logic, the real 

426 # thing is much more complex.) 

427 try: 

428 if __socket__.inet_pton(sock.family, address[0]): 

429 return address 

430 except AttributeError: # pragma: no cover 

431 # inet_pton might not be available. 

432 pass 

433 except _SocketError: 

434 # Not parseable, needs resolved. 

435 pass 

436 

437 

438 # We don't pass the port to getaddrinfo because the C 

439 # socket module doesn't either (on some systems its 

440 # illegal to do that without also passing socket type and 

441 # protocol). Instead we join the port back at the end. 

442 # See https://github.com/gevent/gevent/issues/1252 

443 host, port = address[:2] 

444 r = getaddrinfo(host, None, sock.family) 

445 address = r[0][-1] 

446 if len(address) == 2: 

447 address = (address[0], port) 

448 else: 

449 address = (address[0], port, address[2], address[3]) 

450 return address

591 def connect(self, address): 

592 """ 

593 Connect to *address*. 

594 

595 .. versionchanged:: 20.6.0 

596 If the host part of the address includes an IPv6 scope ID, 

597 it will be used instead of ignored, if the platform supplies 

598 :func:`socket.inet_pton`. 

599 """ 

600 if self.timeout == 0.0: 

601 return self._sock.connect(address) 

602 address = _resolve_addr(self._sock, address) 

603 with Timeout._start_new_or_dummy(self.timeout, __socket__.timeout("timed out")): 

604 while 1: 

605 err = self.getsockopt(__socket__.SOL_SOCKET, __socket__.SO_ERROR) 

606 if err: 

607 raise _SocketError(err, strerror(err)) 

608 result = self._sock.connect_ex(address) 

609 

610 if not result or result == EISCONN: 

611 break 

612 if (result in (EWOULDBLOCK, EINPROGRESS, EALREADY)) or (result == EINVAL and is_windows): 

613 self._wait(self._write_event) 

614 else: 

615 if (isinstance(address, tuple)

617 and result == EHOSTUNREACH):

619 # returned for this link-local address, but it really is

622 # (On previous versions, that code passed the '127.0.0.1' IPv4 address, so

624 # It is something of a mystery how the stdlib socket code doesn't 

625 # produce EHOSTUNREACH---I (JAM) can't see how socketmodule.c would avoid 

626 # that. The normal connect just calls connect_ex much like we do.
