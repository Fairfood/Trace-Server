31""" 

32Offset Codebook (OCB) mode. 

33 

34OCB is Authenticated Encryption with Associated Data (AEAD) cipher mode 

35designed by Prof. Phillip Rogaway and specified in `RFC7253`_. 

36 

37The algorithm provides both authenticity and privacy, it is very efficient, 

38it uses only one key and it can be used in online mode (so that encryption 

39or decryption can start before the end of the message is available). 

40 

41This module implements the third and last variant of OCB (OCB3) and it only 

42works in combination with a 128-bit block symmetric cipher, like AES. 

43 

44OCB is patented in US but `free licenses`_ exist for software implementations 

45meant for non-military purposes. 

46 

47Example: 

48 >>> from Crypto.Cipher import AES 

49 >>> from Crypto.Random import get_random_bytes 

50 >>> 

51 >>> key = get_random_bytes(32) 

52 >>> cipher = AES.new(key, AES.MODE_OCB) 

53 >>> plaintext = b"Attack at dawn" 

54 >>> ciphertext, mac = cipher.encrypt_and_digest(plaintext) 

55 >>> # Deliver cipher.nonce, ciphertext and mac 

56 ... 

57 >>> cipher = AES.new(key, AES.MODE_OCB, nonce=nonce) 

58 >>> try: 

59 >>> plaintext = cipher.decrypt_and_verify(ciphertext, mac) 

60 >>> except ValueError: 

61 >>> print "Invalid message" 

62 >>> else: 

63 >>> print plaintext 

64 

65:undocumented: __package__ 

66 

67.. _RFC7253: http://www.rfc-editor.org/info/rfc7253 

68.. _free licenses: http://web.cs.ucdavis.edu/~rogaway/ocb/license.htm 

69"""

115 def __init__(self, factory, nonce, mac_len, cipher_params): 

116 

117 if factory.block_size != 16: 

118 raise ValueError("OCB mode is only available for ciphers" 

119 " that operate on 128 bits blocks") 

120 

121 self.block_size = 16 

122 """The block size of the underlying cipher, in bytes.""" 

123 

124 self.nonce = _copy_bytes(None, None, nonce) 

125 """Nonce used for this session.""" 

126 if len(nonce) not in range(1, 16): 

127 raise ValueError("Nonce must be at most 15 bytes long") 

128 if not is_buffer(nonce): 

129 raise TypeError("Nonce must be bytes, bytearray or memoryview") 

130 

131 self._mac_len = mac_len 

132 if not 8 <= mac_len <= 16: 

133 raise ValueError("MAC tag must be between 8 and 16 bytes long") 

134 

135 # Cache for MAC tag 

136 self._mac_tag = None 

137 

138 # Cache for unaligned associated data 

139 self._cache_A = b"" 

140 

141 # Cache for unaligned ciphertext/plaintext 

142 self._cache_P = b"" 

143 

144 # Allowed transitions after initialization 

145 self._next = [self.update, self.encrypt, self.decrypt, 

146 self.digest, self.verify] 

147 

148 # Compute Offset_0 

149 params_without_key = dict(cipher_params) 

150 key = params_without_key.pop("key") 

151 nonce = (struct.pack('B', self._mac_len << 4 & 0xFF) + 

152 b'\x00' * (14 - len(nonce)) + 

153 b'\x01' + self.nonce) 

154 

155 bottom_bits = bord(nonce[15]) & 0x3F # 6 bits, 0..63 

156 top_bits = bord(nonce[15]) & 0xC0 # 2 bits 

157 

158 ktop_cipher = factory.new(key, 

159 factory.MODE_ECB, 

160 **params_without_key) 

161 ktop = ktop_cipher.encrypt(struct.pack('15sB', 

162 nonce[:15], 

163 top_bits)) 

164 

165 stretch = ktop + strxor(ktop[:8], ktop[1:9]) # 192 bits 

166 offset_0 = long_to_bytes(bytes_to_long(stretch) >> 

167 (64 - bottom_bits), 24)[8:] 

168 

169 # Create low-level cipher instance 

170 raw_cipher = factory._create_base_cipher(cipher_params) 

171 if cipher_params: 

172 raise TypeError("Unknown keywords: " + str(cipher_params)) 

173 

174 self._state = VoidPointer() 

175 result = _raw_ocb_lib.OCB_start_operation(raw_cipher.get(), 

176 offset_0, 

177 c_size_t(len(offset_0)), 

178 self._state.address_of()) 

179 if result: 

180 raise ValueError("Error %d while instantiating the OCB mode" 

181 % result) 

182 

183 # Ensure that object disposal of this Python object will (eventually) 

184 # free the memory allocated by the raw library for the cipher mode 

185 self._state = SmartPointer(self._state.get(), 

186 _raw_ocb_lib.OCB_stop_operation) 

187 

188 # Memory allocated for the underlying block cipher is now owed 

189 # by the cipher mode 

190 raw_cipher.release() 

191 

192 def _update(self, assoc_data, assoc_data_len): 

193 result = _raw_ocb_lib.OCB_update(self._state.get(), 

194 c_uint8_ptr(assoc_data), 

195 c_size_t(assoc_data_len)) 

196 if result: 

197 raise ValueError("Error %d while computing MAC in OCB mode" % result)

199 def update(self, assoc_data): 

200 """Process the associated data. 

201 

202 If there is any associated data, the caller has to invoke 

203 this method one or more times, before using 

204 ``decrypt`` or ``encrypt``. 

205 

206 By *associated data* it is meant any data (e.g. packet headers) that 

207 will not be encrypted and will be transmitted in the clear. 

208 However, the receiver shall still able to detect modifications. 

209 

210 If there is no associated data, this method must not be called. 

211 

212 The caller may split associated data in segments of any size, and 

213 invoke this method multiple times, each time with the next segment. 

214 

215 :Parameters: 

216 assoc_data : bytes/bytearray/memoryview 

217 A piece of associated data. 

218 """ 

219 

220 if self.update not in self._next: 

221 raise TypeError("update() can only be called" 

222 " immediately after initialization") 

223 

224 self._next = [self.encrypt, self.decrypt, self.digest, 

225 self.verify, self.update] 

226 

227 if len(self._cache_A) > 0: 

228 filler = min(16 - len(self._cache_A), len(assoc_data)) 

229 self._cache_A += _copy_bytes(None, filler, assoc_data) 

230 assoc_data = assoc_data[filler:] 

231 

232 if len(self._cache_A) < 16: 

233 return self 

234 

235 # Clear the cache, and proceeding with any other aligned data 

236 self._cache_A, seg = b"", self._cache_A 

237 self.update(seg) 

238 

239 update_len = len(assoc_data) // 16 * 16 

240 self._cache_A = _copy_bytes(update_len, None, assoc_data) 

241 self._update(assoc_data, update_len) 

242 return self

244 def _transcrypt_aligned(self, in_data, in_data_len, 

245 trans_func, trans_desc): 

246 

247 out_data = create_string_buffer(in_data_len) 

248 result = trans_func(self._state.get(), 

249 in_data, 

250 out_data, 

251 c_size_t(in_data_len)) 

252 if result: 

253 raise ValueError("Error %d while %sing in OCB mode" 

254 % (result, trans_desc)) 

255 return get_raw_buffer(out_data)

257 def _transcrypt(self, in_data, trans_func, trans_desc): 

258 # Last piece to encrypt/decrypt 

259 if in_data is None: 

260 out_data = self._transcrypt_aligned(self._cache_P, 

261 len(self._cache_P), 

262 trans_func, 

263 trans_desc) 

264 self._cache_P = b"" 

265 return out_data 

266 

267 # Try to fill up the cache, if it already contains something 

268 prefix = b"" 

269 if len(self._cache_P) > 0: 

270 filler = min(16 - len(self._cache_P), len(in_data)) 

271 self._cache_P += _copy_bytes(None, filler, in_data) 

272 in_data = in_data[filler:] 

273 

274 if len(self._cache_P) < 16: 

275 # We could not manage to fill the cache, so there is certainly 

276 # no output yet. 

277 return b"" 

278 

279 # Clear the cache, and proceeding with any other aligned data 

280 prefix = self._transcrypt_aligned(self._cache_P, 

281 len(self._cache_P), 

282 trans_func, 

283 trans_desc) 

284 self._cache_P = b"" 

285 

286 # Process data in multiples of the block size 

287 trans_len = len(in_data) // 16 * 16 

288 result = self._transcrypt_aligned(c_uint8_ptr(in_data), 

289 trans_len, 

290 trans_func, 

291 trans_desc) 

292 if prefix: 

293 result = prefix + result 

294 

295 # Left-over 

296 self._cache_P = _copy_bytes(trans_len, None, in_data) 

297 

298 return result

300 def encrypt(self, plaintext=None): 

301 """Encrypt the next piece of plaintext. 

302 

303 After the entire plaintext has been passed (but before `digest`), 

304 you **must** call this method one last time with no arguments to collect 

305 the final piece of ciphertext. 

306 

307 If possible, use the method `encrypt_and_digest` instead. 

308 

309 :Parameters: 

310 plaintext : bytes/bytearray/memoryview 

311 The next piece of data to encrypt or ``None`` to signify 

312 that encryption has finished and that any remaining ciphertext 

313 has to be produced. 

314 :Return: 

315 the ciphertext, as a byte string. 

316 Its length may not match the length of the *plaintext*. 

317 """ 

318 

319 if self.encrypt not in self._next: 

320 raise TypeError("encrypt() can only be called after" 

321 " initialization or an update()") 

322 

323 if plaintext is None: 

324 self._next = [self.digest] 

325 else: 

326 self._next = [self.encrypt] 

327 return self._transcrypt(plaintext, _raw_ocb_lib.OCB_encrypt, "encrypt")

329 def decrypt(self, ciphertext=None): 

330 """Decrypt the next piece of ciphertext. 

331 

332 After the entire ciphertext has been passed (but before `verify`), 

333 you **must** call this method one last time with no arguments to collect 

334 the remaining piece of plaintext. 

335 

336 If possible, use the method `decrypt_and_verify` instead. 

337 

338 :Parameters: 

339 ciphertext : bytes/bytearray/memoryview 

340 The next piece of data to decrypt or ``None`` to signify 

341 that decryption has finished and that any remaining plaintext 

342 has to be produced. 

343 :Return: 

344 the plaintext, as a byte string. 

345 Its length may not match the length of the *ciphertext*. 

346 """ 

347 

348 if self.decrypt not in self._next: 

349 raise TypeError("decrypt() can only be called after" 

350 " initialization or an update()") 

351 

352 if ciphertext is None: 

353 self._next = [self.verify] 

354 else: 

355 self._next = [self.decrypt] 

356 return self._transcrypt(ciphertext, 

357 _raw_ocb_lib.OCB_decrypt, 

358 "decrypt")

360 def _compute_mac_tag(self): 

361 

362 if self._mac_tag is not None: 

363 return 

364 

365 if self._cache_A: 

366 self._update(self._cache_A, len(self._cache_A)) 

367 self._cache_A = b"" 

368 

369 mac_tag = create_string_buffer(16) 

370 result = _raw_ocb_lib.OCB_digest(self._state.get(), 

371 mac_tag, 

372 c_size_t(len(mac_tag)) 

373 ) 

374 if result: 

375 raise ValueError("Error %d while computing digest in OCB mode" 

376 % result) 

377 self._mac_tag = get_raw_buffer(mac_tag)[:self._mac_len]

379 def digest(self): 

380 """Compute the *binary* MAC tag. 

381 

382 Call this method after the final `encrypt` (the one with no arguments) 

383 to obtain the MAC tag. 

384 

385 The MAC tag is needed by the receiver to determine authenticity 

386 of the message. 

387 

388 :Return: the MAC, as a byte string. 

389 """ 

390 

391 if self.digest not in self._next: 

392 raise TypeError("digest() cannot be called now for this cipher") 

393 

394 assert(len(self._cache_P) == 0) 

395 

396 self._next = [self.digest] 

397 

398 if self._mac_tag is None: 

399 self._compute_mac_tag() 

400 

401 return self._mac_tag

403 def hexdigest(self): 

404 """Compute the *printable* MAC tag. 

405 

406 This method is like `digest`. 

407 

408 :Return: the MAC, as a hexadecimal string. 

409 """ 

410 return "".join(["%02x" % bord(x) for x in self.digest()])

412 def verify(self, received_mac_tag): 

413 """Validate the *binary* MAC tag. 

414 

415 Call this method after the final `decrypt` (the one with no arguments) 

416 to check if the message is authentic and valid. 

417 

418 :Parameters: 

419 received_mac_tag : bytes/bytearray/memoryview 

420 This is the *binary* MAC, as received from the sender. 

421 :Raises ValueError: 

422 if the MAC does not match. The message has been tampered with 

423 or the key is incorrect. 

424 """ 

425 

426 if self.verify not in self._next: 

427 raise TypeError("verify() cannot be called now for this cipher") 

428 

429 assert(len(self._cache_P) == 0) 

430 

431 self._next = [self.verify] 

432 

433 if self._mac_tag is None: 

434 self._compute_mac_tag() 

435 

436 secret = get_random_bytes(16) 

437 mac1 = BLAKE2s.new(digest_bits=160, key=secret, data=self._mac_tag) 

438 mac2 = BLAKE2s.new(digest_bits=160, key=secret, data=received_mac_tag) 

439 

440 if mac1.digest() != mac2.digest(): 

441 raise ValueError("MAC check failed") 

442 

443 def hexverify(self, hex_mac_tag): 

444 """Validate the *printable* MAC tag. 

445 

446 This method is like `verify`. 

447 

448 :Parameters: 

449 hex_mac_tag : string 

450 This is the *printable* MAC, as received from the sender. 

451 :Raises ValueError: 

452 if the MAC does not match. The message has been tampered with 

453 or the key is incorrect. 

454 """ 

455 

456 self.verify(unhexlify(hex_mac_tag))

458 def encrypt_and_digest(self, plaintext): 

459 """Encrypt the message and create the MAC tag in one step. 

460 

461 :Parameters: 

462 plaintext : bytes/bytearray/memoryview 

463 The entire message to encrypt. 

464 :Return: 

465 a tuple with two byte strings: 

466 

467 - the encrypted data 

468 - the MAC 

469 """ 

470 

471 return self.encrypt(plaintext) + self.encrypt(), self.digest()

473 def decrypt_and_verify(self, ciphertext, received_mac_tag): 

474 """Decrypted the message and verify its authenticity in one step. 

475 

476 :Parameters: 

477 ciphertext : bytes/bytearray/memoryview 

478 The entire message to decrypt. 

479 received_mac_tag : byte string 

480 This is the *binary* MAC, as received from the sender. 

481 

482 :Return: the decrypted data (byte string). 

483 :Raises ValueError: 

484 if the MAC does not match. The message has been tampered with 

485 or the key is incorrect. 

486 """ 

487 

488 plaintext = self.decrypt(ciphertext) + self.decrypt() 

489 self.verify(received_mac_tag) 

490 return plaintext

493def _create_ocb_cipher(factory, **kwargs): 

494 """Create a new block cipher, configured in OCB mode. 

495 

496 :Parameters: 

497 factory : module 

498 A symmetric cipher module from `Crypto.Cipher` 

499 (like `Crypto.Cipher.AES`). 

500 

501 :Keywords: 

502 nonce : bytes/bytearray/memoryview 

503 A value that must never be reused for any other encryption. 

504 Its length can vary from 1 to 15 bytes. 

505 If not specified, a random 15 bytes long nonce is generated. 

506 

507 mac_len : integer 

508 Length of the MAC, in bytes. 

509 It must be in the range ``[8..16]``. 

510 The default is 16 (128 bits). 

511 

512 Any other keyword will be passed to the underlying block cipher. 

513 See the relevant documentation for details (at least ``key`` will need 

514 to be present). 

515 """ 

516 

517 try: 

518 nonce = kwargs.pop("nonce", None) 

519 if nonce is None: 

520 nonce = get_random_bytes(15) 

521 mac_len = kwargs.pop("mac_len", 16) 

522 except KeyError as e: 

523 raise TypeError("Keyword missing: " + str(e)) 

524 

525 return OcbMode(factory, nonce, mac_len, kwargs)
