21class EventDispatcher(object): 

22 """Dispatches event messages. 

23 

24 Arguments: 

25 connection (kombu.Connection): Connection to the broker. 

26 

27 hostname (str): Hostname to identify ourselves as, 

28 by default uses the hostname returned by 

29 :func:`~celery.utils.anon_nodename`. 

30 

31 groups (Sequence[str]): List of groups to send events for. 

32 :meth:`send` will ignore send requests to groups not in this list. 

33 If this is :const:`None`, all events will be sent. 

34 Example groups include ``"task"`` and ``"worker"``. 

35 

36 enabled (bool): Set to :const:`False` to not actually publish any 

37 events, making :meth:`send` a no-op. 

38 

39 channel (kombu.Channel): Can be used instead of `connection` to specify 

40 an exact channel to use when sending events. 

41 

42 buffer_while_offline (bool): If enabled events will be buffered 

43 while the connection is down. :meth:`flush` must be called 

44 as soon as the connection is re-established. 

45 

46 Note: 

47 You need to :meth:`close` this after use. 

48 """

118 def publish(self, type, fields, producer, 

119 blind=False, Event=Event, **kwargs): 

120 """Publish event using custom :class:`~kombu.Producer`. 

121 

122 Arguments: 

123 type (str): Event type name, with group separated by dash (`-`). 

124 fields: Dictionary of event fields, must be json serializable. 

125 producer (kombu.Producer): Producer instance to use: 

126 only the ``publish`` method will be called. 

127 retry (bool): Retry in the event of connection failure. 

128 retry_policy (Mapping): Map of custom retry policy options. 

129 See :meth:`~kombu.Connection.ensure`. 

130 blind (bool): Don't set logical clock value (also don't forward 

131 the internal logical clock). 

132 Event (Callable): Event type used to create event. 

133 Defaults to :func:`Event`. 

134 utcoffset (Callable): Function returning the current 

135 utc offset in hours. 

136 """ 

137 clock = None if blind else self.clock.forward()

139 pid=self.pid, clock=clock, **fields) 

140 with self.mutex:

164 def send(self, type, blind=False, utcoffset=utcoffset, retry=False, 

165 retry_policy=None, Event=Event, **fields): 

166 """Send event. 

167 

168 Arguments: 

169 type (str): Event type name, with group separated by dash (`-`). 

170 retry (bool): Retry in the event of connection failure. 

171 retry_policy (Mapping): Map of custom retry policy options. 

172 See :meth:`~kombu.Connection.ensure`. 

173 blind (bool): Don't set logical clock value (also don't forward 

174 the internal logical clock). 

175 Event (Callable): Event type used to create event, 

176 defaults to :func:`Event`. 

177 utcoffset (Callable): unction returning the current utc offset 

178 in hours. 

179 **fields (Any): Event fields -- must be json serializable. 

180 """ 

181 if self.enabled: 

182 groups, group = self.groups, group_from(type) 

183 if groups and group not in groups: 

184 return 

185 if group in self.buffer_group: 

186 clock = self.clock.forward() 

187 event = Event(type, hostname=self.hostname, 

188 utcoffset=utcoffset(), 

189 pid=self.pid, clock=clock, **fields) 

190 buf = self._group_buffer[group] 

191 buf.append(event) 

192 if len(buf) >= self.buffer_limit: 

193 self.flush() 

194 elif self.on_send_buffered: 

195 self.on_send_buffered() 

196 else:

198 Event=Event, retry=retry, 

199 retry_policy=retry_policy)

201 def flush(self, errors=True, groups=True): 

202 """Flush the outbound buffer.""" 

203 if errors: 

204 buf = list(self._outbound_buffer) 

205 try: 

206 with self.mutex: 

207 for event, routing_key, _ in buf: 

208 self._publish(event, self.producer, routing_key) 

209 finally: 

210 self._outbound_buffer.clear() 

211 if groups: 

212 with self.mutex: 

213 for group, events in items(self._group_buffer):
