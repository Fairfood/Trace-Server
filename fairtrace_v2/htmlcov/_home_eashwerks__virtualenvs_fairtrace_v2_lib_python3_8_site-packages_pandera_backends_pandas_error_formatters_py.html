63def reshape_failure_cases( 

64 failure_cases: Union[pd.DataFrame, pd.Series], ignore_na: bool = True 

65) -> pd.DataFrame: 

66 """Construct readable error messages for vectorized_error_message. 

67 

68 :param failure_cases: The failure cases encountered by the element-wise 

69 or vectorized validator. 

70 :param ignore_na: whether or not to ignore null failure cases. 

71 :returns: DataFrame where index contains failure cases, the "index" 

72 column contains a list of integer indexes in the validation 

73 DataFrame that caused the failure, and a "count" column 

74 representing how many failures of that case occurred. 

75

80 if not (is_table(failure_cases) or is_field(failure_cases)):

84 )

86 if (

88 and "column" in failure_cases.columns 

89 and "failure_case" in failure_cases.columns

91 reshaped_failure_cases = failure_cases 

92 elif is_table(failure_cases) and is_multiindex(failure_cases.index): 

93 reshaped_failure_cases = ( 

94 failure_cases.rename_axis("column", axis=1) # type: ignore[call-overload] 

95 .assign( 

96 index=lambda df: ( 

97 df.index.to_frame().apply(tuple, axis=1).astype(str)

145def consolidate_failure_cases(schema_errors: List[SchemaError]): 

146 """Consolidate schema error dicts to produce data for error message.""" 

147 assert schema_errors, ( 

148 "schema_errors input cannot be empty. Check how the backend " 

149 "validation logic is handling/raising SchemaError(s)." 

150 ) 

151 check_failure_cases = [] 

152 

153 column_order = [ 

154 "schema_context", 

155 "column", 

156 "check", 

157 "check_number", 

158 "failure_case", 

159 "index", 

160 ] 

161 

162 for schema_error in schema_errors: 

163 err, reason_code = schema_error, schema_error.reason_code 

164 

165 check_identifier = ( 

166 None 

167 if err.check is None 

168 else err.check 

169 if isinstance(err.check, str) 

170 else err.check.error 

171 if err.check.error is not None 

172 else err.check.name 

173 if err.check.name is not None 

174 else str(err.check) 

175 ) 

176 

177 if err.failure_cases is not None: 

178 if "column" in err.failure_cases: 

179 column = err.failure_cases["column"] 

180 else: 

181 column = ( 

182 err.schema.name 

183 if reason_code 

184 in { 

185 SchemaErrorReason.SCHEMA_COMPONENT_CHECK, 

186 SchemaErrorReason.DATAFRAME_CHECK, 

187 } 

188 else None 

189 ) 

190 

191 failure_cases = err.failure_cases.assign( 

192 schema_context=err.schema.__class__.__name__, 

193 check=check_identifier, 

194 check_number=err.check_index, 

195 # if the column key is a tuple (for MultiIndex column 

196 # names), explicitly wrap `column` in a list of the 

197 # same length as the number of failure cases. 

198 column=( 

199 [column] * err.failure_cases.shape[0] 

200 if isinstance(column, tuple) 

201 else column 

202 ), 

203 ) 

204 check_failure_cases.append(failure_cases[column_order]) 

205 

206 # NOTE: this is a hack to support pyspark.pandas and modin 

207 concat_fn = pd.concat # type: ignore 

208 if any( 

209 type(x).__module__.startswith("pyspark.pandas") 

210 for x in check_failure_cases 

211 ): 

212 # pylint: disable=import-outside-toplevel 

213 import pyspark.pandas as ps 

214 

215 concat_fn = ps.concat # type: ignore 

216 check_failure_cases = [ 

217 x if isinstance(x, ps.DataFrame) else ps.DataFrame(x) 

218 for x in check_failure_cases 

219 ] 

220 elif any( 

221 type(x).__module__.startswith("modin.pandas") 

222 for x in check_failure_cases 

223 ): 

224 # pylint: disable=import-outside-toplevel 

225 import modin.pandas as mpd 

226 

227 concat_fn = mpd.concat # type: ignore 

228 check_failure_cases = [ 

229 x if isinstance(x, mpd.DataFrame) else mpd.DataFrame(x) 

230 for x in check_failure_cases 

231 ] 

232 return concat_fn(check_failure_cases).reset_index(drop=True) 

233 

234 return ( 

235 concat_fn(check_failure_cases) 

236 .reset_index(drop=True) 

237 .sort_values("schema_context", ascending=False) 

238 )

258def summarize_failure_cases( 

259 schema_name: str, 

260 schema_errors: List[SchemaError], 

261 failure_cases: pd.DataFrame, 

262) -> Tuple[str, Dict[str, int]]: 

263 """Format error message.""" 

264 

265 error_counts = defaultdict(int) # type: ignore 

266 for schema_error in schema_errors: 

267 reason_code = schema_error.reason_code 

268 error_counts[reason_code] += 1

280 def agg_failure_cases(df): 

281 # Note: hack to support unhashable types, proper solution that only transforms 

282 # when requires https://github.com/unionai-oss/pandera/issues/260 

283 df.failure_case = df.failure_case.astype(str) 

284 # NOTE: this is a hack to add modin support 

285 if type(df).__module__.startswith("modin.pandas"): 

286 return ( 

287 df.groupby(["schema_context", "column", "check"]) 

288 .agg({"failure_case": "unique"}) 

289 .failure_case 

290 ) 

291 return df.groupby( 

292 ["schema_context", "column", "check"] 

293 ).failure_case.unique() 

294 

295 summarized_failure_cases = ( 

296 failure_cases.fillna({"column": "<NA>"}) 

297 .pipe(agg_failure_cases) 

298 .rename("failure_cases") 

299 .to_frame() 

300 .assign(n_failure_cases=lambda df: df.failure_cases.map(len)) 

301 ) 

302 index_labels = [ 

303 summarized_failure_cases.index.names.index(name) 

304 for name in ["schema_context", "column"] 

305 ] 

306 summarized_failure_cases = summarized_failure_cases.sort_index( 

307 level=index_labels, 

308 ascending=[False, True], 

309 ) 

310 msg += "\nSchema Error Summary" 

311 msg += "\n--------------------\n" 

312 with pd.option_context("display.max_colwidth", 100): 

313 msg += summarized_failure_cases.to_string() 

314 msg += SCHEMA_ERRORS_SUFFIX 

315 return msg, error_counts
