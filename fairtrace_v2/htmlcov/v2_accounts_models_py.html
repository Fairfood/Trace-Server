620 """Function to create a unique token for a device. 

621 

622 This function will create a token for the device if force create is 

623 true or retrieve the existing token and refresh it 

624 

625 refresh: refresh on each login, or just get or create token on 

626 each login 

627 """ 

628 response = {"is_granted": True, "token": ""} 

629 if not force_create: 

630 try: 

631 token = AccessToken.objects.get( 

632 device=self, user=self.user, device__active=True 

633 ) 

634 # refresh token only if refresh is True. 

635 response["token"] = ( 

636 token.key if not refresh else token.refresh() 

637 ) 

638 except Exception as exc: 

639 expected = (ObjectDoesNotExist, MultipleObjectsReturned) 

640 if not isinstance(exc, expected): 

641 raise exc 

642 token_exists = AccessToken.objects.filter( 

643 user=self.user, device__active=True 

644 ).exists() 

645 # new token will be issued if not refresh enabled. 

646 if not token_exists or not refresh: 

647 response["token"] = self.user.issue_access_token(self) 

648 else: 

649 response["is_granted"] = False 

650 return response 

651 

652 self.user.last_login = timezone.now() 

653 self.user.save() 

654 return response

656 old_mobile_tokens = AccessToken.objects.filter( 

657 user=self.user, device__type__in=MOBILE_DEVICE_TYPES 

658 ) 

659 old_mobile_tokens.delete() 

660 response["token"] = self.user.issue_access_token(self)
