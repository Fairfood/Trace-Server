19 def __init__(self, http, data_path, seconds_per_quota=100, retries=1, logger=logging.getLogger(__name__), 

20 check=True): 

21 """A wrapper class for the Google Sheets API v4. 

22 

23 All calls to the the API are made in this class. This ensures that the quota is never hit. 

24 

25 The default quota for the API is 100 requests per 100 seconds. Each request is made immediately and counted. 

26 When 100 seconds have passed the counter is reset. Should the counter reach 101 the request is delayed until seconds_per_quota 

27 seconds since the first request pass. 

28 

29 :param http: The http object used to execute the requests. 

30 :param data_path: Where the discovery json file is stored. 

31 :param seconds_per_quota: Default value is 100 seconds 

32 :param retries: How often the requests will be repeated if the connection times out. (Default 1) 

33 :param check: Check for quota error and apply rate limiting. 

34 :param logger: 

35 """ 

36 

37 self.logger = logger 

38 try: 

39 with open(os.path.join(data_path, "sheets_discovery.json")) as jd: 

40 self.service = discovery.build_from_document(json.load(jd), http=http) 

41 except: 

42 self.service = discovery.build('sheets', 'v4', http=http) 

43 self.retries = retries 

44 self.seconds_per_quota = seconds_per_quota 

45 self.check = check 

46 self.batch_mode = False 

47 self.batched_requests = dict()

60 def batch_update(self, spreadsheet_id, requests, **kwargs): 

61 """ 

62 Applies one or more updates to the spreadsheet. 

63 

64 Each request is validated before being applied. If any request is not valid then the entire request will 

65 fail and nothing will be applied. 

66 

67 Some requests have replies to give you some information about how they are applied. The replies will mirror 

68 the requests. For example, if you applied 4 updates and the 3rd one had a reply, then the response will have 

69 2 empty replies, the actual reply, and another empty reply, in that order. 

70 

71 Due to the collaborative nature of spreadsheets, it is not guaranteed that the spreadsheet will reflect exactly 

72 your changes after this completes, however it is guaranteed that the updates in the request will be applied 

73 together atomically. Your changes may be altered with respect to collaborator changes. If there are no 

74 collaborators, the spreadsheet should reflect your changes. 

75 

76 +-----------------------------------+-----------------------------------------------------+ 

77 | Request body params | Description | 

78 +===================================+=====================================================+ 

79 | includeSpreadsheetInResponse | | Determines if the update response should include | 

80 | | | the spreadsheet resource. (default: False) | 

81 +-----------------------------------+-----------------------------------------------------+ 

82 | responseRanges[] | | Limits the ranges included in the response | 

83 | | | spreadsheet. Only applied if the first param is | 

84 | | | True. | 

85 +-----------------------------------+-----------------------------------------------------+ 

86 | responseIncludeGridData | | True if grid data should be returned. Meaningful | 

87 | | | only if if includeSpreadsheetInResponse is 'true'.| 

88 | | | This parameter is ignored if a field mask was set | 

89 | | | in the request. | 

90 +-----------------------------------+-----------------------------------------------------+ 

91 

92 :param spreadsheet_id: The spreadsheet to apply the updates to. 

93 :param requests: A list of updates to apply to the spreadsheet. Requests will be applied in the order 

94 they are specified. If any request is not valid, no requests will be applied. 

95 :param kwargs: Request body params & standard parameters (see reference for details). 

96 :return: 

97 """ 

98 

99 if not isinstance(requests, list): 

100 requests = [requests] 

101 

102 if self.batch_mode: 

103 if spreadsheet_id in self.batched_requests: 

104 self.batched_requests[spreadsheet_id].extend(requests) 

105 else: 

106 self.batched_requests[spreadsheet_id] = requests 

107 return 

108 

109 body = {'requests': requests} 

110 for param in ['includeSpreadsheetInResponse', 'responseRanges', 'responseIncludeGridData']: 

111 if param in kwargs: 

112 body[param] = kwargs[param] 

113 del kwargs[param] 

114 if 'fields' not in kwargs: 

115 kwargs['fields'] = '*' 

116 

117 request = self.service.spreadsheets().batchUpdate(spreadsheetId=spreadsheet_id, 

118 body=body, **kwargs) 

119 return self._execute_requests(request)

327 def values_batch_update(self, spreadsheet_id, body, parse=True): 

328 """ 

329 Impliments batch update 

330 

331 :param spreadsheet_id: id of spreadsheet 

332 :param body: body of request 

333 :param parse: 

334 """ 

335 cformat = 'USER_ENTERED' if parse else 'RAW' 

336 batch_limit = GOOGLE_SHEET_CELL_UPDATES_LIMIT 

337 lengths = [len(x) for x in body['values']] 

338 avg_row_length = (min(lengths) + max(lengths))/2 

339 avg_row_length = 1 if avg_row_length == 0 else avg_row_length 

340 if body['majorDimension'] == 'ROWS': 

341 batch_length = int(batch_limit / avg_row_length) # num of rows to include in a batch 

342 num_rows = len(body['values']) 

343 else: 

344 batch_length = int(batch_limit / len(body['values'])) # num of rows to include in a batch 

345 num_rows = len(body['values'][0]) 

346 if len(body['values']) * len(body['values'][0]) <= batch_limit: 

347 request = self.service.spreadsheets().values().update(spreadsheetId=spreadsheet_id, 

348 range=body['range'], 

349 valueInputOption=cformat, body=body) 

350 self._execute_requests(request) 

351 else: 

352 if batch_length == 0: 

353 raise AssertionError("num_columns < " + str(GOOGLE_SHEET_CELL_UPDATES_LIMIT)) 

354 values = body['values'] 

355 title, value_range = body['range'].split('!') 

356 value_range_start, value_range_end = value_range.split(':') 

357 value_range_end = list(format_addr(str(value_range_end), output='tuple')) 

358 value_range_start = list(format_addr(str(value_range_start), output='tuple')) 

359 max_rows = value_range_end[0] 

360 start_row = value_range_start[0] 

361 for batch_start in range(0, num_rows, batch_length): 

362 if body['majorDimension'] == 'ROWS': 

363 body['values'] = values[batch_start:batch_start + batch_length] 

364 else: 

365 body['values'] = [col[batch_start:batch_start + batch_length] for col in values] 

366 value_range_start[0] = batch_start + start_row 

367 value_range_end[0] = min(batch_start + batch_length, max_rows) + start_row 

368 body['range'] = title + '!' + format_addr(tuple(value_range_start), output='label') + ':' + \ 

369 format_addr(tuple(value_range_end), output='label') 

370 request = self.service.spreadsheets().values().update(spreadsheetId=spreadsheet_id, body=body, 

371 range=body['range'], 

372 valueInputOption=cformat) 

373 self._execute_requests(request)

486 def _execute_requests(self, request): 

487 """Execute a request to the Google Sheets API v4. 

488 

489 When the API returns a 429 Error will sleep for the specified time and try again. 

490 

491 :param request: The request to be made. 

492 :return: Response 

493 """ 

494 try: 

495 response = request.execute(num_retries=self.retries) 

496 except HttpError as error: 

497 if error.resp['status'] == '429' and self.check: 

498 time.sleep(self.seconds_per_quota) # TODO use asyncio 

499 response = request.execute(num_retries=self.retries) 

500 else: 

501 raise 

502 return response
