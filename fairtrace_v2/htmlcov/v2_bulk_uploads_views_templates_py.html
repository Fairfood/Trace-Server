216 @staticmethod 

217 def find_header_row(file): 

218 """ 

219 Find the header row in an Excel file. 

220 

221 This function takes an Excel file and searches for the row where the 

222 header information is located. The header row is determined by finding 

223 columns with at least one non-empty cell and then searching for the 

224 first row with all non-empty cells in those columns. 

225 

226 Parameters: 

227 file (str): The path to the Excel file. 

228 

229 Returns: 

230 int or None: The row number where the header is found, or None if the 

231 header is not found. 

232 """ 

233 workbook = openpyxl.load_workbook(file) 

234 sheet = workbook.active 

235 

236 # Find the dimensions of the sheet (number of rows and columns) 

237 max_row = sheet.max_row 

238 max_column = sheet.max_column 

239 

240 # Find columns with at least one non-empty cell 

241 non_empty_columns = [] 

242 for col in range(1, max_column + 1): 

243 column_data = [sheet.cell(row=row, column=col).value for row in 

244 range(1, max_row + 1)] 

245 if any(cell_value is not None for cell_value in column_data): 

246 non_empty_columns.append(col) 

247 

248 # If no non-empty columns are found, return None 

249 if not non_empty_columns: 

250 return None

252 def is_empty_row(row, columns): 

253 """ 

254 Find the first row with all non-empty cells in the non-empty 

255 columns. 

256 """ 

257 if not columns: 

258 return None 

259 for row in range(1, max_row + 1): 

260 if all(sheet.cell(row=row, column=_col).value is not None 

261 for _col in columns): 

262 return row 

263 return is_empty_row( 

264 row, columns[1:]) # Recursively call is_empty_row 

265 

266 return is_empty_row(1, non_empty_columns)

277 Parameters: 

278 file (BinaryIO): The binary file object of the Excel file to be read.

281 List[List]: A 2D array representing the data from the Excel file.

283 """ 

284 

285 # Load the Excel workbook 

286 try: 

287 workbook = load_workbook(file) 

288 except InvalidFileException: 

289 raise InvalidFileException( 

290 "The provided file is not a valid Excel file.") 

291 

292 # Get the active sheet from the workbook 

293 sheet = workbook.active 

294 

295 # Read data from the sheet and store it in a 2D array 

296 data_2d_array = [] 

297 for row in sheet.iter_rows(values_only=True): 

298 data_2d_array.append(list(row)) 

299 

300 last_row_index = None 

301 # Find last row with at least one non-null value 

302 for index, row in enumerate(data_2d_array): 

303 if any(cell is not None for cell in row): 

304 last_row_index = index 

305 

306 last_column_index = None 

307 # Find last column with at least one non-null value 

308 for i in range(len(data_2d_array[0])): 

309 if any(row[i] is not None for row in data_2d_array): 

310 last_column_index = i 

311 

312 return [ 

313 row[:last_column_index + 1] 

314 for row in data_2d_array[:10]]
