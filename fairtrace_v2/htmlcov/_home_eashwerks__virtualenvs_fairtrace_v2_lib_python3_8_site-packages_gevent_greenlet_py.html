140def _extract_stack(limit): 

141 try: 

142 frame = sys_getframe() 

143 except ValueError: 

144 # In certain embedded cases that directly use the Python C api 

145 # to call Greenlet.spawn (e.g., uwsgi) this can raise 

146 # `ValueError: call stack is not deep enough`. This is because 

147 # the Cython stack frames for Greenlet.spawn -> 

148 # Greenlet.__init__ -> _extract_stack are all on the C level, 

149 # not the Python level. 

150 # See https://github.com/gevent/gevent/issues/1212 

151 frame = None 

152 

153 newest_Frame = None 

154 newer_Frame = None 

155 

156 while limit and frame is not None: 

157 limit -= 1 

158 older_Frame = _Frame() 

159 # Arguments are always passed to the constructor as Python objects, 

160 # meaning we wind up boxing the f_lineno just to unbox it if we pass it. 

161 # It's faster to simply assign once the object is created. 

162 older_Frame.f_code = Gevent_PyFrame_GetCode(frame) # pylint:disable=undefined-variable 

163 older_Frame.f_lineno = Gevent_PyFrame_GetLineNumber(frame) # pylint:disable=undefined-variable 

164 if newer_Frame is not None: 

165 newer_Frame.f_back = older_Frame 

166 newer_Frame = older_Frame 

167 if newest_Frame is None: 

168 newest_Frame = newer_Frame 

169 

170 frame = Gevent_PyFrame_GetBack(frame) # pylint:disable=undefined-variable 

171 

172 return newest_Frame

460 def __handle_death_before_start(self, args): 

461 # args is (t, v, tb) or simply t or v. 

462 # The last two cases are transformed into (t, v, None); 

463 # if the single argument is an exception type, a new instance 

464 # is created; if the single argument is not an exception type and also 

465 # not an exception, it is wrapped in a BaseException (this is not 

466 # documented, but should result in better behaviour in the event of a 

467 # user error---instead of silently printing something to stderr, we still 

468 # kill the greenlet). 

469 if self._exc_info is None and self.dead: 

470 # the greenlet was never switched to before and it will 

471 # never be; _report_error was not called, the result was 

472 # not set, and the links weren't notified. Let's do it 

473 # here. 

474 # 

475 # checking that self.dead is true is essential, because 

476 # throw() does not necessarily kill the greenlet (if the 

477 # exception raised by throw() is caught somewhere inside 

478 # the greenlet). 

479 if len(args) == 1: 

480 arg = args[0] 

481 if isinstance(arg, type) and issubclass(arg, BaseException): 

482 args = (arg, arg(), None) 

483 else: 

484 args = (type(arg), arg, None) 

485 elif not args: 

486 args = (GreenletExit, GreenletExit(), None) 

487 if not issubclass(args[0], BaseException): 

488 # Random non-type, non-exception arguments. 

489 print("RANDOM CRAP", args) 

490 import traceback; traceback.print_stack() 

491 args = (BaseException, BaseException(args), None) 

492 assert issubclass(args[0], BaseException) 

493 self.__report_error(args)

596 def throw(self, *args): 

597 """Immediately switch into the greenlet and raise an exception in it. 

598 

599 Should only be called from the HUB, otherwise the current greenlet is left unscheduled forever. 

600 To raise an exception in a safe manner from any greenlet, use :meth:`kill`. 

601 

602 If a greenlet was started but never switched to yet, then also 

603 a) cancel the event that will start it 

604 b) fire the notifications as if an exception was raised in a greenlet 

605 """ 

606 self.__cancel_start() 

607 

608 try: 

609 if not self.dead: 

610 # Prevent switching into a greenlet *at all* if we had never 

611 # started it. Usually this is the same thing that happens by throwing, 

612 # but if this is done from the hub with nothing else running, prevents a 

613 # LoopExit. 

614 greenlet.throw(self, *args) 

615 finally:

728 def kill(self, exception=GreenletExit, block=True, timeout=None): 

729 """ 

730 Raise the ``exception`` in the greenlet. 

731 

732 If ``block`` is ``True`` (the default), wait until the greenlet 

733 dies or the optional timeout expires; this may require switching 

734 greenlets. 

735 If block is ``False``, the current greenlet is not unscheduled. 

736 

737 This function always returns ``None`` and never raises an error. It 

738 may be called multpile times on the same greenlet object, and may be 

739 called on an unstarted or dead greenlet. 

740 

741 .. note:: 

742 

743 Depending on what this greenlet is executing and the state 

744 of the event loop, the exception may or may not be raised 

745 immediately when this greenlet resumes execution. It may 

746 be raised on a subsequent green call, or, if this greenlet 

747 exits before making such a call, it may not be raised at 

748 all. As of 1.1, an example where the exception is raised 

749 later is if this greenlet had called :func:`sleep(0) 

750 <gevent.sleep>`; an example where the exception is raised 

751 immediately is if this greenlet had called 

752 :func:`sleep(0.1) <gevent.sleep>`. 

753 

754 .. caution:: 

755 

756 Use care when killing greenlets. If the code executing is not 

757 exception safe (e.g., makes proper use of ``finally``) then an 

758 unexpected exception could result in corrupted state. Using 

759 a :meth:`link` or :meth:`rawlink` (cheaper) may be a safer way to 

760 clean up resources. 

761 

762 See also :func:`gevent.kill` and :func:`gevent.killall`. 

763 

764 :keyword type exception: The type of exception to raise in the greenlet. The default 

765 is :class:`GreenletExit`, which indicates a :meth:`successful` completion 

766 of the greenlet. 

767 

768 .. versionchanged:: 0.13.0 

769 *block* is now ``True`` by default. 

770 .. versionchanged:: 1.1a2 

771 If this greenlet had never been switched to, killing it will 

772 prevent it from *ever* being switched to. Links (:meth:`rawlink`) 

773 will still be executed, though. 

774 .. versionchanged:: 20.12.1 

775 If this greenlet is :meth:`ready`, immediately return instead of 

776 requiring a trip around the event loop. 

777 """ 

778 if not self._maybe_kill_before_start(exception): 

779 if self.ready(): 

780 return 

781 

782 waiter = Waiter() if block else None # pylint:disable=undefined-variable 

783 hub = get_my_hub(self) # pylint:disable=undefined-variable 

784 hub.loop.run_callback(_kill, self, exception, waiter) 

785 if waiter is not None: 

786 waiter.get() 

787 self.join(timeout)

789 def get(self, block=True, timeout=None): 

790 """ 

791 get(block=True, timeout=None) -> object 

792 

793 Return the result the greenlet has returned or re-raise the 

794 exception it has raised. 

795 

796 If block is ``False``, raise :class:`gevent.Timeout` if the 

797 greenlet is still alive. If block is ``True``, unschedule the 

798 current greenlet until the result is available or the timeout 

799 expires. In the latter case, :class:`gevent.Timeout` is 

800 raised. 

801 """ 

802 if self.ready(): 

803 if self.successful(): 

804 return self.value 

805 self._raise_exception() 

806 if not block: 

807 raise Timeout() 

808 

809 switch = getcurrent().switch # pylint:disable=undefined-variable 

810 self.rawlink(switch) 

811 try: 

812 t = Timeout._start_new_or_dummy(timeout) 

813 try: 

814 result = get_my_hub(self).switch() # pylint:disable=undefined-variable 

815 if result is not self: 

816 raise InvalidSwitchError('Invalid switch into Greenlet.get(): %r' % (result, )) 

817 finally: 

818 t.cancel() 

819 except:

821 # if switch occurred normally then link was already removed in _notify_links 

822 # and there's no need to touch the links set. 

823 # Note, however, that if "Invalid switch" assert was removed and invalid switch 

824 # did happen, the link would remain, causing another invalid switch later in this greenlet. 

825 self.unlink(switch) 

826 raise 

827 

828 if self.ready(): 

829 if self.successful(): 

830 return self.value 

831 self._raise_exception()

1119def killall(greenlets, exception=GreenletExit, block=True, timeout=None): 

1120 """ 

1121 Forceably terminate all the *greenlets* by causing them to raise *exception*. 

1122 

1123 .. caution:: Use care when killing greenlets. If they are not prepared for exceptions, 

1124 this could result in corrupted state. 

1125 

1126 :param greenlets: A **bounded** iterable of the non-None greenlets to terminate. 

1127 *All* the items in this iterable must be greenlets that belong to the same hub, 

1128 which should be the hub for this current thread. If this is a generator or iterator 

1129 that switches greenlets, the results are undefined. 

1130 :keyword exception: The type of exception to raise in the greenlets. By default this is 

1131 :class:`GreenletExit`. 

1132 :keyword bool block: If True (the default) then this function only returns when all the 

1133 greenlets are dead; the current greenlet is unscheduled during that process. 

1134 If greenlets ignore the initial exception raised in them, 

1135 then they will be joined (with :func:`gevent.joinall`) and allowed to die naturally. 

1136 If False, this function returns immediately and greenlets will raise 

1137 the exception asynchronously. 

1138 :keyword float timeout: A time in seconds to wait for greenlets to die. If given, it is 

1139 only honored when ``block`` is True. 

1140 :raise Timeout: If blocking and a timeout is given that elapses before 

1141 all the greenlets are dead. 

1142 

1143 .. versionchanged:: 1.1a2 

1144 *greenlets* can be any iterable of greenlets, like an iterator or a set. 

1145 Previously it had to be a list or tuple. 

1146 .. versionchanged:: 1.5a3 

1147 Any :class:`Greenlet` in the *greenlets* list that hadn't been switched to before 

1148 calling this method will never be switched to. This makes this function 

1149 behave like :meth:`Greenlet.kill`. This does not apply to raw greenlets. 

1150 .. versionchanged:: 1.5a3 

1151 Now accepts raw greenlets created by :func:`gevent.spawn_raw`. 

1152 """ 

1153 

1154 need_killed = [] 

1155 for glet in greenlets: 

1156 # Quick pass through to prevent any greenlet from 

1157 # actually being switched to if it hasn't already. 

1158 # (Previously we called ``list(greenlets)`` so we're still 

1159 # linear.) 

1160 # 

1161 # We don't use glet.kill() here because we don't want to schedule 

1162 # any callbacks in the loop; we're about to handle that more directly. 

1163 try: 

1164 cancel = glet._maybe_kill_before_start 

1165 except AttributeError: 

1166 need_killed.append(glet) 

1167 else: 

1168 if not cancel(exception): 

1169 need_killed.append(glet) 

1170 

1171 if not need_killed: 

1172 return 

1173 

1174 loop = glet.loop # pylint:disable=undefined-loop-variable 

1175 if block: 

1176 waiter = Waiter() # pylint:disable=undefined-variable 

1177 loop.run_callback(_killall3, need_killed, exception, waiter) 

1178 t = Timeout._start_new_or_dummy(timeout) 

1179 try: 

1180 alive = waiter.get() 

1181 if alive: 

1182 joinall(alive, raise_error=False) 

1183 finally: 

1184 t.cancel() 

1185 else: 

1186 loop.run_callback(_killall, need_killed, exception)
