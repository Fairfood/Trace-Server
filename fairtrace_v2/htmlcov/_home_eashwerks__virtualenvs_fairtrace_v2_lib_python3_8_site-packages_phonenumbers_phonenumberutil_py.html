302def _create_extn_pattern(single_extn_symbols): 

303 """Helper initialiser method to create the regular-expression pattern to 

304 match extensions, allowing the one-char extension symbols provided by 

305 single_extn_symbols.""" 

306 # There are three regular expressions here. The first covers RFC 3966 

307 # format, where the extension is added using ";ext=". The second more 

308 # generic one starts with optional white space and ends with an optional 

309 # full stop (.), followed by zero or more spaces/tabs/commas and then the 

310 # numbers themselves. The other one covers the special case of American 

311 # numbers where the extension is written with a hash at the end, such as 

312 # "- 503#". Note that the only capturing groups should be around the 

313 # digits that you want to capture as part of the extension, or else 

314 # parsing will fail! Canonical-equivalence doesn't seem to be an option 

315 # with Android java, so we allow two options for representing the accented 

316 # o - the character itself, and one in the unicode decomposed form with 

317 # the combining acute accent. 

318 return (_RFC3966_EXTN_PREFIX + _CAPTURING_EXTN_DIGITS + u("|") + 

319 u("[ \u00A0\\t,]*(?:e?xt(?:ensi(?:o\u0301?|\u00F3))?n?|") + 

320 u("\uFF45?\uFF58\uFF54\uFF4E?|") + 

321 u("\u0434\u043e\u0431|") + u("[") + single_extn_symbols + u("]|int|anexo|\uFF49\uFF4E\uFF54)") + 

322 u("[:\\.\uFF0E]?[ \u00A0\\t,-]*") + _CAPTURING_EXTN_DIGITS + u("#?|") + 

323 u("[- ]+(") + _DIGITS + u("{1,5})#"))

509def _extract_possible_number(number): 

510 """Attempt to extract a possible number from the string passed in. 

511 

512 This currently strips all leading characters that cannot be used to 

513 start a phone number. Characters that can be used to start a phone number 

514 are defined in the VALID_START_CHAR_PATTERN. If none of these characters 

515 are found in the number passed in, an empty string is returned. This 

516 function also attempts to strip off any alternative extensions or endings 

517 if two or more are present, such as in the case of: (530) 583-6985 

518 x302/x2303. The second extension here makes this actually two phone 

519 numbers, (530) 583-6985 x302 and (530) 583-6985 x2303. We remove the 

520 second extension so that the first number is parsed correctly. 

521 

522 Arguments: 

523 number -- The string that might contain a phone number. 

524 

525 Returns the number, stripped of any non-phone-number prefix (such 

526 as "Tel:") or an empty string if no character used to start phone 

527 numbers (such as + or any digit) is found in the number 

528 """ 

529 match = _VALID_START_CHAR_PATTERN.search(number) 

530 if match: 

531 number = number[match.start():] 

532 # Remove trailing non-alpha non-numberical characters. 

533 trailing_chars_match = _UNWANTED_END_CHAR_PATTERN.search(number) 

534 if trailing_chars_match: 

535 number = number[:trailing_chars_match.start()] 

536 # Check for extra numbers at the end. 

537 second_number_match = _SECOND_NUMBER_START_PATTERN.search(number) 

538 if second_number_match: 

539 number = number[:second_number_match.start()] 

540 return number 

541 else: 

542 return U_EMPTY_STRING

566def _normalize(number): 

567 """Normalizes a string of characters representing a phone number. 

568 

569 This performs the following conversions: 

570 - Punctuation is stripped. 

571 - For ALPHA/VANITY numbers: 

572 - Letters are converted to their numeric representation on a telephone 

573 keypad. The keypad used here is the one defined in ITU 

574 Recommendation E.161. This is only done if there are 3 or more 

575 letters in the number, to lessen the risk that such letters are 

576 typos. 

577 - For other numbers: 

578 - Wide-ascii digits are converted to normal ASCII (European) digits. 

579 - Arabic-Indic numerals are converted to European numerals. 

580 - Spurious alpha characters are stripped. 

581 

582 Arguments: 

583 number -- string representing a phone number 

584 

585 Returns the normalized string version of the phone number. 

586 """ 

587 m = fullmatch(_VALID_ALPHA_PHONE_PATTERN, number) 

588 if m: 

589 return _normalize_helper(number, _ALPHA_PHONE_MAPPINGS, True) 

590 else: 

591 return normalize_digits_only(number)

594def normalize_digits_only(number, keep_non_digits=False): 

595 """Normalizes a string of characters representing a phone number. 

596 

597 This converts wide-ascii and arabic-indic numerals to European numerals, 

598 and strips punctuation and alpha characters (optional). 

599 

600 Arguments: 

601 number -- a string representing a phone number 

602 keep_non_digits -- whether to keep non-digits 

603 

604 Returns the normalized string version of the phone number. 

605 """ 

606 number = unicod(number) 

607 number_length = len(number) 

608 normalized_digits = U_EMPTY_STRING 

609 for ii in range(number_length): 

610 d = unicode_digit(number[ii], -1) 

611 if d != -1: 

612 normalized_digits += unicod(d) 

613 elif keep_non_digits: 

614 normalized_digits += number[ii] 

615 return normalized_digits

638def length_of_geographical_area_code(numobj): 

639 """Return length of the geographical area code for a number. 

640 

641 Gets the length of the geographical area code from the PhoneNumber object 

642 passed in, so that clients could use it to split a national significant 

643 number into geographical area code and subscriber number. It works in such 

644 a way that the resultant subscriber number should be diallable, at least 

645 on some devices. An example of how this could be used: 

646 

647 >>> import phonenumbers 

648 >>> numobj = phonenumbers.parse("16502530000", "US") 

649 >>> nsn = phonenumbers.national_significant_number(numobj) 

650 >>> ac_len = phonenumbers.length_of_geographical_area_code(numobj) 

651 >>> if ac_len > 0: 

652 ... area_code = nsn[:ac_len] 

653 ... subscriber_number = nsn[ac_len:] 

654 ... else: 

655 ... area_code = "" 

656 ... subscriber_number = nsn 

657 

658 N.B.: area code is a very ambiguous concept, so the I18N team generally 

659 recommends against using it for most purposes, but recommends using the 

660 more general national_number instead. Read the following carefully before 

661 deciding to use this method: 

662 

663 - geographical area codes change over time, and this method honors those 

664 changes; therefore, it doesn't guarantee the stability of the result it 

665 produces. 

666 - subscriber numbers may not be diallable from all devices (notably 

667 mobile devices, which typically require the full national_number to be 

668 dialled in most countries). 

669 - most non-geographical numbers have no area codes, including numbers 

670 from non-geographical entities. 

671 - some geographical numbers have no area codes. 

672 

673 Arguments: 

674 numobj -- The PhoneNumber object to find the length of the area code form. 

675 

676 Returns the length of area code of the PhoneNumber object passed in. 

677 """ 

678 metadata = PhoneMetadata.metadata_for_region(region_code_for_number(numobj), None) 

679 if metadata is None: 

680 return 0 

681 

682 # If a country doesn't use a national prefix, and this number doesn't have 

683 # an Italian leading zero, we assume it is a closed dialling plan with no 

684 # area codes. 

685 if metadata.national_prefix is None and not numobj.italian_leading_zero: 

686 return 0 

687 

688 ntype = number_type(numobj) 

689 country_code = numobj.country_code 

690 if (ntype == PhoneNumberType.MOBILE and 

691 (country_code in _GEO_MOBILE_COUNTRIES_WITHOUT_MOBILE_AREA_CODES)): 

692 # Note this is a rough heuristic; it doesn't cover Indonesia well, for 

693 # example, where area codes are present for some mobile phones but not 

694 # for others. We have no better way of representing this in the 

695 # metadata at this point. 

696 return 0 

697 

698 if not is_number_type_geographical(ntype, country_code): 

699 return 0 

700 

701 return length_of_national_destination_code(numobj)

704def length_of_national_destination_code(numobj): 

705 """Return length of the national destination code code for a number. 

706 

707 Gets the length of the national destination code (NDC) from the 

708 PhoneNumber object passed in, so that clients could use it to split a 

709 national significant number into NDC and subscriber number. The NDC of a 

710 phone number is normally the first group of digit(s) right after the 

711 country calling code when the number is formatted in the international 

712 format, if there is a subscriber number part that follows. 

713 

714 N.B.: similar to an area code, not all numbers have an NDC! 

715 

716 An example of how this could be used: 

717 

718 >>> import phonenumbers 

719 >>> numobj = phonenumbers.parse("18002530000", "US") 

720 >>> nsn = phonenumbers.national_significant_number(numobj) 

721 >>> ndc_len = phonenumbers.length_of_national_destination_code(numobj) 

722 >>> if ndc_len > 0: 

723 ... national_destination_code = nsn[:ndc_len] 

724 ... subscriber_number = nsn[ndc_len:] 

725 ... else: 

726 ... national_destination_code = "" 

727 ... subscriber_number = nsn 

728 

729 Refer to the unittests to see the difference between this function and 

730 length_of_geographical_area_code. 

731 

732 Arguments: 

733 numobj -- The PhoneNumber object to find the length of the NDC from. 

734 

735 Returns the length of NDC of the PhoneNumber object passed in, which 

736 could be zero. 

737 """ 

738 if numobj.extension is not None: 

739 # We don't want to alter the object given to us, but we don't want to 

740 # include the extension when we format it, so we copy it and clear the 

741 # extension here. 

742 copied_numobj = PhoneNumber() 

743 copied_numobj.merge_from(numobj) 

744 copied_numobj.extension = None 

745 else: 

746 copied_numobj = numobj 

747 

748 nsn = format_number(copied_numobj, PhoneNumberFormat.INTERNATIONAL) 

749 number_groups = re.split(NON_DIGITS_PATTERN, nsn) 

750 

751 # The pattern will start with "+COUNTRY_CODE " so the first group will 

752 # always be the empty string (before the + symbol) and the second group 

753 # will be the country calling code. The third group will be area code if 

754 # it is not the last group. 

755 if len(number_groups) <= 3: 

756 return 0 

757 

758 if number_type(numobj) == PhoneNumberType.MOBILE: 

759 # For example Argentinian mobile numbers, when formatted in the 

760 # international format, are in the form of +54 9 NDC XXXX... As a 

761 # result, we take the length of the third group (NDC) and add the 

762 # length of the second group (which is the mobile token), which also 

763 # forms part of the national significant number. This assumes that 

764 # the mobile token is always formatted separately from the rest of the 

765 # phone number. 

766 mobile_token = country_mobile_token(numobj.country_code) 

767 if mobile_token != U_EMPTY_STRING: 

768 return len(number_groups[2]) + len(number_groups[3]) 

769 return len(number_groups[2])

784def _normalize_helper(number, replacements, remove_non_matches): 

785 """Normalizes a string of characters representing a phone number by 

786 replacing all characters found in the accompanying map with the values 

787 therein, and stripping all other characters if remove_non_matches is true. 

788 

789 Arguments: 

790 number -- a string representing a phone number 

791 replacements -- a mapping of characters to what they should be replaced 

792 by in the normalized version of the phone number 

793 remove_non_matches -- indicates whether characters that are not able to be 

794 replaced should be stripped from the number. If this is False, 

795 they will be left unchanged in the number. 

796 

797 Returns the normalized string version of the phone number. 

798 """ 

799 normalized_number = [] 

800 for char in number: 

801 new_digit = replacements.get(char.upper(), None) 

802 if new_digit is not None: 

803 normalized_number.append(new_digit) 

804 elif not remove_non_matches: 

805 normalized_number.append(char) 

806 # If neither of the above are true, we remove this character 

807 return U_EMPTY_STRING.join(normalized_number)

939def format_number(numobj, num_format): 

940 """Formats a phone number in the specified format using default rules. 

941 

942 Note that this does not promise to produce a phone number that the user 

943 can dial from where they are - although we do format in either 'national' 

944 or 'international' format depending on what the client asks for, we do not 

945 currently support a more abbreviated format, such as for users in the same 

946 "area" who could potentially dial the number without area code. Note that 

947 if the phone number has a country calling code of 0 or an otherwise 

948 invalid country calling code, we cannot work out which formatting rules to 

949 apply so we return the national significant number with no formatting 

950 applied. 

951 

952 Arguments: 

953 numobj -- The phone number to be formatted. 

954 num_format -- The format the phone number should be formatted into 

955 

956 Returns the formatted phone number. 

957 """ 

958 if numobj.national_number == 0 and numobj.raw_input is not None: 

959 # Unparseable numbers that kept their raw input just use that. This 

960 # is the only case where a number can be formatted as E164 without a 

961 # leading '+' symbol (but the original number wasn't parseable 

962 # anyway). 

963 # TODO: Consider removing the 'if' above so that unparseable strings 

964 # without raw input format to the empty string instead of "+00". 

965 if len(numobj.raw_input) > 0: 

966 return numobj.raw_input 

967 country_calling_code = numobj.country_code 

968 nsn = national_significant_number(numobj) 

969 if num_format == PhoneNumberFormat.E164: 

970 # Early exit for E164 case (even if the country calling code is 

971 # invalid) since no formatting of the national number needs to be 

972 # applied. Extensions are not formatted. 

973 return _prefix_number_with_country_calling_code(country_calling_code, num_format, nsn) 

974 if not _has_valid_country_calling_code(country_calling_code): 

975 return nsn 

976 # Note region_code_for_country_code() is used because formatting 

977 # information for regions which share a country calling code is contained 

978 # by only one region for performance reasons. For example, for NANPA 

979 # regions it will be contained in the metadata for US. 

980 region_code = region_code_for_country_code(country_calling_code) 

981 # Metadata cannot be None because the country calling code is valid (which 

982 # means that the region code cannot be ZZ and must be one of our supported 

983 # region codes). 

984 metadata = PhoneMetadata.metadata_for_region_or_calling_code(country_calling_code, region_code.upper()) 

985 formatted_number = _format_nsn(nsn, metadata, num_format) 

986 formatted_number = _maybe_append_formatted_extension(numobj, 

987 metadata, 

988 num_format, 

989 formatted_number) 

990 return _prefix_number_with_country_calling_code(country_calling_code, 

991 num_format, 

992 formatted_number)

995def format_by_pattern(numobj, number_format, user_defined_formats): 

996 """Formats a phone number using client-defined formatting rules. 

997 

998 Note that if the phone number has a country calling code of zero or an 

999 otherwise invalid country calling code, we cannot work out things like 

1000 whether there should be a national prefix applied, or how to format 

1001 extensions, so we return the national significant number with no 

1002 formatting applied. 

1003 

1004 Arguments: 

1005 numobj -- The phone number to be formatted 

1006 number_format -- The format the phone number should be formatted into, 

1007 as a PhoneNumberFormat value. 

1008 user_defined_formats -- formatting rules specified by clients, as a list 

1009 of NumberFormat objects. 

1010 

1011 Returns the formatted phone number. 

1012 """ 

1013 country_code = numobj.country_code 

1014 nsn = national_significant_number(numobj) 

1015 if not _has_valid_country_calling_code(country_code): 

1016 return nsn 

1017 # Note region_code_for_country_code() is used because formatting 

1018 # information for regions which share a country calling code is contained 

1019 # by only one region for performance reasons. For example, for NANPA 

1020 # regions it will be contained in the metadata for US. 

1021 region_code = region_code_for_country_code(country_code) 

1022 # Metadata cannot be None because the country calling code is valid. 

1023 metadata = PhoneMetadata.metadata_for_region_or_calling_code(country_code, region_code) 

1024 

1025 formatted_number = U_EMPTY_STRING 

1026 formatting_pattern = _choose_formatting_pattern_for_number(user_defined_formats, nsn) 

1027 if formatting_pattern is None: 

1028 # If no pattern above is matched, we format the number as a whole. 

1029 formatted_number = nsn 

1030 else: 

1031 num_format_copy = _copy_number_format(formatting_pattern) 

1032 # Before we do a replacement of the national prefix pattern $NP with 

1033 # the national prefix, we need to copy the rule so that subsequent 

1034 # replacements for different numbers have the appropriate national 

1035 # prefix. 

1036 np_formatting_rule = formatting_pattern.national_prefix_formatting_rule 

1037 if np_formatting_rule: 

1038 national_prefix = metadata.national_prefix 

1039 if national_prefix: 

1040 # Replace $NP with national prefix and $FG with the first 

1041 # group (\1) matcher. 

1042 np_formatting_rule = np_formatting_rule.replace(_NP_STRING, national_prefix) 

1043 np_formatting_rule = np_formatting_rule.replace(_FG_STRING, unicod("\\1")) 

1044 num_format_copy.national_prefix_formatting_rule = np_formatting_rule 

1045 else: 

1046 # We don't want to have a rule for how to format the national 

1047 # prefix if there isn't one.

1134def format_number_for_mobile_dialing(numobj, region_calling_from, with_formatting): 

1135 """Returns a number formatted in such a way that it can be dialed from a 

1136 mobile phone in a specific region. 

1137 

1138 If the number cannot be reached from the region (e.g. some countries block 

1139 toll-free numbers from being called outside of the country), the method 

1140 returns an empty string. 

1141 

1142 Arguments: 

1143 numobj -- The phone number to be formatted 

1144 region_calling_from -- The region where the call is being placed. 

1145 

1146 with_formatting -- whether the number should be returned with formatting 

1147 symbols, such as spaces and dashes. 

1148 

1149 Returns the formatted phone number. 

1150 """ 

1151 country_calling_code = numobj.country_code 

1152 if not _has_valid_country_calling_code(country_calling_code): 

1153 if numobj.raw_input is None: 

1154 return U_EMPTY_STRING 

1155 else: 

1156 return numobj.raw_input 

1157 formatted_number = U_EMPTY_STRING 

1158 # Clear the extension, as that part cannot normally be dialed together with the main number. 

1159 numobj_no_ext = PhoneNumber() 

1160 numobj_no_ext.merge_from(numobj) 

1161 numobj_no_ext.extension = None 

1162 region_code = region_code_for_country_code(country_calling_code) 

1163 numobj_type = number_type(numobj_no_ext) 

1164 is_valid_number = (numobj_type != PhoneNumberType.UNKNOWN) 

1165 if region_calling_from == region_code: 

1166 is_fixed_line_or_mobile = ((numobj_type == PhoneNumberType.FIXED_LINE) or 

1167 (numobj_type == PhoneNumberType.MOBILE) or 

1168 (numobj_type == PhoneNumberType.FIXED_LINE_OR_MOBILE)) 

1169 # Carrier codes may be needed in some countries. We handle this here. 

1170 if region_code == "CO" and numobj_type == PhoneNumberType.FIXED_LINE: 

1171 formatted_number = format_national_number_with_carrier_code(numobj_no_ext, 

1172 _COLOMBIA_MOBILE_TO_FIXED_LINE_PREFIX) 

1173 elif region_code == "BR" and is_fixed_line_or_mobile: 

1174 # Historically, we set this to an empty string when parsing with 

1175 # raw input if none was found in the input string. However, this 

1176 # doesn't result in a number we can dial. For this reason, we 

1177 # treat the empty string the same as if it isn't set at all. 

1178 if (numobj_no_ext.preferred_domestic_carrier_code is not None and 

1179 len(numobj_no_ext.preferred_domestic_carrier_code) > 0): 

1180 formatted_number = format_national_number_with_preferred_carrier_code(numobj_no_ext, "") 

1181 else: 

1182 # Brazilian fixed line and mobile numbers need to be dialed with a 

1183 # carrier code when called within Brazil. Without that, most of 

1184 # the carriers won't connect the call. Because of that, we return 

1185 # an empty string here. 

1186 formatted_number = U_EMPTY_STRING 

1187 elif is_valid_number and region_code == "HU": 

1188 # The national format for HU numbers doesn't contain the national 

1189 # prefix, because that is how numbers are normally written 

1190 # down. However, the national prefix is obligatory when dialing 

1191 # from a mobile phone, except for short numbers. As a result, we 

1192 # add it back here if it is a valid regular length phone number. 

1193 formatted_number = (ndd_prefix_for_region(region_code, True) + # strip non-digits 

1194 U_SPACE + format_number(numobj_no_ext, PhoneNumberFormat.NATIONAL)) 

1195 elif country_calling_code == _NANPA_COUNTRY_CODE: 

1196 # For NANPA countries, we output international format for numbers 

1197 # that can be dialed internationally, since that always works, 

1198 # except for numbers which might potentially be short numbers, 

1199 # which are always dialled in national format. 

1200 metadata = PhoneMetadata.metadata_for_region(region_calling_from) 

1201 if (can_be_internationally_dialled(numobj_no_ext) and 

1202 _test_number_length(national_significant_number(numobj_no_ext), 

1203 metadata) != ValidationResult.TOO_SHORT): 

1204 formatted_number = format_number(numobj_no_ext, PhoneNumberFormat.INTERNATIONAL) 

1205 else: 

1206 formatted_number = format_number(numobj_no_ext, PhoneNumberFormat.NATIONAL) 

1207 else: 

1208 # For non-geographical countries, and Mexican, Chilean, and Uzbek 

1209 # fixed line and mobile numbers, we output international format for 

1210 # numbers that can be dialed internationally as that always works. 

1211 if ((region_code == REGION_CODE_FOR_NON_GEO_ENTITY or 

1212 ((region_code == unicod("MX") or region_code == unicod("CL") or 

1213 region_code == unicod("UZ")) and 

1214 is_fixed_line_or_mobile)) and 

1215 can_be_internationally_dialled(numobj_no_ext)): 

1216 # MX fixed line and mobile numbers should always be formatted 

1217 # in international format, even when dialed within MX. For 

1218 # national format to work, a carrier code needs to be used, 

1219 # and the correct carrier code depends on if the caller and 

1220 # callee are from the same local area. It is trickier to get 

1221 # that to work correctly than using international format, 

1222 # which is tested to work fine on all carriers. 

1223 # CL fixed line numbers need the national prefix when dialing 

1224 # in the national format, but don't have it when used for 

1225 # display. The reverse is true for mobile numbers. As a 

1226 # result, we output them in the international format to make 

1227 # it work. 

1228 # UZ mobile and fixed-line numbers have to be formatted in 

1229 # international format or prefixed with special codes like 03, 

1230 # 04 (for fixed-line) and 05 (for mobile) for dialling 

1231 # successfully from mobile devices. As we do not have complete 

1232 # information on special codes and to be consistent with 

1233 # formatting across all phone types we return the number in 

1234 # international format here. 

1235 formatted_number = format_number(numobj_no_ext, PhoneNumberFormat.INTERNATIONAL) 

1236 else: 

1237 formatted_number = format_number(numobj_no_ext, PhoneNumberFormat.NATIONAL) 

1238 elif is_valid_number and can_be_internationally_dialled(numobj_no_ext): 

1239 # We assume that short numbers are not diallable from outside their 

1240 # region, so if a number is not a valid regular length phone number, 

1241 # we treat it as if it cannot be internationally dialled. 

1242 if with_formatting: 

1243 return format_number(numobj_no_ext, PhoneNumberFormat.INTERNATIONAL) 

1244 else: 

1245 return format_number(numobj_no_ext, PhoneNumberFormat.E164) 

1246 

1247 if with_formatting: 

1248 return formatted_number 

1249 else: 

1250 return normalize_diallable_chars_only(formatted_number)

1253def format_out_of_country_calling_number(numobj, region_calling_from): 

1254 """Formats a phone number for out-of-country dialing purposes. 

1255 

1256 If no region_calling_from is supplied, we format the number in its 

1257 INTERNATIONAL format. If the country calling code is the same as that of 

1258 the region where the number is from, then NATIONAL formatting will be 

1259 applied. 

1260 

1261 If the number itself has a country calling code of zero or an otherwise 

1262 invalid country calling code, then we return the number with no formatting 

1263 applied. 

1264 

1265 Note this function takes care of the case for calling inside of NANPA and 

1266 between Russia and Kazakhstan (who share the same country calling 

1267 code). In those cases, no international prefix is used. For regions which 

1268 have multiple international prefixes, the number in its INTERNATIONAL 

1269 format will be returned instead. 

1270 

1271 Arguments: 

1272 numobj -- The phone number to be formatted 

1273 region_calling_from -- The region where the call is being placed 

1274 

1275 Returns the formatted phone number 

1276 """ 

1277 if not _is_valid_region_code(region_calling_from): 

1278 return format_number(numobj, PhoneNumberFormat.INTERNATIONAL) 

1279 country_code = numobj.country_code 

1280 nsn = national_significant_number(numobj) 

1281 if not _has_valid_country_calling_code(country_code): 

1282 return nsn 

1283 if country_code == _NANPA_COUNTRY_CODE: 

1284 if is_nanpa_country(region_calling_from): 

1285 # For NANPA regions, return the national format for these regions 

1286 # but prefix it with the country calling code. 

1287 return (unicod(country_code) + U_SPACE + 

1288 format_number(numobj, PhoneNumberFormat.NATIONAL)) 

1289 elif country_code == country_code_for_valid_region(region_calling_from): 

1290 # If regions share a country calling code, the country calling code 

1291 # need not be dialled. This also applies when dialling within a 

1292 # region, so this if clause covers both these cases. Technically this 

1293 # is the case for dialling from La Reunion to other overseas 

1294 # departments of France (French Guiana, Martinique, Guadeloupe), but 

1295 # not vice versa - so we don't cover this edge case for now and for 

1296 # those cases return the version including country calling code. 

1297 # Details here: 

1298 # http://www.petitfute.com/voyage/225-info-pratiques-reunion 

1299 return format_number(numobj, PhoneNumberFormat.NATIONAL) 

1300 

1301 # Metadata cannot be None because we checked '_is_valid_region_code()' above. 

1302 metadata_for_region_calling_from = PhoneMetadata.metadata_for_region_or_calling_code(country_code, region_calling_from.upper()) 

1303 international_prefix = metadata_for_region_calling_from.international_prefix 

1304 

1305 # For regions that have multiple international prefixes, the international 

1306 # format of the number is returned, unless there is a preferred 

1307 # international prefix. 

1308 i18n_prefix_for_formatting = U_EMPTY_STRING 

1309 i18n_match = fullmatch(_SINGLE_INTERNATIONAL_PREFIX, international_prefix) 

1310 if i18n_match: 

1311 i18n_prefix_for_formatting = international_prefix 

1312 elif metadata_for_region_calling_from.preferred_international_prefix is not None: 

1313 i18n_prefix_for_formatting = metadata_for_region_calling_from.preferred_international_prefix 

1314 

1315 region_code = region_code_for_country_code(country_code) 

1316 # Metadata cannot be None because the country calling code is valid. 

1317 metadata_for_region = PhoneMetadata.metadata_for_region_or_calling_code(country_code, region_code.upper()) 

1318 formatted_national_number = _format_nsn(nsn, 

1319 metadata_for_region, 

1320 PhoneNumberFormat.INTERNATIONAL) 

1321 formatted_number = _maybe_append_formatted_extension(numobj, 

1322 metadata_for_region, 

1323 PhoneNumberFormat.INTERNATIONAL, 

1324 formatted_national_number) 

1325 if len(i18n_prefix_for_formatting) > 0: 

1326 formatted_number = (i18n_prefix_for_formatting + U_SPACE + 

1327 unicod(country_code) + U_SPACE + formatted_number) 

1328 else: 

1329 formatted_number = _prefix_number_with_country_calling_code(country_code, 

1330 PhoneNumberFormat.INTERNATIONAL, 

1331 formatted_number) 

1332 return formatted_number

1335def format_in_original_format(numobj, region_calling_from): 

1336 """Format a number using the original format that the number was parsed from. 

1337 

1338 The original format is embedded in the country_code_source field of the 

1339 PhoneNumber object passed in. If such information is missing, the number 

1340 will be formatted into the NATIONAL format by default. 

1341 

1342 When we don't have a formatting pattern for the number, the method 

1343 returns the raw input when it is available. 

1344 

1345 Note this method guarantees no digit will be inserted, removed or modified 

1346 as a result of formatting. 

1347 

1348 Arguments: 

1349 number -- The phone number that needs to be formatted in its original 

1350 number format 

1351 region_calling_from -- The region whose IDD needs to be prefixed if the 

1352 original number has one. 

1353 

1354 Returns the formatted phone number in its original number format. 

1355 """ 

1356 if (numobj.raw_input is not None and not _has_formatting_pattern_for_number(numobj)): 

1357 # We check if we have the formatting pattern because without that, we 

1358 # might format the number as a group without national prefix. 

1359 return numobj.raw_input 

1360 if numobj.country_code_source is CountryCodeSource.UNSPECIFIED: 

1361 return format_number(numobj, PhoneNumberFormat.NATIONAL) 

1362 

1363 formatted_number = _format_original_allow_mods(numobj, region_calling_from) 

1364 num_raw_input = numobj.raw_input 

1365 # If no digit is inserted/removed/modified as a result of our formatting, 

1366 # we return the formatted phone number; otherwise we return the raw input 

1367 # the user entered. 

1368 if (formatted_number is not None and num_raw_input): 

1369 normalized_formatted_number = normalize_diallable_chars_only(formatted_number) 

1370 normalized_raw_input = normalize_diallable_chars_only(num_raw_input) 

1371 if normalized_formatted_number != normalized_raw_input: 

1372 formatted_number = num_raw_input 

1373 return formatted_number

1376def _format_original_allow_mods(numobj, region_calling_from): 

1377 if (numobj.country_code_source == CountryCodeSource.FROM_NUMBER_WITH_PLUS_SIGN): 

1378 return format_number(numobj, PhoneNumberFormat.INTERNATIONAL) 

1379 elif numobj.country_code_source == CountryCodeSource.FROM_NUMBER_WITH_IDD: 

1380 return format_out_of_country_calling_number(numobj, region_calling_from) 

1381 elif (numobj.country_code_source == CountryCodeSource.FROM_NUMBER_WITHOUT_PLUS_SIGN): 

1382 return format_number(numobj, PhoneNumberFormat.INTERNATIONAL)[1:] 

1383 else: 

1384 region_code = region_code_for_country_code(numobj.country_code) 

1385 # We strip non-digits from the NDD here, and from the raw input later, so that we can 

1386 # compare them easily. 

1387 national_prefix = ndd_prefix_for_region(region_code, True) # strip non-digits 

1388 national_format = format_number(numobj, PhoneNumberFormat.NATIONAL) 

1389 if (national_prefix is None or len(national_prefix) == 0): 

1390 # If the region doesn't have a national prefix at all, we can 

1391 # safely return the national format without worrying about a 

1392 # national prefix being added. 

1393 return national_format 

1394 # Otherwise, we check if the original number was entered with a national prefix. 

1395 if (_raw_input_contains_national_prefix(numobj.raw_input, national_prefix, region_code)): 

1396 # If so, we can safely return the national format. 

1397 return national_format 

1398 # Metadata cannot be None here because ndd_prefix_for_region() (above) returns None if 

1399 # there is no metadata for the region. 

1400 metadata = PhoneMetadata.metadata_for_region(region_code) 

1401 national_number = national_significant_number(numobj) 

1402 format_rule = _choose_formatting_pattern_for_number(metadata.number_format, national_number) 

1403 # The format rule could still be null here if the national number was 

1404 # 0 and there was no raw input (this should not be possible for 

1405 # numbers generated by the phonenumber library as they would also not 

1406 # have a country calling code and we would have exited earlier). 

1407 if format_rule is None: 

1408 return national_format 

1409 # When the format we apply to this number doesn't contain national 

1410 # prefix, we can just return the national format. 

1411 # TODO: Refactor the code below with the code in isNationalPrefixPresentIfRequired. 

1412 candidate_national_prefix_rule = format_rule.national_prefix_formatting_rule 

1413 # We assume that the first-group symbol will never be _before_ the national prefix. 

1414 if candidate_national_prefix_rule is None: 

1415 return national_format 

1416 index_of_first_group = candidate_national_prefix_rule.find("\\1") 

1417 if (index_of_first_group <= 0): 

1418 return national_format 

1419 candidate_national_prefix_rule = candidate_national_prefix_rule[:index_of_first_group] 

1420 candidate_national_prefix_rule = normalize_digits_only(candidate_national_prefix_rule) 

1421 if len(candidate_national_prefix_rule) == 0: 

1422 # National prefix not used when formatting this number. 

1423 return national_format 

1424 # Otherwise, we need to remove the national prefix from our output. 

1425 new_format_rule = _copy_number_format(format_rule) 

1426 new_format_rule.national_prefix_formatting_rule = None 

1427 return format_by_pattern(numobj, PhoneNumberFormat.NATIONAL, [new_format_rule])

1430def _raw_input_contains_national_prefix(raw_input, national_prefix, region_code): 

1431 """Check if raw_input, which is assumed to be in the national format, has a 

1432 national prefix. The national prefix is assumed to be in digits-only 

1433 form.""" 

1434 nnn = normalize_digits_only(raw_input) 

1435 if nnn.startswith(national_prefix): 

1436 try: 

1437 # Some Japanese numbers (e.g. 00777123) might be mistaken to 

1438 # contain the national prefix when written without it 

1439 # (e.g. 0777123) if we just do prefix matching. To tackle that, we 

1440 # check the validity of the number if the assumed national prefix 

1441 # is removed (777123 won't be valid in Japan). 

1442 return is_valid_number(parse(nnn[len(national_prefix):], region_code)) 

1443 except NumberParseException: 

1444 return False 

1445 return False

1459def format_out_of_country_keeping_alpha_chars(numobj, region_calling_from): 

1460 """Formats a phone number for out-of-country dialing purposes. 

1461 

1462 Note that in this version, if the number was entered originally using 

1463 alpha characters and this version of the number is stored in raw_input, 

1464 this representation of the number will be used rather than the digit 

1465 representation. Grouping information, as specified by characters such as 

1466 "-" and " ", will be retained. 

1467 

1468 Caveats: 

1469 

1470 - This will not produce good results if the country calling code is both 

1471 present in the raw input _and_ is the start of the national 

1472 number. This is not a problem in the regions which typically use alpha 

1473 numbers. 

1474 

1475 - This will also not produce good results if the raw input has any 

1476 grouping information within the first three digits of the national 

1477 number, and if the function needs to strip preceding digits/words in 

1478 the raw input before these digits. Normally people group the first 

1479 three digits together so this is not a huge problem - and will be fixed 

1480 if it proves to be so. 

1481 

1482 Arguments: 

1483 numobj -- The phone number that needs to be formatted. 

1484 region_calling_from -- The region where the call is being placed. 

1485 

1486 Returns the formatted phone number 

1487 """ 

1488 num_raw_input = numobj.raw_input 

1489 # If there is no raw input, then we can't keep alpha characters because there aren't any. 

1490 # In this case, we return format_out_of_country_calling_number. 

1491 if num_raw_input is None or len(num_raw_input) == 0: 

1492 return format_out_of_country_calling_number(numobj, region_calling_from) 

1493 country_code = numobj.country_code 

1494 if not _has_valid_country_calling_code(country_code): 

1495 return num_raw_input 

1496 # Strip any prefix such as country calling code, IDD, that was present. We 

1497 # do this by comparing the number in raw_input with the parsed number. To 

1498 # do this, first we normalize punctuation. We retain number grouping 

1499 # symbols such as " " only. 

1500 num_raw_input = _normalize_helper(num_raw_input, 

1501 _ALL_PLUS_NUMBER_GROUPING_SYMBOLS, 

1502 True) 

1503 # Now we trim everything before the first three digits in the parsed 

1504 # number. We choose three because all valid alpha numbers have 3 digits at 

1505 # the start - if it does not, then we don't trim anything at 

1506 # all. Similarly, if the national number was less than three digits, we 

1507 # don't trim anything at all. 

1508 national_number = national_significant_number(numobj) 

1509 if len(national_number) > 3: 

1510 first_national_number_digit = num_raw_input.find(national_number[:3]) 

1511 if first_national_number_digit != -1: 

1512 num_raw_input = num_raw_input[first_national_number_digit:] 

1513 

1514 metadata_for_region_calling_from = PhoneMetadata.metadata_for_region(region_calling_from.upper(), None) 

1515 if country_code == _NANPA_COUNTRY_CODE: 

1516 if is_nanpa_country(region_calling_from): 

1517 return unicod(country_code) + U_SPACE + num_raw_input 

1518 elif (metadata_for_region_calling_from is not None and 

1519 country_code == country_code_for_region(region_calling_from)): 

1520 formatting_pattern = _choose_formatting_pattern_for_number(metadata_for_region_calling_from.number_format, 

1521 national_number) 

1522 if formatting_pattern is None: 

1523 # If no pattern above is matched, we format the original input 

1524 return num_raw_input 

1525 new_format = _copy_number_format(formatting_pattern) 

1526 # The first group is the first group of digits that the user 

1527 # wrote together. 

1528 new_format.pattern = u("(\\d+)(.*)") 

1529 # Here we just concatenate them back together after the national 

1530 # prefix has been fixed. 

1531 new_format.format = u(r"\1\2") 

1532 # Now we format using this pattern instead of the default pattern, 

1533 # but with the national prefix prefixed if necessary. 

1534 # This will not work in the cases where the pattern (and not the 

1535 # leading digits) decide whether a national prefix needs to be used, 

1536 # since we have overridden the pattern to match anything, but that is 

1537 # not the case in the metadata to date. 

1538 return _format_nsn_using_pattern(num_raw_input, 

1539 new_format, 

1540 PhoneNumberFormat.NATIONAL) 

1541 i18n_prefix_for_formatting = U_EMPTY_STRING 

1542 # If an unsupported region-calling-from is entered, or a country with 

1543 # multiple international prefixes, the international format of the number 

1544 # is returned, unless there is a preferred international prefix. 

1545 if metadata_for_region_calling_from is not None: 

1546 international_prefix = metadata_for_region_calling_from.international_prefix 

1547 i18n_match = fullmatch(_SINGLE_INTERNATIONAL_PREFIX, international_prefix) 

1548 if i18n_match: 

1549 i18n_prefix_for_formatting = international_prefix 

1550 else: 

1551 i18n_prefix_for_formatting = metadata_for_region_calling_from.preferred_international_prefix 

1552 

1553 region_code = region_code_for_country_code(country_code) 

1554 # Metadata cannot be None because the country calling code is valid. 

1555 metadata_for_region = PhoneMetadata.metadata_for_region_or_calling_code(country_code, region_code) 

1556 formatted_number = _maybe_append_formatted_extension(numobj, 

1557 metadata_for_region, 

1558 PhoneNumberFormat.INTERNATIONAL, 

1559 num_raw_input) 

1560 if i18n_prefix_for_formatting: 

1561 formatted_number = (i18n_prefix_for_formatting + U_SPACE + 

1562 unicod(country_code) + U_SPACE + formatted_number) 

1563 else: 

1564 # Invalid region entered as country-calling-from (so no metadata was 

1565 # found for it) or the region chosen has multiple international 

1566 # dialling prefixes. 

1567 formatted_number = _prefix_number_with_country_calling_code(country_code, 

1568 PhoneNumberFormat.INTERNATIONAL, 

1569 formatted_number) 

1570 return formatted_number

1611def _format_nsn(number, metadata, num_format, carrier_code=None): 

1612 """Format a national number.""" 

1613 # Note in some regions, the national number can be written in two 

1614 # completely different ways depending on whether it forms part of the 

1615 # NATIONAL format or INTERNATIONAL format. The num_format parameter here 

1616 # is used to specify which format to use for those cases. If a carrier_code 

1617 # is specified, this will be inserted into the formatted string to replace 

1618 # $CC. 

1619 intl_number_formats = metadata.intl_number_format 

1620 

1621 # When the intl_number_formats exists, we use that to format national 

1622 # number for the INTERNATIONAL format instead of using the 

1623 # number_desc.number_formats. 

1624 if (len(intl_number_formats) == 0 or 

1625 num_format == PhoneNumberFormat.NATIONAL): 

1626 available_formats = metadata.number_format 

1627 else: 

1628 available_formats = metadata.intl_number_format 

1629 formatting_pattern = _choose_formatting_pattern_for_number(available_formats, number) 

1630 if formatting_pattern is None: 

1631 return number 

1632 else: 

1633 return _format_nsn_using_pattern(number, formatting_pattern, num_format, carrier_code)

1636def _choose_formatting_pattern_for_number(available_formats, national_number): 

1637 for num_format in available_formats: 

1638 size = len(num_format.leading_digits_pattern) 

1639 # We always use the last leading_digits_pattern, as it is the most detailed. 

1640 if size > 0: 

1641 ld_pattern = re.compile(num_format.leading_digits_pattern[-1]) 

1642 ld_match = ld_pattern.match(national_number) 

1643 if size == 0 or ld_match: 

1644 format_pattern = re.compile(num_format.pattern) 

1645 if fullmatch(format_pattern, national_number): 

1646 return num_format 

1647 return None

1650def _format_nsn_using_pattern(national_number, formatting_pattern, number_format, 

1651 carrier_code=None): 

1652 # Note that carrier_code is optional - if None or an empty string, no 

1653 # carrier code replacement will take place. 

1654 number_format_rule = formatting_pattern.format 

1655 m_re = re.compile(formatting_pattern.pattern) 

1656 formatted_national_number = U_EMPTY_STRING 

1657 

1658 if (number_format == PhoneNumberFormat.NATIONAL and carrier_code and 

1659 formatting_pattern.domestic_carrier_code_formatting_rule): 

1660 # Replace the $CC in the formatting rule with the desired 

1661 # carrier code. 

1662 cc_format_rule = formatting_pattern.domestic_carrier_code_formatting_rule 

1663 cc_format_rule = cc_format_rule.replace(_CC_STRING, carrier_code) 

1664 

1665 # Now replace the $FG in the formatting rule with the 

1666 # first group and the carrier code combined in the 

1667 # appropriate way. 

1668 number_format_rule = re.sub(_FIRST_GROUP_PATTERN, 

1669 cc_format_rule, 

1670 number_format_rule, 

1671 count=1) 

1672 formatted_national_number = re.sub(m_re, number_format_rule, national_number) 

1673 else: 

1674 # Use the national prefix formatting rule instead. 

1675 national_prefix_formatting_rule = formatting_pattern.national_prefix_formatting_rule 

1676 if (number_format == PhoneNumberFormat.NATIONAL and 

1677 national_prefix_formatting_rule): 

1678 first_group_rule = re.sub(_FIRST_GROUP_PATTERN, 

1679 national_prefix_formatting_rule, 

1680 number_format_rule, 

1681 count=1) 

1682 formatted_national_number = re.sub(m_re, first_group_rule, national_number) 

1683 else: 

1684 formatted_national_number = re.sub(m_re, number_format_rule, national_number) 

1685 

1686 if number_format == PhoneNumberFormat.RFC3966: 

1687 # Strip any leading punctuation. 

1688 m = _SEPARATOR_PATTERN.match(formatted_national_number) 

1689 if m: 

1690 formatted_national_number = re.sub(_SEPARATOR_PATTERN, U_EMPTY_STRING, formatted_national_number, count=1) 

1691 # Replace the rest with a dash between each number group 

1692 formatted_national_number = re.sub(_SEPARATOR_PATTERN, U_DASH, formatted_national_number) 

1693 

1694 return formatted_national_number

1711def invalid_example_number(region_code): 

1712 """Gets an invalid number for the specified region. 

1713 

1714 This is useful for unit-testing purposes, where you want to test what 

1715 will happen with an invalid number. Note that the number that is 

1716 returned will always be able to be parsed and will have the correct 

1717 country code. It may also be a valid *short* number/code for this 

1718 region. Validity checking such numbers is handled with shortnumberinfo. 

1719 

1720 Arguments: 

1721 region_code -- The region for which an example number is needed. 

1722 

1723 

1724 Returns an invalid number for the specified region. Returns None when an 

1725 unsupported region or the region 001 (Earth) is passed in. 

1726 """ 

1727 if not _is_valid_region_code(region_code): 

1728 return None 

1729 # We start off with a valid fixed-line number since every country 

1730 # supports this. Alternatively we could start with a different number 

1731 # type, since fixed-line numbers typically have a wide breadth of valid 

1732 # number lengths and we may have to make it very short before we get an 

1733 # invalid number. 

1734 metadata = PhoneMetadata.metadata_for_region(region_code.upper()) 

1735 desc = _number_desc_by_type(metadata, PhoneNumberType.FIXED_LINE) 

1736 if desc is None or desc.example_number is None: 

1737 # This shouldn't happen; we have a test for this. 

1738 return None # pragma no cover 

1739 example_number = desc.example_number 

1740 # Try and make the number invalid. We do this by changing the length. We 

1741 # try reducing the length of the number, since currently no region has a 

1742 # number that is the same length as MIN_LENGTH_FOR_NSN. This is probably 

1743 # quicker than making the number longer, which is another 

1744 # alternative. We could also use the possible number pattern to extract 

1745 # the possible lengths of the number to make this faster, but this 

1746 # method is only for unit-testing so simplicity is preferred to 

1747 # performance. We don't want to return a number that can't be parsed, 

1748 # so we check the number is long enough. We try all possible lengths 

1749 # because phone number plans often have overlapping prefixes so the 

1750 # number 123456 might be valid as a fixed-line number, and 12345 as a 

1751 # mobile number. It would be faster to loop in a different order, but we 

1752 # prefer numbers that look closer to real numbers (and it gives us a 

1753 # variety of different lengths for the resulting phone numbers - 

1754 # otherwise they would all be MIN_LENGTH_FOR_NSN digits long.) 

1755 phone_number_length = len(example_number) - 1 

1756 while phone_number_length >= _MIN_LENGTH_FOR_NSN: 

1757 number_to_try = example_number[:phone_number_length] 

1758 try: 

1759 possibly_valid_number = parse(number_to_try, region_code) 

1760 if not is_valid_number(possibly_valid_number): 

1761 return possibly_valid_number 

1762 except NumberParseException: # pragma no cover 

1763 # Shouldn't happen: we have already checked the length, we know 

1764 # example numbers have only valid digits, and we know the region 

1765 # code is fine. 

1766 pass 

1767 phone_number_length -= 1 

1768 

1769 # We have a test to check that this doesn't happen for any of our 

1770 # supported regions. 

1771 return None # pragma no cover

1901 """Gets the type of a valid phone number. 

1902

1904 numobj -- The PhoneNumber object that we want to know the type of.

1911 if metadata is None:

1917def _number_type_helper(national_number, metadata): 

1918 """Return the type of the given number against the metadata""" 

1919 if not _is_number_matching_desc(national_number, metadata.general_desc): 

1920 return PhoneNumberType.UNKNOWN 

1921 if _is_number_matching_desc(national_number, metadata.premium_rate): 

1922 return PhoneNumberType.PREMIUM_RATE 

1923 if _is_number_matching_desc(national_number, metadata.toll_free): 

1924 return PhoneNumberType.TOLL_FREE 

1925 if _is_number_matching_desc(national_number, metadata.shared_cost): 

1926 return PhoneNumberType.SHARED_COST 

1927 if _is_number_matching_desc(national_number, metadata.voip): 

1928 return PhoneNumberType.VOIP 

1929 if _is_number_matching_desc(national_number, metadata.personal_number): 

1930 return PhoneNumberType.PERSONAL_NUMBER 

1931 if _is_number_matching_desc(national_number, metadata.pager): 

1932 return PhoneNumberType.PAGER 

1933 if _is_number_matching_desc(national_number, metadata.uan): 

1934 return PhoneNumberType.UAN 

1935 if _is_number_matching_desc(national_number, metadata.voicemail): 

1936 return PhoneNumberType.VOICEMAIL 

1937 

1938 if _is_number_matching_desc(national_number, metadata.fixed_line): 

1939 if metadata.same_mobile_and_fixed_line_pattern: 

1940 return PhoneNumberType.FIXED_LINE_OR_MOBILE 

1941 elif _is_number_matching_desc(national_number, metadata.mobile): 

1942 return PhoneNumberType.FIXED_LINE_OR_MOBILE 

1943 return PhoneNumberType.FIXED_LINE 

1944 

1945 # Otherwise, test to see if the number is mobile. Only do this if certain 

1946 # that the patterns for mobile and fixed line aren't the same. 

1947 if (not metadata.same_mobile_and_fixed_line_pattern and 

1948 _is_number_matching_desc(national_number, metadata.mobile)): 

1949 return PhoneNumberType.MOBILE 

1950 return PhoneNumberType.UNKNOWN

1953def _is_number_matching_desc(national_number, number_desc): 

1954 """Determine if the number matches the given PhoneNumberDesc""" 

1955 # Check if any possible number lengths are present; if so, we use them to avoid checking the 

1956 # validation pattern if they don't match. If they are absent, this means they match the general 

1957 # description, which we have already checked before checking a specific number type. 

1958 if number_desc is None: 

1959 return False 

1960 actual_length = len(national_number) 

1961 possible_lengths = number_desc.possible_length 

1962 if len(possible_lengths) > 0 and actual_length not in possible_lengths: 

1963 return False 

1964 return _match_national_number(national_number, number_desc, False)

1967def is_valid_number(numobj): 

1968 """Tests whether a phone number matches a valid pattern. 

1969 

1970 Note this doesn't verify the number is actually in use, which is 

1971 impossible to tell by just looking at a number itself. It only verifies 

1972 whether the parsed, canonicalised number is valid: not whether a 

1973 particular series of digits entered by the user is diallable from the 

1974 region provided when parsing. For example, the number +41 (0) 78 927 2696 

1975 can be parsed into a number with country code "41" and national 

1976 significant number "789272696". This is valid, while the original string 

1977 is not diallable. 

1978 

1979 Arguments: 

1980 numobj -- The phone number object that we want to validate 

1981 

1982 Returns a boolean that indicates whether the number is of a valid pattern. 

1983 """ 

1984 region_code = region_code_for_number(numobj) 

1985 return is_valid_number_for_region(numobj, region_code)

1988def is_valid_number_for_region(numobj, region_code): 

1989 """Tests whether a phone number is valid for a certain region. 

1990 

1991 Note this doesn't verify the number is actually in use, which is 

1992 impossible to tell by just looking at a number itself. If the country 

1993 calling code is not the same as the country calling code for the region, 

1994 this immediately exits with false. After this, the specific number pattern 

1995 rules for the region are examined. This is useful for determining for 

1996 example whether a particular number is valid for Canada, rather than just 

1997 a valid NANPA number. 

1998 

1999 Warning: In most cases, you want to use is_valid_number instead. For 

2000 example, this method will mark numbers from British Crown dependencies 

2001 such as the Isle of Man as invalid for the region "GB" (United Kingdom), 

2002 since it has its own region code, "IM", which may be undesirable. 

2003 

2004 Arguments: 

2005 numobj -- The phone number object that we want to validate. 

2006 region_code -- The region that we want to validate the phone number for. 

2007 

2008 Returns a boolean that indicates whether the number is of a valid pattern. 

2009 """ 

2010 country_code = numobj.country_code 

2011 if region_code is None: 

2012 return False 

2013 metadata = PhoneMetadata.metadata_for_region_or_calling_code(country_code, region_code.upper()) 

2014 if (metadata is None or 

2015 (region_code != REGION_CODE_FOR_NON_GEO_ENTITY and 

2016 country_code != country_code_for_valid_region(region_code))): 

2017 # Either the region code was invalid, or the country calling code for 

2018 # this number does not match that of the region code. 

2019 return False 

2020 nsn = national_significant_number(numobj) 

2021 return (_number_type_helper(nsn, metadata) != PhoneNumberType.UNKNOWN)

2037 """

2040 if regions is None: 

2041 return None 

2042

2049def _region_code_for_number_from_list(numobj, regions): 

2050 """Find the region in a list that matches a number""" 

2051 national_number = national_significant_number(numobj) 

2052 for region_code in regions: 

2053 # If leading_digits is present, use this. Otherwise, do full 

2054 # validation. 

2055 # Metadata cannot be None because the region codes come from 

2056 # the country calling code map. 

2057 metadata = PhoneMetadata.metadata_for_region(region_code.upper(), None) 

2058 if metadata is None: 

2059 continue

2616def _maybe_strip_national_prefix_carrier_code(number, metadata): 

2617 """Strips any national prefix (such as 0, 1) present in a number. 

2618 

2619 Arguments: 

2620 number -- The normalized telephone number that we wish to strip any 

2621 national dialing prefix from 

2622 metadata -- The metadata for the region that we think this number 

2623 is from. 

2624 

2625 Returns a 3-tuple of 

2626 - The carrier code extracted if it is present, otherwise an empty string. 

2627 - The number with the prefix stripped. 

2628 - Boolean indicating if a national prefix or carrier code (or both) could be extracted. 

2629 """ 

2630 carrier_code = U_EMPTY_STRING 

2631 possible_national_prefix = metadata.national_prefix_for_parsing 

2632 if (len(number) == 0 or 

2633 possible_national_prefix is None or 

2634 len(possible_national_prefix) == 0): 

2635 # Early return for numbers of zero length. 

2636 return (U_EMPTY_STRING, number, False) 

2637 

2638 # Attempt to parse the first digits as a national prefix. 

2639 prefix_pattern = re.compile(possible_national_prefix) 

2640 prefix_match = prefix_pattern.match(number) 

2641 if prefix_match: 

2642 general_desc = metadata.general_desc 

2643 # Check if the original number is viable. 

2644 is_viable_original_number = _match_national_number(number, general_desc, False) 

2645 # prefix_match.groups() == () implies nothing was captured by the 

2646 # capturing groups in possible_national_prefix; therefore, no 

2647 # transformation is necessary, and we just remove the national prefix. 

2648 num_groups = len(prefix_match.groups()) 

2649 transform_rule = metadata.national_prefix_transform_rule 

2650 if (transform_rule is None or 

2651 len(transform_rule) == 0 or 

2652 prefix_match.groups()[num_groups - 1] is None): 

2653 # If the original number was viable, and the resultant number is not, we return. 

2654 # Check that the resultant number is viable. If not, return. 

2655 national_number_match = _match_national_number(number[prefix_match.end():], general_desc, False) 

2656 if (is_viable_original_number and not national_number_match): 

2657 return (U_EMPTY_STRING, number, False) 

2658 

2659 if (num_groups > 0 and 

2660 prefix_match.groups(num_groups) is not None): 

2661 carrier_code = prefix_match.group(1) 

2662 return (carrier_code, number[prefix_match.end():], True) 

2663 else: 

2664 # Check that the resultant number is still viable. If not, 

2665 # return. Check this by copying the number and making the 

2666 # transformation on the copy first. 

2667 transformed_number = re.sub(prefix_pattern, transform_rule, number, count=1) 

2668 national_number_match = _match_national_number(transformed_number, general_desc, False) 

2669 if (is_viable_original_number and not national_number_match): 

2670 return ("", number, False) 

2671 if num_groups > 1: 

2672 carrier_code = prefix_match.group(1) 

2673 return (carrier_code, transformed_number, True) 

2674 else: 

2675 return (carrier_code, number, False)

2737def parse(number, region=None, keep_raw_input=False, 

2738 numobj=None, _check_region=True): 

2739 """Parse a string and return a corresponding PhoneNumber object. 

2740 

2741 The method is quite lenient and looks for a number in the input text 

2742 (raw input) and does not check whether the string is definitely only a 

2743 phone number. To do this, it ignores punctuation and white-space, as 

2744 well as any text before the number (e.g. a leading "Tel: ") and trims 

2745 the non-number bits. It will accept a number in any format (E164, 

2746 national, international etc), assuming it can be interpreted with the 

2747 defaultRegion supplied. It also attempts to convert any alpha characters 

2748 into digits if it thinks this is a vanity number of the type "1800 

2749 MICROSOFT". 

2750 

2751 This method will throw a NumberParseException if the number is not 

2752 considered to be a possible number. Note that validation of whether the 

2753 number is actually a valid number for a particular region is not 

2754 performed. This can be done separately with is_valid_number. 

2755 

2756 Note this method canonicalizes the phone number such that different 

2757 representations can be easily compared, no matter what form it was 

2758 originally entered in (e.g. national, international). If you want to 

2759 record context about the number being parsed, such as the raw input that 

2760 was entered, how the country code was derived etc. then ensure 

2761 keep_raw_input is set.

2764 in, even when keep_raw_input is False, it should also be handled in the 

2765 _copy_core_fields_only() function.

2775 supplied. If the number is guaranteed to start with a '+' 

2776 followed by the country calling code, then None or

2784 should always be True for external callers.

2787

2789 NumberParseException if the string is not considered to be a viable 

2790 phone number (e.g. too few or too many digits) or if no default

2808 "The string supplied did not seem to be a phone number.") 

2809 

2810 # Check the region supplied is valid, or that the extracted number starts 

2811 # with some sort of + sign so the number's region can be determined. 

2812 if _check_region and not _check_region_for_parsing(national_number, region): 

2813 raise NumberParseException(NumberParseException.INVALID_COUNTRY_CODE, 

2814 "Missing or invalid default region.") 

2815 if keep_raw_input: 

2816 numobj.raw_input = number 

2817 

2818 # Attempt to parse extension first, since it doesn't require 

2819 # region-specific data and we want to have the non-normalised number here. 

2820 extension, national_number = _maybe_strip_extension(national_number) 

2821 if len(extension) > 0: 

2822 numobj.extension = extension 

2823 if region is None: 

2824 metadata = None 

2825 else: 

2826 metadata = PhoneMetadata.metadata_for_region(region.upper(), None) 

2827 

2828 country_code = 0 

2829 try: 

2830 country_code, normalized_national_number = _maybe_extract_country_code(national_number, 

2831 metadata, 

2832 keep_raw_input, 

2833 numobj) 

2834 except NumberParseException:

2837 if (e.error_type == NumberParseException.INVALID_COUNTRY_CODE and 

2838 matchobj is not None): 

2839 # Strip the plus-char, and try again. 

2840 country_code, normalized_national_number = _maybe_extract_country_code(national_number[matchobj.end():], 

2841 metadata,

2884 len_national_number = len(normalized_national_number) 

2885 if len_national_number < _MIN_LENGTH_FOR_NSN: # pragma no cover 

2886 # Check of _is_viable_phone_number() at the top of this function makes 

2887 # this effectively unhittable. 

2888 raise NumberParseException(NumberParseException.TOO_SHORT_NSN, 

2889 "The string supplied is too short to be a phone number.") 

2890 if len_national_number > _MAX_LENGTH_FOR_NSN: 

2891 raise NumberParseException(NumberParseException.TOO_LONG, 

2892 "The string supplied is too long to be a phone number.") 

2893 _set_italian_leading_zeros_for_phone_number(normalized_national_number, numobj)

2898def _build_national_number_for_parsing(number): 

2899 """Converts number to a form that we can parse and return it if it is 

2900 written in RFC3966; otherwise extract a possible number out of it and return it.""" 

2901 index_of_phone_context = number.find(_RFC3966_PHONE_CONTEXT) 

2902 if index_of_phone_context >= 0: 

2903 phone_context_start = index_of_phone_context + len(_RFC3966_PHONE_CONTEXT) 

2904 # If the phone context contains a phone number prefix, we need to 

2905 # capture it, whereas domains will be ignored. 

2906 if (phone_context_start < (len(number) - 1) and 

2907 number[phone_context_start] == _PLUS_SIGN): 

2908 # Additional parameters might follow the phone context. If so, we 

2909 # will remove them here because the parameters after phone context 

2910 # are not important for parsing the phone number. 

2911 phone_context_end = number.find(U_SEMICOLON, phone_context_start) 

2912 if phone_context_end > 0: 

2913 national_number = number[phone_context_start:phone_context_end] 

2914 else: 

2915 national_number = number[phone_context_start:] 

2916 else: 

2917 national_number = U_EMPTY_STRING 

2918 # Now append everything between the "tel:" prefix and the 

2919 # phone-context. This should include the national number, an optional 

2920 # extension or isdn-subaddress component. Note we also handle the case 

2921 # when "tel:" is missing, as we have seen in some of the phone number 

2922 # inputs. In that case we append everything from the beginning. 

2923 index_of_rfc3996_prefix = number.find(_RFC3966_PREFIX) 

2924 index_of_national_number = ((index_of_rfc3996_prefix + len(_RFC3966_PREFIX)) 

2925 if (index_of_rfc3996_prefix >= 0) else 0) 

2926 national_number += number[index_of_national_number:index_of_phone_context] 

2927 else: 

2928 # Extract a possible number from the string passed in (this strips leading characters that 

2929 # could not be the start of a phone number.) 

2930 national_number = _extract_possible_number(number) 

2931 

2932 # Delete the isdn-subaddress and everything after it if it is 

2933 # present. Note extension won't appear at the same time with 

2934 # isdn-subaddress according to paragraph 5.3 of the RFC3966 spec, 

2935 index_of_isdn = national_number.find(_RFC3966_ISDN_SUBADDRESS) 

2936 if index_of_isdn > 0: 

2937 national_number = national_number[:index_of_isdn] 

2938 # If both phone context and isdn-subaddress are absent but other 

2939 # parameters are present, the parameters are left in national_number. This 

2940 # is because we are concerned about deleting content from a potential 

2941 # number string when there is no strong evidence that the number is 

2942 # actually written in RFC3966. 

2943 return national_number
