39@register_extension_dtype 

40class ArrowDtype(StorageExtensionDtype): 

41 """ 

42 An ExtensionDtype for PyArrow data types. 

43 

44 .. warning:: 

45 

46 ArrowDtype is considered experimental. The implementation and 

47 parts of the API may change without warning. 

48 

49 While most ``dtype`` arguments can accept the "string" 

50 constructor, e.g. ``"int64[pyarrow]"``, ArrowDtype is useful 

51 if the data type contains parameters like ``pyarrow.timestamp``. 

52 

53 Parameters 

54 ---------- 

55 pyarrow_dtype : pa.DataType 

56 An instance of a `pyarrow.DataType <https://arrow.apache.org/docs/python/api/datatypes.html#factory-functions>`__. 

57 

58 Attributes 

59 ---------- 

60 pyarrow_dtype 

61 

62 Methods 

63 ------- 

64 None 

65 

66 Returns 

67 ------- 

68 ArrowDtype 

69 

70 Examples 

71 -------- 

72 >>> import pyarrow as pa 

73 >>> pd.ArrowDtype(pa.int64()) 

74 int64[pyarrow] 

75 

76 Types with parameters must be constructed with ArrowDtype. 

77 

78 >>> pd.ArrowDtype(pa.timestamp("s", tz="America/New_York")) 

79 timestamp[s, tz=America/New_York][pyarrow] 

80 >>> pd.ArrowDtype(pa.list_(pa.int64())) 

81 list<item: int64>[pyarrow] 

82 """ # noqa: E501

195 @classmethod 

196 def construct_from_string(cls, string: str) -> ArrowDtype: 

197 """ 

198 Construct this type from a string. 

199 

200 Parameters 

201 ---------- 

202 string : str 

203 string should follow the format f"{pyarrow_type}[pyarrow]" 

204 e.g. int64[pyarrow] 

205 """ 

206 if not isinstance(string, str): 

207 raise TypeError( 

208 f"'construct_from_string' expects a string, got {type(string)}" 

209 ) 

210 if not string.endswith("[pyarrow]"): 

211 raise TypeError(f"'{string}' must end with '[pyarrow]'") 

212 if string == "string[pyarrow]": 

213 # Ensure Registry.find skips ArrowDtype to use StringDtype instead 

214 raise TypeError("string[pyarrow] should be constructed by StringDtype") 

215 

216 base_type = string[:-9] # get rid of "[pyarrow]" 

217 try: 

218 pa_dtype = pa.type_for_alias(base_type) 

219 except ValueError as err: 

220 has_parameters = re.search(r"[\[\(].*[\]\)]", base_type) 

221 if has_parameters: 

222 # Fallback to try common temporal types 

223 try: 

224 return cls._parse_temporal_dtype_string(base_type) 

225 except (NotImplementedError, ValueError): 

226 # Fall through to raise with nice exception message below 

227 pass 

228 

229 raise NotImplementedError( 

230 "Passing pyarrow type specific parameters " 

231 f"({has_parameters.group()}) in the string is not supported. " 

232 "Please construct an ArrowDtype object with a pyarrow_dtype " 

233 "instance with specific parameters." 

234 ) from err 

235 raise TypeError(f"'{base_type}' is not a valid pyarrow data type.") from err 

236 return cls(pa_dtype)

239 @classmethod 

240 def _parse_temporal_dtype_string(cls, string: str) -> ArrowDtype: 

241 """ 

242 Construct a temporal ArrowDtype from string. 

243 """ 

244 # we assume 

245 # 1) "[pyarrow]" has already been stripped from the end of our string. 

246 # 2) we know "[" is present 

247 head, tail = string.split("[", 1) 

248 

249 if not tail.endswith("]"): 

250 raise ValueError 

251 tail = tail[:-1] 

252 

253 if head == "timestamp": 

254 assert "," in tail # otherwise type_for_alias should work 

255 unit, tz = tail.split(",", 1) 

256 unit = unit.strip() 

257 tz = tz.strip() 

258 if tz.startswith("tz="): 

259 tz = tz[3:] 

260 

261 pa_type = pa.timestamp(unit, tz=tz) 

262 dtype = cls(pa_type) 

263 return dtype 

264 

265 raise NotImplementedError(string)
