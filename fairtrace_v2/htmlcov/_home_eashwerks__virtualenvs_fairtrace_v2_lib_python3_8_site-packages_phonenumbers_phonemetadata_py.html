25class NumberFormat(UnicodeMixin, ImmutableMixin): 

26 """Representation of way that a phone number can be formatted for output""" 

27 @mutating_method 

28 def __init__(self, 

29 pattern=None, 

30 format=None, 

31 leading_digits_pattern=None, 

32 national_prefix_formatting_rule=None, 

33 national_prefix_optional_when_formatting=None, 

34 domestic_carrier_code_formatting_rule=None): 

35 # pattern is a regex that is used to match the national (significant) 

36 # number. For example, the pattern "(20)(\d{4})(\d{4})" will match 

37 # number "2070313000", which is the national (significant) number for 

38 # Google London. Note the presence of the parentheses, which are 

39 # capturing groups what specifies the grouping of numbers. 

40 self.pattern = force_unicode(pattern) # Unicode string holding regexp 

41 

42 # format specifies how the national (significant) number matched by 

43 # pattern should be formatted. Using the same example as above, format 

44 # could contain "$1 $2 $3", meaning that the number should be 

45 # formatted as "20 7031 3000". Each $x is replaced by the numbers 

46 # captured by group x in the regex specified by pattern. 

47 self.format = force_unicode(format) # None or Unicode string 

48 

49 # This field is a regex that is used to match a certain number of 

50 # digits at the beginning of the national (significant) number. When 

51 # the match is successful, the accompanying pattern and format should 

52 # be used to format this number. For example, if 

53 # leading_digits="[1-3]|44", then all the national numbers starting 

54 # with 1, 2, 3 or 44 should be formatted using the accompanying 

55 # pattern and format. 

56 # 

57 # The first leading_digits_pattern matches up to the first three digits 

58 # of the national (significant) number; the next one matches the first 

59 # four digits, then the first five and so on, until the 

60 # leading_digits_pattern can uniquely identify one pattern and format 

61 # to be used to format the number. 

62 # 

63 # In the case when only one formatting pattern exists, no 

64 # leading_digits_pattern is needed. 

65 self.leading_digits_pattern = [] # list of Unicode strings holding regexps 

66 if leading_digits_pattern is not None: 

67 self.leading_digits_pattern = [force_unicode(p) for p in leading_digits_pattern] 

68 

69 # This field specifies how the national prefix ($NP) together with the 

70 # first group ($FG) in the national significant number should be 

71 # formatted in the NATIONAL format when a national prefix exists for a 

72 # certain country. For example, when this field contains "($NP$FG)", a 

73 # number from Beijing, China (whose $NP = 0), which would by default 

74 # be formatted without national prefix as 10 1234 5678 in NATIONAL 

75 # format, will instead be formatted as (010) 1234 5678; to format it 

76 # as (0)10 1234 5678, the field would contain "($NP)$FG". Note $FG 

77 # should always be present in this field, but $NP can be omitted. For 

78 # example, having "$FG" could indicate the number should be formatted 

79 # in NATIONAL format without the national prefix. This is commonly 

80 # used to override the rule specified for the territory in the XML 

81 # file. 

82 # 

83 # When this field is missing, a number will be formatted without 

84 # national prefix in NATIONAL format. This field does not affect how a 

85 # number is formatted in other formats, such as INTERNATIONAL. 

86 self.national_prefix_formatting_rule = force_unicode(national_prefix_formatting_rule) # None or Unicode string 

87 

88 # This field specifies whether the $NP can be omitted when formatting 

89 # a number in national format, even though it usually wouldn't be. For 

90 # example, a UK number would be formatted by our library as 020 XXXX 

91 # XXXX. If we have commonly seen this number written by people without 

92 # the leading 0, for example as (20) XXXX XXXX, this field would be 

93 # set to true. This will be inherited from the value set for the 

94 # territory in the XML file, unless a national_prefix_formatting_rule 

95 # is defined specifically for this NumberFormat. 

96 if national_prefix_optional_when_formatting is not None: 

97 self.national_prefix_optional_when_formatting = bool(national_prefix_optional_when_formatting) 

98 else: 

99 self.national_prefix_optional_when_formatting = None 

100 

101 # This field specifies how any carrier code ($CC) together with the 

102 # first group ($FG) in the national significant number should be 

103 # formatted when format_with_carrier_code is called, if carrier codes 

104 # are used for a certain country. 

105 self.domestic_carrier_code_formatting_rule = force_unicode(domestic_carrier_code_formatting_rule) # None or Unicode string 

106 

107 def merge_from(self, other): 

108 """Merge information from another NumberFormat object into this one.""" 

109 if other.pattern is not None: 

110 self.pattern = other.pattern 

111 if other.format is not None: 

112 self.format = other.format 

113 self.leading_digits_pattern.extend(other.leading_digits_pattern) 

114 if other.national_prefix_formatting_rule is not None: 

115 self.national_prefix_formatting_rule = other.national_prefix_formatting_rule 

116 if other.national_prefix_optional_when_formatting is not None: 

117 self.national_prefix_optional_when_formatting = other.national_prefix_optional_when_formatting 

118 if other.domestic_carrier_code_formatting_rule is not None: 

119 self.domestic_carrier_code_formatting_rule = other.domestic_carrier_code_formatting_rule 

120 

121 def __eq__(self, other): 

122 if not isinstance(other, NumberFormat): 

123 return False 

124 return (repr(self) == repr(other)) 

125 

126 def __ne__(self, other): 

127 return not self.__eq__(other) 

128 

129 def __repr__(self): 

130 return str(self) 

131 

132 def __unicode__(self): 

133 # Generate a string that is valid Python input for the constructor. 

134 # Note that we use rpr (variant of repr), which generates its own quotes. 

135 result = unicod("NumberFormat(pattern=%s, format=%s") % (rpr(self.pattern), rpr(self.format)) 

136 if self.leading_digits_pattern: 

137 result += (unicod(", leading_digits_pattern=[%s]") % 

138 unicod(", ").join([rpr(ld) for ld in self.leading_digits_pattern])) 

139 if self.national_prefix_formatting_rule is not None: 

140 result += unicod(", national_prefix_formatting_rule=%s") % rpr(self.national_prefix_formatting_rule) 

141 if self.national_prefix_optional_when_formatting is not None: 

142 result += unicod(", national_prefix_optional_when_formatting=%s") % str(self.national_prefix_optional_when_formatting) 

143 if self.domestic_carrier_code_formatting_rule is not None: 

144 result += unicod(", domestic_carrier_code_formatting_rule=%s") % rpr(self.domestic_carrier_code_formatting_rule) 

145 result += unicod(")") 

146 return result

149class PhoneNumberDesc(UnicodeMixin, ImmutableMixin): 

150 """Class representing the description of a set of phone numbers.""" 

151 @mutating_method 

152 def __init__(self, 

153 national_number_pattern=None,

156 possible_length_local_only=None): 

157 # The national_number_pattern is the pattern that a valid national 

158 # significant number would match. This specifies information such as 

159 # its total length and leading digits. 

160 self.national_number_pattern = force_unicode(national_number_pattern) # None or Unicode string holding regexp 

161 

162 # An example national significant number for the specific type. It 

163 # should not contain any formatting information. 

164 self.example_number = force_unicode(example_number) # None or Unicode string 

165 

166 # These represent the lengths a phone number from this region can be. They 

167 # will be sorted from smallest to biggest. Note that these lengths are for 

168 # the full number, without country calling code or national prefix. For 

169 # example, for the Swiss number +41789270000, in local format 0789270000, 

170 # this would be 9. 

171 # This could be used to highlight tokens in a text that may be a phone 

172 # number, or to quickly prune numbers that could not possibly be a phone 

173 # number for this locale. 

174 if possible_length is None: 

175 possible_length = () 

176 self.possible_length = possible_length # sequence of int 

177 

178 # These represent the lengths that only local phone numbers (without an area 

179 # code) from this region can be. They will be sorted from smallest to 

180 # biggest. For example, since the American number 456-1234 may be locally 

181 # diallable, although not diallable from outside the area, 7 could be a 

182 # possible value. 

183 # This could be used to highlight tokens in a text that may be a phone 

184 # number. 

185 # To our knowledge, area codes are usually only relevant for some fixed-line 

186 # and mobile numbers, so this field should only be set for those types of 

187 # numbers (and the general description) - however there are exceptions for 

188 # NANPA countries. 

189 if possible_length_local_only is None: 

190 possible_length_local_only = () 

191 self.possible_length_local_only = possible_length_local_only # sequence of int

252 # If a region code is a key in this dict, metadata for that region is available.

254 # - a function which loads the region's metadata 

255 # - None, to indicate that the metadata is already loaded 

256 _region_available = {} # ISO 3166-1 alpha 2 => function or None 

257 # Likewise for short number metadata.

265 # the PhoneMetadata for that country calling code. Examples of the country 

266 # calling codes include 800 (International Toll Free Service) and 808 

267 # (International Shared Cost Service).
