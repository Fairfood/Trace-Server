79class Signature(dict): 

80 """Task Signature. 

81 

82 Class that wraps the arguments and execution options 

83 for a single task invocation. 

84 

85 Used as the parts in a :class:`group` and other constructs, 

86 or to pass tasks around as callbacks while being compatible 

87 with serializers with a strict type subset. 

88 

89 Signatures can also be created from tasks: 

90 

91 - Using the ``.signature()`` method that has the same signature 

92 as ``Task.apply_async``: 

93 

94 .. code-block:: pycon 

95 

96 >>> add.signature(args=(1,), kwargs={'kw': 2}, options={}) 

97 

98 - or the ``.s()`` shortcut that works for star arguments: 

99 

100 .. code-block:: pycon 

101 

102 >>> add.s(1, kw=2) 

103 

104 - the ``.s()`` shortcut does not allow you to specify execution options 

105 but there's a chaning `.set` method that returns the signature: 

106 

107 .. code-block:: pycon 

108 

109 >>> add.s(2, 2).set(countdown=10).set(expires=30).delay() 

110 

111 Note: 

112 You should use :func:`~celery.signature` to create new signatures. 

113 The ``Signature`` class is the type returned by that function and 

114 should be used for ``isinstance`` checks for signatures. 

115 

116 See Also: 

117 :ref:`guide-canvas` for the complete guide. 

118 

119 Arguments: 

120 task (Union[Type[celery.app.task.Task], str]): Either a task 

121 class/instance, or the name of a task. 

122 args (Tuple): Positional arguments to apply. 

123 kwargs (Dict): Keyword arguments to apply. 

124 options (Dict): Additional options to :meth:`Task.apply_async`. 

125 

126 Note: 

127 If the first argument is a :class:`dict`, the other 

128 arguments will be ignored and the values in the dict will be used 

129 instead:: 

130 

131 >>> s = signature('tasks.add', args=(2, 2)) 

132 >>> signature(s) 

133 {'task': 'tasks.add', args=(2, 2), kwargs={}, options={}} 

134 """

530def _prepare_chain_from_options(options, tasks, use_link): 

531 # When we publish groups we reuse the same options dictionary for all of 

532 # the tasks in the group. See: 

533 # https://github.com/celery/celery/blob/fb37cb0b8/celery/canvas.py#L1022. 

534 # Issue #5354 reported that the following type of canvases 

535 # causes a Celery worker to hang: 

536 # group( 

537 # add.s(1, 1), 

538 # add.s(1, 1) 

539 # ) | tsum.s() | add.s(1) | group(add.s(1), add.s(1)) 

540 # The resolution of #5354 in PR #5681 was to only set the `chain` key 

541 # in the options dictionary if it is not present. 

542 # Otherwise we extend the existing list of tasks in the chain with the new 

543 # tasks: options['chain'].extend(chain_). 

544 # Before PR #5681 we overrode the `chain` key in each iteration 

545 # of the loop which applies all the tasks in the group: 

546 # options['chain'] = tasks if not use_link else None 

547 # This caused Celery to execute chains correctly in most cases since 

548 # in each iteration the `chain` key would reset itself to a new value 

549 # and the side effect of mutating the key did not propagate 

550 # to the next task in the group. 

551 # Since we now mutated the `chain` key, a *list* which is passed 

552 # by *reference*, the next task in the group will extend the list 

553 # of tasks in the chain instead of setting a new one from the chain_ 

554 # variable above. 

555 # This causes Celery to execute a chain, even though there might not be 

556 # one to begin with. Alternatively, it causes Celery to execute more tasks 

557 # that were previously present in the previous task in the group. 

558 # The solution is to be careful and never mutate the options dictionary 

559 # to begin with. 

560 # Here is an example of a canvas which triggers this issue: 

561 # add.s(5, 6) | group((add.s(1) | add.s(2), add.s(3))). 

562 # The expected result is [14, 14]. However, when we extend the `chain` 

563 # key the `add.s(3)` task erroneously has `add.s(2)` in its chain since 

564 # it was previously applied to `add.s(1)`. 

565 # Without being careful not to mutate the options dictionary, the result 

566 # in this case is [16, 14]. 

567 # To avoid deep-copying the entire options dictionary every single time we 

568 # run a chain we use a ChainMap and ensure that we never mutate 

569 # the original `chain` key, hence we use list_a + list_b to create a new 

570 # list. 

571 if use_link: 

572 return ChainMap({'chain': None}, options) 

573 elif 'chain' not in options: 

574 return ChainMap({'chain': tasks}, options) 

575 elif tasks is not None: 

576 # chain option may already be set, resulting in 

577 # "multiple values for keyword argument 'chain'" error. 

578 # Issue #3379. 

579 # If a chain already exists, we need to extend it with the next 

580 # tasks in the chain. 

581 # Issue #5354. 

582 # WARNING: Be careful not to mutate `options['chain']`. 

583 return ChainMap({'chain': options['chain'] + tasks}, 

584 options)

682 def prepare_steps(self, args, kwargs, tasks, 

683 root_id=None, parent_id=None, link_error=None, app=None, 

684 last_task_id=None, group_id=None, chord_body=None, 

685 clone=True, from_dict=Signature.from_dict): 

686 app = app or self.app 

687 # use chain message field for protocol 2 and later. 

688 # this avoids pickle blowing the stack on the recursion 

689 # required by linking task together in a tree structure. 

690 # (why is pickle using recursion? or better yet why cannot python 

691 # do tail call optimization making recursion actually useful?) 

692 use_link = self._use_link 

693 if use_link is None and app.conf.task_protocol == 1: 

694 use_link = True 

695 steps = deque(tasks) 

696 

697 steps_pop = steps.pop 

698 steps_extend = steps.extend 

699 

700 prev_task = None 

701 prev_res = None 

702 tasks, results = [], [] 

703 i = 0 

704 # NOTE: We are doing this in reverse order. 

705 # The result is a list of tasks in reverse order, that is 

706 # passed as the ``chain`` message field. 

707 # As it's reversed the worker can just do ``chain.pop()`` to 

708 # get the next task in the chain. 

709 while steps: 

710 task = steps_pop() 

711 is_first_task, is_last_task = not steps, not i 

712 

713 if not isinstance(task, abstract.CallableSignature): 

714 task = from_dict(task, app=app) 

715 if isinstance(task, group): 

716 task = maybe_unroll_group(task) 

717 

718 # first task gets partial args from chain 

719 if clone: 

720 if is_first_task: 

721 task = task.clone(args, kwargs) 

722 else: 

723 task = task.clone() 

724 elif is_first_task: 

725 task.args = tuple(args) + tuple(task.args) 

726 

727 if isinstance(task, _chain): 

728 # splice the chain 

729 steps_extend(task.tasks) 

730 continue 

731 

732 if isinstance(task, group) and prev_task: 

733 # automatically upgrade group(...) | s to chord(group, s) 

734 # for chords we freeze by pretending it's a normal 

735 # signature instead of a group. 

736 tasks.pop() 

737 results.pop() 

738 try: 

739 task = chord( 

740 task, body=prev_task, 

741 task_id=prev_res.task_id, root_id=root_id, app=app, 

742 ) 

743 except AttributeError: 

744 # A GroupResult does not have a task_id since it consists 

745 # of multiple tasks. 

746 # We therefore, have to construct the chord without it. 

747 # Issues #5467, #3585. 

748 task = chord( 

749 task, body=prev_task, 

750 root_id=root_id, app=app, 

751 ) 

752 

753 if is_last_task: 

754 # chain(task_id=id) means task id is set for the last task 

755 # in the chain. If the chord is part of a chord/group 

756 # then that chord/group must synchronize based on the 

757 # last task in the chain, so we only set the group_id and 

758 # chord callback for the last task. 

759 res = task.freeze( 

760 last_task_id, 

761 root_id=root_id, group_id=group_id, chord=chord_body, 

762 ) 

763 else: 

764 res = task.freeze(root_id=root_id) 

765 

766 i += 1 

767 

768 if prev_task: 

769 if use_link: 

770 # link previous task to this task. 

771 task.link(prev_task) 

772 

773 if prev_res and not prev_res.parent: 

774 prev_res.parent = res 

775 

776 if link_error: 

777 for errback in maybe_list(link_error): 

778 task.link_error(errback) 

779 

780 tasks.append(task) 

781 results.append(res) 

782 

783 prev_task, prev_res = task, res 

784 if isinstance(task, chord): 

785 app.backend.ensure_chords_allowed() 

786 # If the task is a chord, and the body is a chain 

787 # the chain has already been prepared, and res is 

788 # set to the last task in the callback chain. 

789 

790 # We need to change that so that it points to the 

791 # group result object. 

792 node = res 

793 while node.parent: 

794 node = node.parent 

795 prev_res = node 

796 return tasks, results

827class chain(_chain): 

828 """Chain tasks together. 

829 

830 Each tasks follows one another, 

831 by being applied as a callback of the previous task. 

832 

833 Note: 

834 If called with only one argument, then that argument must 

835 be an iterable of tasks to chain: this allows us 

836 to use generator expressions. 

837 

838 Example: 

839 This is effectively :math:`((2 + 2) + 4)`: 

840 

841 .. code-block:: pycon 

842 

843 >>> res = chain(add.s(2, 2), add.s(4))() 

844 >>> res.get() 

845 8 

846 

847 Calling a chain will return the result of the last task in the chain. 

848 You can get to the other tasks by following the ``result.parent``'s: 

849 

850 .. code-block:: pycon 

851 

852 >>> res.parent.get() 

853 4 

854 

855 Using a generator expression: 

856 

857 .. code-block:: pycon 

858 

859 >>> lazy_chain = chain(add.s(i) for i in range(10)) 

860 >>> res = lazy_chain(3) 

861 

862 Arguments: 

863 *tasks (Signature): List of task signatures to chain. 

864 If only one argument is passed and that argument is 

865 an iterable, then that'll be used as the list of signatures 

866 to chain instead. This means that you can use a generator 

867 expression. 

868 

869 Returns: 

870 ~celery.chain: A lazy signature that can be called to apply the first 

871 task in the chain. When that task succeeds the next task in the 

872 chain is applied, and so on. 

873 """ 

874 

875 # could be function, but must be able to reference as :class:`chain`. 

876 def __new__(cls, *tasks, **kwargs): 

877 # This forces `chain(X, Y, Z)` to work the same way as `X | Y | Z` 

878 if not kwargs and tasks: 

879 if len(tasks) != 1 or is_list(tasks[0]): 

880 tasks = tasks[0] if len(tasks) == 1 else tasks 

881 return reduce(operator.or_, tasks) 

882 return super(chain, cls).__new__(cls, *tasks, **kwargs)

1251class chord(Signature): 

1252 r"""Barrier synchronization primitive. 

1253 

1254 A chord consists of a header and a body. 

1255 

1256 The header is a group of tasks that must complete before the callback is 

1257 called. A chord is essentially a callback for a group of tasks. 

1258 

1259 The body is applied with the return values of all the header 

1260 tasks as a list. 

1261 

1262 Example: 

1263 

1264 The chord: 

1265 

1266 .. code-block:: pycon 

1267 

1268 >>> res = chord([add.s(2, 2), add.s(4, 4)])(sum_task.s()) 

1269 

1270 is effectively :math:`\Sigma ((2 + 2) + (4 + 4))`: 

1271 

1272 .. code-block:: pycon 

1273 

1274 >>> res.get() 

1275 12 

1276 """ 

1277 

1278 @classmethod 

1279 def from_dict(cls, d, app=None): 

1280 options = d.copy() 

1281 args, options['kwargs'] = cls._unpack_args(**options['kwargs']) 

1282 return _upgrade(d, cls(*args, app=app, **options))

1284 @staticmethod 

1285 def _unpack_args(header=None, body=None, **kwargs):
