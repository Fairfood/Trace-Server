1""" 

2This module is an implementation of `section 3.4`_ of RFC 5849. 

3 

4**Usage** 

5 

6Steps for signing a request: 

7 

81. Collect parameters from the request using ``collect_parameters``. 

92. Normalize those parameters using ``normalize_parameters``. 

103. Create the *base string URI* using ``base_string_uri``. 

114. Create the *signature base string* from the above three components 

12 using ``signature_base_string``. 

135. Pass the *signature base string* and the client credentials to one of the 

14 sign-with-client functions. The HMAC-based signing functions needs 

15 client credentials with secrets. The RSA-based signing functions needs 

16 client credentials with an RSA private key. 

17 

18To verify a request, pass the request and credentials to one of the verify

23

29

54def signature_base_string( 

55 http_method: str, 

56 base_str_uri: str, 

57 normalized_encoded_request_parameters: str) -> str: 

58 """ 

59 Construct the signature base string. 

60 

61 The *signature base string* is the value that is calculated and signed by 

62 the client. It is also independently calculated by the server to verify 

63 the signature, and therefore must produce the exact same value at both 

64 ends or the signature won't verify. 

65 

66 The rules for calculating the *signature base string* are defined in 

67 section 3.4.1.1`_ of RFC 5849. 

68 

69 .. _`section 3.4.1.1`: https://tools.ietf.org/html/rfc5849#section-3.4.1.1 

70 """ 

71 

72 # The signature base string is constructed by concatenating together, 

73 # in order, the following HTTP request elements: 

74 

75 # 1. The HTTP request method in uppercase. For example: "HEAD", 

76 # "GET", "POST", etc. If the request uses a custom HTTP method, it 

77 # MUST be encoded (`Section 3.6`_). 

78 # 

79 # .. _`Section 3.6`: https://tools.ietf.org/html/rfc5849#section-3.6 

80 base_string = utils.escape(http_method.upper()) 

81 

82 # 2. An "&" character (ASCII code 38). 

83 base_string += '&' 

84 

85 # 3. The base string URI from `Section 3.4.1.2`_, after being encoded 

86 # (`Section 3.6`_). 

87 # 

88 # .. _`Section 3.4.1.2`: https://tools.ietf.org/html/rfc5849#section-3.4.1.2 

89 # .. _`Section 3.6`: https://tools.ietf.org/html/rfc5849#section-3.6 

90 base_string += utils.escape(base_str_uri) 

91 

92 # 4. An "&" character (ASCII code 38). 

93 base_string += '&' 

94 

95 # 5. The request parameters as normalized in `Section 3.4.1.3.2`_, after 

96 # being encoded (`Section 3.6`). 

97 # 

98 # .. _`Sec 3.4.1.3.2`: https://tools.ietf.org/html/rfc5849#section-3.4.1.3.2 

99 # .. _`Section 3.6`: https://tools.ietf.org/html/rfc5849#section-3.6 

100 base_string += utils.escape(normalized_encoded_request_parameters) 

101 

102 return base_string

105def base_string_uri(uri: str, host: str = None) -> str: 

106 """ 

107 Calculates the _base string URI_. 

108 

109 The *base string URI* is one of the components that make up the 

110 *signature base string*. 

111 

112 The ``host`` is optional. If provided, it is used to override any host and 

113 port values in the ``uri``. The value for ``host`` is usually extracted from 

114 the "Host" request header from the HTTP request. Its value may be just the 

115 hostname, or the hostname followed by a colon and a TCP/IP port number 

116 (hostname:port). If a value for the``host`` is provided but it does not 

117 contain a port number, the default port number is used (i.e. if the ``uri`` 

118 contained a port number, it will be discarded). 

119 

120 The rules for calculating the *base string URI* are defined in 

121 section 3.4.1.2`_ of RFC 5849. 

122 

123 .. _`section 3.4.1.2`: https://tools.ietf.org/html/rfc5849#section-3.4.1.2 

124 

125 :param uri: URI 

126 :param host: hostname with optional port number, separated by a colon 

127 :return: base string URI 

128 """ 

129 

130 if not isinstance(uri, str): 

131 raise ValueError('uri must be a string.') 

132 

133 # FIXME: urlparse does not support unicode 

134 scheme, netloc, path, params, query, fragment = urlparse.urlparse(uri) 

135 

136 # The scheme, authority, and path of the request resource URI `RFC3986` 

137 # are included by constructing an "http" or "https" URI representing 

138 # the request resource (without the query or fragment) as follows: 

139 # 

140 # .. _`RFC3986`: https://tools.ietf.org/html/rfc3986 

141 

142 if not scheme: 

143 raise ValueError('missing scheme') 

144 

145 # Per `RFC 2616 section 5.1.2`_: 

146 # 

147 # Note that the absolute path cannot be empty; if none is present in 

148 # the original URI, it MUST be given as "/" (the server root). 

149 # 

150 # .. _`RFC 2616 5.1.2`: https://tools.ietf.org/html/rfc2616#section-5.1.2 

151 if not path: 

152 path = '/' 

153 

154 # 1. The scheme and host MUST be in lowercase. 

155 scheme = scheme.lower() 

156 netloc = netloc.lower() 

157 # Note: if ``host`` is used, it will be converted to lowercase below 

158 

159 # 2. The host and port values MUST match the content of the HTTP 

160 # request "Host" header field. 

161 if host is not None: 

162 netloc = host.lower() # override value in uri with provided host 

163 

164 # 3. The port MUST be included if it is not the default port for the 

165 # scheme, and MUST be excluded if it is the default. Specifically, 

166 # the port MUST be excluded when making an HTTP request `RFC2616`_ 

167 # to port 80 or when making an HTTPS request `RFC2818`_ to port 443. 

168 # All other non-default port numbers MUST be included. 

169 # 

170 # .. _`RFC2616`: https://tools.ietf.org/html/rfc2616 

171 # .. _`RFC2818`: https://tools.ietf.org/html/rfc2818 

172 

173 if ':' in netloc: 

174 # Contains a colon ":", so try to parse as "host:port" 

175 

176 hostname, port_str = netloc.split(':', 1) 

177 

178 if len(hostname) == 0: 

179 raise ValueError('missing host') # error: netloc was ":port" or ":" 

180 

181 if len(port_str) == 0: 

182 netloc = hostname # was "host:", so just use the host part 

183 else: 

184 try: 

185 port_num = int(port_str) # try to parse into an integer number 

186 except ValueError: 

187 raise ValueError('port is not an integer') 

188 

189 if port_num <= 0 or 65535 < port_num: 

190 raise ValueError('port out of range') # 16-bit unsigned ints 

191 if (scheme, port_num) in (('http', 80), ('https', 443)): 

192 netloc = hostname # default port for scheme: exclude port num 

193 else: 

194 netloc = hostname + ':' + str(port_num) # use hostname:port 

195 else: 

196 # Does not contain a colon, so entire value must be the hostname 

197 

198 if len(netloc) == 0: 

199 raise ValueError('missing host') # error: netloc was empty string 

200 

201 v = urlparse.urlunparse((scheme, netloc, path, params, '', '')) 

202 

203 # RFC 5849 does not specify which characters are encoded in the 

204 # "base string URI", nor how they are encoded - which is very bad, since 

205 # the signatures won't match if there are any differences. Fortunately, 

206 # most URIs only use characters that are clearly not encoded (e.g. digits 

207 # and A-Z, a-z), so have avoided any differences between implementations. 

208 # 

209 # The example from its section 3.4.1.2 illustrates that spaces in 

210 # the path are percent encoded. But it provides no guidance as to what other 

211 # characters (if any) must be encoded (nor how); nor if characters in the 

212 # other components are to be encoded or not. 

213 # 

214 # This implementation **assumes** that **only** the space is percent-encoded 

215 # and it is done to the entire value (not just to spaces in the path). 

216 # 

217 # This code may need to be changed if it is discovered that other characters 

218 # are expected to be encoded. 

219 # 

220 # Note: the "base string URI" returned by this function will be encoded 

221 # again before being concatenated into the "signature base string". So any 

222 # spaces in the URI will actually appear in the "signature base string" 

223 # as "%2520" (the "%20" further encoded according to section 3.6). 

224 

225 return v.replace(' ', '%20')

228def collect_parameters(uri_query='', body=None, headers=None, 

229 exclude_oauth_signature=True, with_realm=False): 

230 """ 

231 Gather the request parameters from all the parameter sources. 

232 

233 This function is used to extract all the parameters, which are then passed 

234 to ``normalize_parameters`` to produce one of the components that make up 

235 the *signature base string*. 

236 

237 Parameters starting with `oauth_` will be unescaped. 

238 

239 Body parameters must be supplied as a dict, a list of 2-tuples, or a 

240 form encoded query string. 

241 

242 Headers must be supplied as a dict. 

243 

244 The rules where the parameters must be sourced from are defined in 

245 `section 3.4.1.3.1`_ of RFC 5849. 

246 

247 .. _`Sec 3.4.1.3.1`: https://tools.ietf.org/html/rfc5849#section-3.4.1.3.1 

248 """ 

249 if body is None: 

250 body = [] 

251 headers = headers or {} 

252 params = [] 

253 

254 # The parameters from the following sources are collected into a single 

255 # list of name/value pairs: 

256 

257 # * The query component of the HTTP request URI as defined by 

258 # `RFC3986, Section 3.4`_. The query component is parsed into a list 

259 # of name/value pairs by treating it as an 

260 # "application/x-www-form-urlencoded" string, separating the names 

261 # and values and decoding them as defined by W3C.REC-html40-19980424 

262 # `W3C-HTML-4.0`_, Section 17.13.4. 

263 # 

264 # .. _`RFC3986, Sec 3.4`: https://tools.ietf.org/html/rfc3986#section-3.4 

265 # .. _`W3C-HTML-4.0`: https://www.w3.org/TR/1998/REC-html40-19980424/ 

266 if uri_query: 

267 params.extend(urldecode(uri_query)) 

268 

269 # * The OAuth HTTP "Authorization" header field (`Section 3.5.1`_) if 

270 # present. The header's content is parsed into a list of name/value 

271 # pairs excluding the "realm" parameter if present. The parameter 

272 # values are decoded as defined by `Section 3.5.1`_. 

273 # 

274 # .. _`Section 3.5.1`: https://tools.ietf.org/html/rfc5849#section-3.5.1 

275 if headers: 

276 headers_lower = {k.lower(): v for k, v in headers.items()} 

277 authorization_header = headers_lower.get('authorization') 

278 if authorization_header is not None: 

279 params.extend([i for i in utils.parse_authorization_header( 

280 authorization_header) if with_realm or i[0] != 'realm']) 

281 

282 # * The HTTP request entity-body, but only if all of the following 

283 # conditions are met: 

284 # * The entity-body is single-part. 

285 # 

286 # * The entity-body follows the encoding requirements of the 

287 # "application/x-www-form-urlencoded" content-type as defined by 

288 # W3C.REC-html40-19980424 `W3C-HTML-4.0`_. 

289 

290 # * The HTTP request entity-header includes the "Content-Type" 

291 # header field set to "application/x-www-form-urlencoded". 

292 # 

293 # .. _`W3C-HTML-4.0`: https://www.w3.org/TR/1998/REC-html40-19980424/ 

294 

295 # TODO: enforce header param inclusion conditions 

296 bodyparams = extract_params(body) or [] 

297 params.extend(bodyparams) 

298 

299 # ensure all oauth params are unescaped 

300 unescaped_params = [] 

301 for k, v in params: 

302 if k.startswith('oauth_'): 

303 v = utils.unescape(v) 

304 unescaped_params.append((k, v)) 

305 

306 # The "oauth_signature" parameter MUST be excluded from the signature 

307 # base string if present. 

308 if exclude_oauth_signature: 

309 unescaped_params = list(filter(lambda i: i[0] != 'oauth_signature', 

310 unescaped_params)) 

311 

312 return unescaped_params

315def normalize_parameters(params) -> str: 

316 """ 

317 Calculate the normalized request parameters. 

318 

319 The *normalized request parameters* is one of the components that make up 

320 the *signature base string*. 

321 

322 The rules for parameter normalization are defined in `section 3.4.1.3.2`_ of 

323 RFC 5849. 

324 

325 .. _`Sec 3.4.1.3.2`: https://tools.ietf.org/html/rfc5849#section-3.4.1.3.2 

326 """ 

327 

328 # The parameters collected in `Section 3.4.1.3`_ are normalized into a 

329 # single string as follows: 

330 # 

331 # .. _`Section 3.4.1.3`: https://tools.ietf.org/html/rfc5849#section-3.4.1.3 

332 

333 # 1. First, the name and value of each parameter are encoded 

334 # (`Section 3.6`_). 

335 # 

336 # .. _`Section 3.6`: https://tools.ietf.org/html/rfc5849#section-3.6 

337 key_values = [(utils.escape(k), utils.escape(v)) for k, v in params] 

338 

339 # 2. The parameters are sorted by name, using ascending byte value 

340 # ordering. If two or more parameters share the same name, they 

341 # are sorted by their value. 

342 key_values.sort() 

343 

344 # 3. The name of each parameter is concatenated to its corresponding 

345 # value using an "=" character (ASCII code 61) as a separator, even 

346 # if the value is empty. 

347 parameter_parts = ['{}={}'.format(k, v) for k, v in key_values] 

348 

349 # 4. The sorted name/value pairs are concatenated together into a 

350 # single string by using an "&" character (ASCII code 38) as 

351 # separator. 

352 return '&'.join(parameter_parts)

357def _sign_hmac(hash_algorithm_name: str, 

358 sig_base_str: str, 

359 client_secret: str, 

360 resource_owner_secret: str): 

361 """ 

362 **HMAC-SHA256** 

363 

364 The "HMAC-SHA256" signature method uses the HMAC-SHA256 signature 

365 algorithm as defined in `RFC4634`_:: 

366 

367 digest = HMAC-SHA256 (key, text) 

368 

369 Per `section 3.4.2`_ of the spec. 

370 

371 .. _`RFC4634`: https://tools.ietf.org/html/rfc4634 

372 .. _`section 3.4.2`: https://tools.ietf.org/html/rfc5849#section-3.4.2 

373 """ 

374 

375 # The HMAC-SHA256 function variables are used in following way: 

376 

377 # text is set to the value of the signature base string from 

378 # `Section 3.4.1.1`_. 

379 # 

380 # .. _`Section 3.4.1.1`: https://tools.ietf.org/html/rfc5849#section-3.4.1.1 

381 text = sig_base_str 

382 

383 # key is set to the concatenated values of: 

384 # 1. The client shared-secret, after being encoded (`Section 3.6`_). 

385 # 

386 # .. _`Section 3.6`: https://tools.ietf.org/html/rfc5849#section-3.6 

387 key = utils.escape(client_secret or '') 

388 

389 # 2. An "&" character (ASCII code 38), which MUST be included 

390 # even when either secret is empty. 

391 key += '&' 

392 

393 # 3. The token shared-secret, after being encoded (`Section 3.6`_). 

394 # 

395 # .. _`Section 3.6`: https://tools.ietf.org/html/rfc5849#section-3.6 

396 key += utils.escape(resource_owner_secret or '') 

397 

398 # Get the hashing algorithm to use 

399 

400 m = { 

401 'SHA-1': hashlib.sha1, 

402 'SHA-256': hashlib.sha256, 

403 'SHA-512': hashlib.sha512, 

404 } 

405 hash_alg = m[hash_algorithm_name] 

406 

407 # Calculate the signature 

408 

409 # FIXME: HMAC does not support unicode! 

410 key_utf8 = key.encode('utf-8') 

411 text_utf8 = text.encode('utf-8') 

412 signature = hmac.new(key_utf8, text_utf8, hash_alg) 

413 

414 # digest is used to set the value of the "oauth_signature" protocol 

415 # parameter, after the result octet string is base64-encoded 

416 # per `RFC2045, Section 6.8`. 

417 # 

418 # .. _`RFC2045, Sec 6.8`: https://tools.ietf.org/html/rfc2045#section-6.8 

419 return binascii.b2a_base64(signature.digest())[:-1].decode('utf-8')

422def _verify_hmac(hash_algorithm_name: str, 

423 request, 

424 client_secret=None, 

425 resource_owner_secret=None): 

426 """Verify a HMAC-SHA1 signature. 

427 

428 Per `section 3.4`_ of the spec. 

429 

430 .. _`section 3.4`: https://tools.ietf.org/html/rfc5849#section-3.4 

431 

432 To satisfy `RFC2616 section 5.2`_ item 1, the request argument's uri 

433 attribute MUST be an absolute URI whose netloc part identifies the 

434 origin server or gateway on which the resource resides. Any Host 

435 item of the request argument's headers dict attribute will be 

436 ignored. 

437 

438 .. _`RFC2616 section 5.2`: https://tools.ietf.org/html/rfc2616#section-5.2 

439 

440 """ 

441 norm_params = normalize_parameters(request.params) 

442 bs_uri = base_string_uri(request.uri) 

443 sig_base_str = signature_base_string(request.http_method, bs_uri, 

444 norm_params) 

445 signature = _sign_hmac(hash_algorithm_name, sig_base_str, 

446 client_secret, resource_owner_secret) 

447 match = safe_string_equals(signature, request.signature) 

448 if not match: 

449 log.debug('Verify HMAC failed: signature base string: %s', sig_base_str) 

450 return match

547def _get_jwt_rsa_algorithm(hash_algorithm_name: str): 

548 """ 

549 Obtains an RSAAlgorithm object that implements RSA with the hash algorithm. 

550 

551 This method maintains the ``_jwt_rsa`` cache.

554 """

566 'SHA-512': jwt_algorithms.hashes.SHA512, 

567 } 

568 v = jwt_algorithms.RSAAlgorithm(m[hash_algorithm_name]) 

569 

570 _jwt_rsa[hash_algorithm_name] = v # populate cache 

571 

572 return v

591 rsa_private_key: str): 

592 """ 

593 Calculate the signature for an RSA-based signature method. 

594 

595 The ``alg`` is used to calculate the digest over the signature base string. 

596 For the "RSA_SHA1" signature method, the alg must be SHA-1. While OAuth 1.0a 

597 only defines the RSA-SHA1 signature method, this function can be used for 

598 other non-standard signature methods that only differ from RSA-SHA1 by the 

599 digest algorithm. 

600 

601 Signing for the RSA-SHA1 signature method is defined in 

602 `section 3.4.3`_ of RFC 5849. 

603 

604 The RSASSA-PKCS1-v1_5 signature algorithm used defined by 

605 `RFC3447, Section 8.2`_ (also known as PKCS#1), with the `alg` as the 

606 hash function for EMSA-PKCS1-v1_5. To 

607 use this method, the client MUST have established client credentials 

608 with the server that included its RSA public key (in a manner that is 

609 beyond the scope of this specification). 

610 

611 .. _`section 3.4.3`: https://tools.ietf.org/html/rfc5849#section-3.4.3 

612 .. _`RFC3447, Section 8.2`: https://tools.ietf.org/html/rfc3447#section-8.2 

613 """ 

614 

615 # Get the implementation of RSA-hash 

616 

617 alg = _get_jwt_rsa_algorithm(hash_algorithm_name) 

618 

619 # Check private key 

620 

621 if not rsa_private_key: 

622 raise ValueError('rsa_private_key required for RSA with ' + 

623 alg.hash_alg.name + ' signature method') 

624 

625 # Convert the "signature base string" into a sequence of bytes (M) 

626 # 

627 # The signature base string, by definition, only contain printable US-ASCII 

628 # characters. So encoding it as 'ascii' will always work. It will raise a 

629 # ``UnicodeError`` if it can't encode the value, which will never happen 

630 # if the signature base string was created correctly. Therefore, using 

631 # 'ascii' encoding provides an extra level of error checking. 

632 

633 m = sig_base_str.encode('ascii') 

634 

635 # Perform signing: S = RSASSA-PKCS1-V1_5-SIGN (K, M) 

636 

637 key = _prepare_key_plus(alg, rsa_private_key) 

638 s = alg.sign(m, key) 

639 

640 # base64-encoded per RFC2045 section 6.8. 

641 # 

642 # 1. While b2a_base64 implements base64 defined by RFC 3548. As used here, 

643 # it is the same as base64 defined by RFC 2045. 

644 # 2. b2a_base64 includes a "\n" at the end of its result ([:-1] removes it) 

645 # 3. b2a_base64 produces a binary string. Use decode to produce a str. 

646 # It should only contain only printable US-ASCII characters. 

647 

648 return binascii.b2a_base64(s)[:-1].decode('ascii') 

649 

650 

651def _verify_rsa(hash_algorithm_name: str, 

652 request, 

653 rsa_public_key: str): 

654 """ 

655 Verify a base64 encoded signature for a RSA-based signature method. 

656 

657 The ``alg`` is used to calculate the digest over the signature base string. 

658 For the "RSA_SHA1" signature method, the alg must be SHA-1. While OAuth 1.0a 

659 only defines the RSA-SHA1 signature method, this function can be used for 

660 other non-standard signature methods that only differ from RSA-SHA1 by the 

661 digest algorithm. 

662 

663 Verification for the RSA-SHA1 signature method is defined in 

664 `section 3.4.3`_ of RFC 5849. 

665 

666 .. _`section 3.4.3`: https://tools.ietf.org/html/rfc5849#section-3.4.3 

667 

668 To satisfy `RFC2616 section 5.2`_ item 1, the request argument's uri 

669 attribute MUST be an absolute URI whose netloc part identifies the 

670 origin server or gateway on which the resource resides. Any Host 

671 item of the request argument's headers dict attribute will be 

672 ignored. 

673 

674 .. _`RFC2616 Sec 5.2`: https://tools.ietf.org/html/rfc2616#section-5.2 

675 """ 

676 

677 try: 

678 # Calculate the *signature base string* of the actual received request 

679 

680 norm_params = normalize_parameters(request.params) 

681 bs_uri = base_string_uri(request.uri) 

682 sig_base_str = signature_base_string( 

683 request.http_method, bs_uri, norm_params) 

684 

685 # Obtain the signature that was received in the request 

686 

687 sig = binascii.a2b_base64(request.signature.encode('ascii')) 

688 

689 # Get the implementation of RSA-with-hash algorithm to use 

690 

691 alg = _get_jwt_rsa_algorithm(hash_algorithm_name) 

692 

693 # Verify the received signature was produced by the private key 

694 # corresponding to the `rsa_public_key`, signing exact same 

695 # *signature base string*. 

696 # 

697 # RSASSA-PKCS1-V1_5-VERIFY ((n, e), M, S) 

698 

699 key = _prepare_key_plus(alg, rsa_public_key) 

700 

701 # The signature base string only contain printable US-ASCII characters. 

702 # The ``encode`` method with the default "strict" error handling will 

703 # raise a ``UnicodeError`` if it can't encode the value. So using 

704 # "ascii" will always work. 

705 

706 verify_ok = alg.verify(sig_base_str.encode('ascii'), key, sig) 

707 

708 if not verify_ok: 

709 log.debug('Verify failed: RSA with ' + alg.hash_alg.name + 

710 ': signature base string=%s' + sig_base_str) 

711 return verify_ok 

712 

713 except UnicodeError: 

714 # A properly encoded signature will only contain printable US-ASCII 

715 # characters. The ``encode`` method with the default "strict" error 

716 # handling will raise a ``UnicodeError`` if it can't decode the value. 

717 # So using "ascii" will work with all valid signatures. But an 

718 # incorrectly or maliciously produced signature could contain other 

719 # bytes. 

720 # 

721 # This implementation treats that situation as equivalent to the 

722 # signature verification having failed. 

723 # 

724 # Note: simply changing the encode to use 'utf-8' will not remove this 

725 # case, since an incorrect or malicious request can contain bytes which 

726 # are invalid as UTF-8. 

727 return False 

728

797def sign_plaintext(client_secret, resource_owner_secret): 

798 """Sign a request using plaintext. 

799 

800 Per `section 3.4.4`_ of the spec. 

801 

802 The "PLAINTEXT" method does not employ a signature algorithm. It 

803 MUST be used with a transport-layer mechanism such as TLS or SSL (or 

804 sent over a secure channel with equivalent protections). It does not 

805 utilize the signature base string or the "oauth_timestamp" and 

806 "oauth_nonce" parameters. 

807 

808 .. _`section 3.4.4`: https://tools.ietf.org/html/rfc5849#section-3.4.4 

809 

810 """ 

811 

812 # The "oauth_signature" protocol parameter is set to the concatenated 

813 # value of: 

814 

815 # 1. The client shared-secret, after being encoded (`Section 3.6`_). 

816 # 

817 # .. _`Section 3.6`: https://tools.ietf.org/html/rfc5849#section-3.6 

818 signature = utils.escape(client_secret or '') 

819 

820 # 2. An "&" character (ASCII code 38), which MUST be included even 

821 # when either secret is empty. 

822 signature += '&' 

823 

824 # 3. The token shared-secret, after being encoded (`Section 3.6`_). 

825 # 

826 # .. _`Section 3.6`: https://tools.ietf.org/html/rfc5849#section-3.6 

827 signature += utils.escape(resource_owner_secret or '') 

828 

829 return signature 

830 

831 

832def verify_plaintext(request, client_secret=None, resource_owner_secret=None): 

833 """Verify a PLAINTEXT signature. 

834 

835 Per `section 3.4`_ of the spec. 

836 

837 .. _`section 3.4`: https://tools.ietf.org/html/rfc5849#section-3.4 

838 """ 

839 signature = sign_plaintext(client_secret, resource_owner_secret) 

840 match = safe_string_equals(signature, request.signature) 

841 if not match: 

842 log.debug('Verify PLAINTEXT failed') 

843 return match
