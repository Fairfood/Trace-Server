40class ConsumerMixin(object): 

41 """Convenience mixin for implementing consumer programs. 

42 

43 It can be used outside of threads, with threads, or greenthreads 

44 (eventlet/gevent) too. 

45 

46 The basic class would need a :attr:`connection` attribute 

47 which must be a :class:`~kombu.Connection` instance, 

48 and define a :meth:`get_consumers` method that returns a list 

49 of :class:`kombu.Consumer` instances to use. 

50 Supporting multiple consumers is important so that multiple 

51 channels can be used for different QoS requirements. 

52 

53 Example: 

54 .. code-block:: python 

55 

56 class Worker(ConsumerMixin): 

57 task_queue = Queue('tasks', Exchange('tasks'), 'tasks') 

58 

59 def __init__(self, connection): 

60 self.connection = None 

61 

62 def get_consumers(self, Consumer, channel): 

63 return [Consumer(queues=[self.task_queue], 

64 callbacks=[self.on_task])] 

65 

66 def on_task(self, body, message): 

67 print('Got task: {0!r}'.format(body)) 

68 message.ack() 

69 

70 Methods: 

71 

72 * :meth:`extra_context` 

73 

74 Optional extra context manager that will be entered 

75 after the connection and consumers have been set up. 

76 

77 Takes arguments ``(connection, channel)``. 

78 

79 * :meth:`on_connection_error` 

80 

81 Handler called if the connection is lost/ or 

82 is unavailable. 

83 

84 Takes arguments ``(exc, interval)``, where interval 

85 is the time in seconds when the connection will be retried. 

86 

87 The default handler will log the exception. 

88 

89 * :meth:`on_connection_revived` 

90 

91 Handler called as soon as the connection is re-established 

92 after connection failure. 

93 

94 Takes no arguments. 

95 

96 * :meth:`on_consume_ready` 

97 

98 Handler called when the consumer is ready to accept 

99 messages. 

100 

101 Takes arguments ``(connection, channel, consumers)``. 

102 Also keyword arguments to ``consume`` are forwarded 

103 to this handler. 

104 

105 * :meth:`on_consume_end` 

106 

107 Handler called after the consumers are canceled. 

108 Takes arguments ``(connection, channel)``. 

109 

110 * :meth:`on_iteration` 

111 

112 Handler called for every iteration while draining 

113 events. 

114 

115 Takes no arguments. 

116 

117 * :meth:`on_decode_error` 

118 

119 Handler called if a consumer was unable to decode 

120 the body of a message. 

121 

122 Takes arguments ``(message, exc)`` where message is the 

123 original message object. 

124 

125 The default handler will log the error and 

126 acknowledge the message, so if you override make 

127 sure to call super, or perform these steps yourself. 

128 

129 """

168 def run(self, _tokens=1, **kwargs): 

169 restart_limit = self.restart_limit 

170 errors = (self.connection.connection_errors + 

171 self.connection.channel_errors) 

172 while not self.should_stop: 

173 try: 

174 if restart_limit.can_consume(_tokens): # pragma: no cover 

175 for _ in self.consume(limit=None, **kwargs): 

176 pass 

177 else: 

178 sleep(restart_limit.expected_time(_tokens)) 

179 except errors: 

180 warn(W_CONN_LOST, exc_info=1)

182 @contextmanager 

183 def consumer_context(self, **kwargs): 

184 with self.Consumer() as (connection, channel, consumers): 

185 with self.extra_context(connection, channel): 

186 self.on_consume_ready(connection, channel, consumers, **kwargs) 

187 yield connection, channel, consumers

189 def consume(self, limit=None, timeout=None, safety_interval=1, **kwargs): 

190 elapsed = 0 

191 with self.consumer_context(**kwargs) as (conn, channel, consumers): 

192 for i in limit and range(limit) or count(): 

193 if self.should_stop: 

194 break 

195 self.on_iteration() 

196 try: 

197 conn.drain_events(timeout=safety_interval) 

198 except socket.timeout: 

199 conn.heartbeat_check() 

200 elapsed += safety_interval 

201 if timeout and elapsed >= timeout: 

202 raise 

203 except socket.error: 

204 if not self.should_stop: 

205 raise 

206 else: 

207 yield 

208 elapsed = 0 

209 debug('consume exiting')
