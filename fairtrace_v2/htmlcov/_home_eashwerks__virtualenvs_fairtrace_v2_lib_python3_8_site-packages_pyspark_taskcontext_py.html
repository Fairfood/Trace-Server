259def _load_from_socket( 

260 port: Optional[Union[str, int]], 

261 auth_secret: str, 

262 function: int, 

263 all_gather_message: Optional[str] = None, 

264) -> List[str]: 

265 """ 

266 Load data from a given socket, this is a blocking method thus only return when the socket 

267 connection has been closed. 

268 """ 

269 (sockfile, sock) = local_connect_and_auth(port, auth_secret) 

270 

271 # The call may block forever, so no timeout 

272 sock.settimeout(None) 

273 

274 if function == BARRIER_FUNCTION: 

275 # Make a barrier() function call. 

276 write_int(function, sockfile) 

277 elif function == ALL_GATHER_FUNCTION: 

278 # Make a all_gather() function call. 

279 write_int(function, sockfile) 

280 write_with_length(cast(str, all_gather_message).encode("utf-8"), sockfile) 

281 else: 

282 raise ValueError("Unrecognized function type") 

283 sockfile.flush() 

284 

285 # Collect result. 

286 len = read_int(sockfile) 

287 res = [] 

288 for i in range(len): 

289 res.append(UTF8Deserializer().loads(sockfile)) 

290 

291 # Release resources. 

292 sockfile.close() 

293 sock.close() 

294 

295 return res

298class BarrierTaskContext(TaskContext): 

299 

300 """ 

301 A :class:`TaskContext` with extra contextual info and tooling for tasks in a barrier stage. 

302 Use :func:`BarrierTaskContext.get` to obtain the barrier context for a running barrier task. 

303 

304 .. versionadded:: 2.4.0 

305 

306 Notes 

307 ----- 

308 This API is experimental 

309 

310 Examples 

311 -------- 

312 Set a barrier, and execute it with RDD. 

313 

314 >>> from pyspark import BarrierTaskContext 

315 >>> def block_and_do_something(itr): 

316 ... taskcontext = BarrierTaskContext.get() 

317 ... # Do something. 

318 ... 

319 ... # Wait until all tasks finished. 

320 ... taskcontext.barrier() 

321 ... 

322 ... return itr 

323 ... 

324 >>> rdd = spark.sparkContext.parallelize([1]) 

325 >>> rdd.barrier().mapPartitions(block_and_do_something).collect() 

326 [1] 

327 """

335 Internal function to get or create global :class:`BarrierTaskContext`. We need to make sure

372 def barrier(self) -> None: 

373 """ 

374 Sets a global barrier and waits until all tasks in this stage hit this barrier. 

375 Similar to `MPI_Barrier` function in MPI, this function blocks until all tasks 

376 in the same stage have reached this routine. 

377 

378 .. versionadded:: 2.4.0 

379 

380 Notes 

381 ----- 

382 This API is experimental 

383 

384 In a barrier stage, each task much have the same number of `barrier()` 

385 calls, in all possible code branches. Otherwise, you may get the job hanging 

386 or a `SparkException` after timeout. 

387 """ 

388 if self._port is None or self._secret is None: 

389 raise RuntimeError( 

390 "Not supported to call barrier() before initialize BarrierTaskContext." 

391 ) 

392 else: 

393 _load_from_socket(self._port, self._secret, BARRIER_FUNCTION) 

394 

395 def allGather(self, message: str = "") -> List[str]: 

396 """ 

397 This function blocks until all tasks in the same stage have reached this routine. 

398 Each task passes in a message and returns with a list of all the messages passed in 

399 by each of those tasks. 

400 

401 .. versionadded:: 3.0.0 

402 

403 Notes 

404 ----- 

405 This API is experimental 

406 

407 In a barrier stage, each task much have the same number of `barrier()` 

408 calls, in all possible code branches. Otherwise, you may get the job hanging 

409 or a `SparkException` after timeout. 

410 """ 

411 if not isinstance(message, str): 

412 raise TypeError("Argument `message` must be of type `str`") 

413 elif self._port is None or self._secret is None: 

414 raise RuntimeError( 

415 "Not supported to call barrier() before initialize BarrierTaskContext." 

416 ) 

417 else: 

418 return _load_from_socket(self._port, self._secret, ALL_GATHER_FUNCTION, message)
